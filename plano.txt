<script>
  let capacityOverviewCache = null;
  let capacitySkillFilter = 'Alle';

  function isIgnoredDashboardActivity(activity) {
    const a = String(activity || "").toLowerCase().trim();

    return (
      !a ||
      a === "-" ||
      a.includes("pause") ||
      a.includes("break") ||
      a.includes("systemstörung") ||
      a.includes("systemstoerung") ||
      a.includes("störung") ||
      a.includes("stoerung") ||
      a.includes("meeting") ||
      a.includes("coaching") ||
      a.includes("coaching lks de") ||
      a.includes("schulung") ||
      a.includes("schulung lks") ||
      a.includes("paypal") ||
      a.includes("briefdruck") ||
      a.includes("andere tätigkeiten") ||
      a.includes("andere taetigkeiten") ||
      a.includes("trainer") ||
      a.includes("quality mng") ||
      a.includes("wissen & training") ||
      a.includes("training") ||
      a.includes("bot")
    );
  }

  function getVisibleShiftDetails(emp) {
    if (!emp || !emp.schicht || !Array.isArray(emp.schicht.details)) return [];

    return emp.schicht.details.filter(detail => {
      return !isIgnoredDashboardActivity(detail.activity);
    });
  }

  function getVisibleShiftActivities(emp) {
    if (!emp || !emp.schicht) return [];

    const fromActivities = Array.isArray(emp.schicht.activities)
      ? emp.schicht.activities
      : [];

    let visible = fromActivities.filter(activity => !isIgnoredDashboardActivity(activity));

    if (visible.length === 0) {

  function renderControl(data) {
    const listFrueh = document.getElementById('list-frueh');
    const listMittel = document.getElementById('list-mittel');
    const listSpaet = document.getElementById('list-spaet');

    if (!listFrueh || !listMittel || !listSpaet) return;

    let allActivities = new Set();

    const sorted = [...data].sort((a, b) => {
      if (activeControlSort === 'plus') return b.stundenSaldo - a.stundenSaldo;
      if (activeControlSort === 'minus') return a.stundenSaldo - b.stundenSaldo;

      if (activeControlSort === 'time') {
        const timeA = parseStartTimeMinutes(getRelevantTimeStr(a, activeActivityFilter));
        const timeB = parseStartTimeMinutes(getRelevantTimeStr(b, activeActivityFilter));

        return timeA - timeB;
      }

      return 0;
    });

    sorted.forEach(e => {
      const visibleActivities = getVisibleShiftActivities(e);

      if (visibleActivities.length > 0) {
        visibleActivities.forEach(act => allActivities.add(act));
      }
    });

    buildActivityFilters(Array.from(allActivities).sort());

    listFrueh.innerHTML = '';
    listMittel.innerHTML = '';
    listSpaet.innerHTML = '';

    let cF = 0;
    let cM = 0;
    let cS = 0;

    let sF = 0;
    let sM = 0;
    let sS = 0;

    sorted.forEach(e => {
      if (e.schicht && e.schicht.type) {
        const visibleActivities = getVisibleShiftActivities(e);

        let showItem = true;

        if (activeActivityFilter !== 'Alle') {
          if (!visibleActivities.includes(activeActivityFilter)) {
            showItem = false;
          }
        }

        if (showItem) {
          const col = getBalanceColor(e.stundenSaldo);

          let actHtml = "";

          if (visibleActivities.length > 0) {
            const actsStr = visibleActivities.join(' • ');
            actHtml = `<div class="text-[9px] text-gray-400 mt-1 truncate" title="${escapeHtml(actsStr)}"><i class="fa-solid fa-briefcase mr-1"></i>${escapeHtml(actsStr)}</div>`;
          }

          const displayTime = getRelevantTimeStr(e, activeActivityFilter);

          let timeClass = "text-lidl-blue bg-blue-50 border border-blue-100";

          if (activeActivityFilter !== 'Alle' && displayTime !== e.schicht.time) {
            timeClass = "text-white bg-lidl-blue shadow-sm border-transparent";
          }

          const item = `
            <div onclick="openShiftDetailsModal(${e.id})" class="p-3 border border-gray-100 rounded-xl bg-white hover:border-blue-300 hover:shadow-md cursor-pointer flex flex-col gap-1 text-sm transition-all group">

                  </div>
                </div>
                <div class="text-right flex flex-col items-end shrink-0">
                  <div class="font-mono text-[10px] font-bold px-2 py-0.5 rounded ${timeClass} whitespace-nowrap">${escapeHtml(displayTime)}</div>
                </div>
              </div>
              ${actHtml}
            </div>`;

          const saldoVal = parseFloat(e.stundenSaldo) || 0;

          if (e.schicht.type === "Früh") {
            listFrueh.innerHTML += item;
            cF++;
            if (saldoVal > 0) sF += saldoVal;
          } else if (e.schicht.type === "Mittel") {

            cM++;
            if (saldoVal > 0) sM += saldoVal;
          } else if (e.schicht.type === "Spät") {

  function getRelevantTimeStr(emp, filter) {
    if (!emp || !emp.schicht) return "";

    const visibleDetails = getVisibleShiftDetails(emp);

    if (filter === 'Alle' || visibleDetails.length === 0) {
      return emp.schicht.time || "";
    }

    const matches = visibleDetails.filter(d => d.activity === filter);

    if (matches.length > 0) {
      if (matches.length === 1) {
        return matches[0].time;
      }

      let minStart = null;
      let maxEnd = null;
      let minStartMins = 99999;
      let maxEndMins = -1;

      matches.forEach(m => {
        const tMatch = String(m.time || '').match(/(\d{2}:\d{2})\s*-\s*(\d{2}:\d{2})/);

        if (tMatch) {
          const sStr = tMatch[1];
          const eStr = tMatch[2];

          const sMins = parseInt(sStr.split(':')[0], 10) * 60 + parseInt(sStr.split(':')[1], 10);
          let eMins = parseInt(eStr.split(':')[0], 10) * 60 + parseInt(eStr.split(':')[1], 10);

          if (eMins <= sMins) {
            eMins += 24 * 60;
          }

          if (sMins < minStartMins) {
            minStartMins = sMins;
            minStart = sStr;
          }

          if (eMins > maxEndMins) {
            maxEndMins = eMins;
            maxEnd = eStr;
          }
        }
      });

      if (minStart && maxEnd) {
        return `${minStart}-${maxEnd}`;
      }

      return matches.map(m => m.time).join(', ');
    }

    return emp.schicht.time || "";
  }

  function parseStartTimeMinutes(timeStr) {
    if (!timeStr) return 9999;

    const match = String(timeStr).match(/(\d{2}):(\d{2})/);

    if (match) {
      return parseInt(match[1], 10) * 60 + parseInt(match[2], 10);
    }

    return 9999;
  }

  function buildActivityFilters(activities) {
    const container = document.getElementById('activity-filter-container');

    if (!container) return;

    const cleanActivities = (activities || [])
      .filter(act => act && !isIgnoredDashboardActivity(act))
      .filter((act, index, arr) => arr.indexOf(act) === index);

    container.innerHTML = '';

    const btnAll = document.createElement('button');

    btnAll.innerText = 'Alle anzeigen';
    btnAll.className = `act-filter-chip px-3 py-1 rounded-full border shadow-sm ${activeActivityFilter === 'Alle' ? 'active' : 'default'}`;
    btnAll.onclick = () => {
      activeActivityFilter = 'Alle';
      applyFilters();
    };

    container.appendChild(btnAll);

    cleanActivities.forEach(act => {
      const btn = document.createElement('button');

      btn.innerText = act;
      btn.className = `act-filter-chip px-3 py-1 rounded-full border shadow-sm ${activeActivityFilter === act ? 'active' : 'default'}`;
      btn.onclick = () => {
        activeActivityFilter = act;
        applyFilters();
      };

      container.appendChild(btn);
    });

    if (activeActivityFilter !== 'Alle' && !cleanActivities.includes(activeActivityFilter)) {
      activeActivityFilter = 'Alle';
    }
  }

  function updateRadar() {
    const list = document.getElementById('radar-list');
    const badge = document.getElementById('radar-badge');
    const time = document.getElementById('radar-time');

    if (time) {
      time.innerText = new Date().toLocaleTimeString('de-DE', { hour: '2-digit', minute: '2-digit' });
    }

    if (!list || !badge || !allData) return;

    const now = new Date();
    const nowMins = now.getHours() * 60 + now.getMinutes();

    const upcoming = [];

    allData.forEach(emp => {
      const visibleDetails = getVisibleShiftDetails(emp);

      visibleDetails.forEach(detail => {
        const match = String(detail.time || '').match(/(\d{2}):(\d{2})\s*-\s*(\d{2}):(\d{2})/);

        if (!match) return;

        const start = Number(match[1]) * 60 + Number(match[2]);
        const diff = start - nowMins;

        if (diff >= -5 && diff <= 30) {
          upcoming.push({
            id: emp.id,
            name: emp.fullName,
            time: detail.time,
            activity: detail.activity,
            diff: diff
          });
        }
      });
    });

    upcoming.sort((a, b) => a.diff - b.diff);

    badge.innerText = upcoming.length;
    badge.classList.toggle('hidden', upcoming.length === 0);

    if (upcoming.length === 0) {
      list.innerHTML = '<div class="p-5 text-center text-xs text-gray-400">Keine produktiven Wechsel in den nächsten 30 Minuten.</div>';
      return;
    }

    list.innerHTML = '';

    upcoming.forEach(item => {
      const diffText = item.diff < 0
        ? `seit ${Math.abs(item.diff)} Min.`
        : `in ${item.diff} Min.`;

      list.innerHTML += `
        <button type="button" onclick="openShiftDetailsModal(${item.id})" class="w-full text-left p-3 border-b border-gray-100 bg-white hover:bg-blue-50 transition cursor-pointer">
          <div class="flex justify-between items-center gap-2">
            <div class="font-bold text-sm text-slate-700 truncate">${escapeHtml(item.name)}</div>
            <div class="font-mono text-xs font-bold text-lidl-blue shrink-0">${escapeHtml(item.time)}</div>
          </div>
          <div class="text-xs text-gray-500 mt-1 truncate">${escapeHtml(item.activity)}</div>
          <div class="text-[10px] font-bold text-gray-400 mt-1">${escapeHtml(diffText)}</div>
        </button>
      `;
    });
  }

      areas = allAreas.filter(a => a.area === capacitySkillFilter);
    }

    const totalHours = areas.reduce((sum, area) => sum + Number(area.totalHours || 0), 0);
    const remainingHours = areas.reduce((sum, area) => sum + Number(area.remainingHours || 0), 0);
    const usedHours = areas.reduce((sum, area) => sum + Number(area.usedHours || 0), 0);

    let html = `
      <div class="space-y-5">
        <div class="grid grid-cols-1 md:grid-cols-3 gap-4">
          <div class="bg-white rounded-2xl border border-gray-100 p-5 shadow-sm">
            <div class="text-[10px] uppercase font-bold text-gray-400">Gesamtkapazität Tag</div>
            <div class="font-mono text-3xl font-bold text-slate-700 mt-1">${capacityDecimalToTime(totalHours)}</div>
          </div>
          <div class="bg-white rounded-2xl border border-gray-100 p-5 shadow-sm">
            <div class="text-[10px] uppercase font-bold text-gray-400">Bereits vergangen</div>
            <div class="font-mono text-3xl font-bold text-lidl-blue mt-1">${capacityDecimalToTime(usedHours)}</div>
          </div>
          <div class="bg-white rounded-2xl border border-gray-100 p-5 shadow-sm">
            <div class="text-[10px] uppercase font-bold text-gray-400">Restkapazität ab jetzt</div>
            <div class="font-mono text-3xl font-bold text-modern-green mt-1">${capacityDecimalToTime(remainingHours)}</div>
          </div>
        </div>

        ${renderCapacitySkillFilters(skills)}

        <div class="bg-white border border-gray-100 rounded-2xl shadow-sm overflow-hidden">
          <div class="px-5 py-4 border-b border-gray-100 bg-gradient-to-r from-blue-50 via-white to-yellow-50 flex items-center justify-between">
            <div>
              <h4 class="font-bold text-slate-700 flex items-center gap-2">
                <i class="fa-solid fa-layer-group text-lidl-blue"></i> Kapazität nach Skill
              </h4>
              <p class="text-xs text-gray-400 mt-0.5">Nur produktive Skill-Zeiträume aus den Tätigkeitsblöcken</p>
            </div>
            <div class="text-xs text-gray-400">Stand: ${escapeHtml(data.updatedAt || '-')}</div>
          </div>

          <div class="p-4 grid grid-cols-1 md:grid-cols-2 xl:grid-cols-4 gap-4">
    `;

    if (areas.length === 0) {
      html += `
        <div class="col-span-full text-center py-10 text-gray-400">
          <i class="fa-regular fa-folder-open text-3xl mb-3"></i>
          <p>Keine produktiven Skill-Zeiten gefunden.</p>
        </div>
      `;
    }

    areas.forEach(area => {
      html += `
        <div class="border border-gray-100 rounded-2xl p-4 bg-white hover:shadow-md transition">
          <div class="flex items-start justify-between gap-3 mb-4">
            <div>
              <div class="font-bold text-slate-700">${escapeHtml(area.area)}</div>
              <div class="text-[10px] text-gray-400 mt-0.5 truncate" title="${escapeHtml((area.originalSkills || []).join(', '))}">
                ${(area.originalSkills || []).length ? escapeHtml((area.originalSkills || []).join(' • ')) : 'Produktiver Skill'}
              </div>
            </div>
            <div class="w-9 h-9 rounded-xl bg-blue-50 text-lidl-blue flex items-center justify-center">
              <i class="fa-solid fa-clock"></i>
            </div>
          </div>

          <div class="grid grid-cols-3 gap-2 text-right">
            <div class="rounded-xl bg-gray-50 border border-gray-100 p-2">
              <div class="text-[9px] uppercase font-bold text-gray-400">Gesamt</div>
              <div class="font-mono font-bold text-slate-700">${capacityDecimalToTime(area.totalHours)}</div>
            </div>
            <div class="rounded-xl bg-blue-50 border border-blue-100 p-2">
              <div class="text-[9px] uppercase font-bold text-gray-400">Vergangen</div>
              <div class="font-mono font-bold text-lidl-blue">${capacityDecimalToTime(area.usedHours)}</div>
            </div>
            <div class="rounded-xl bg-emerald-50 border border-emerald-100 p-2">
              <div class="text-[9px] uppercase font-bold text-gray-400">Rest</div>
              <div class="font-mono font-bold text-modern-green">${capacityDecimalToTime(area.remainingHours)}</div>
            </div>
          </div>
        </div>
      `;
    });

    html += `
          </div>
        </div>
      </div>
    `;

    content.innerHTML = html;
  }

  function renderCapacitySkillFilters(skills) {
    const safeSkills = (skills || []).filter(s => s && String(s).trim() !== "" && String(s).toLowerCase() !== "ohne skill");

    let html = `
      <div class="bg-white border border-gray-100 rounded-xl p-3 shadow-sm">
        <div class="flex flex-wrap gap-2 items-center">
          <span class="text-[10px] uppercase font-bold text-gray-400 mr-1">
            <i class="fa-solid fa-filter text-lidl-blue mr-1"></i> Skill Filter:
          </span>
    `;

    const allClass = capacitySkillFilter === 'Alle'
      ? 'bg-lidl-blue text-white border-lidl-blue'
      : 'bg-white text-gray-500 border-gray-200 hover:bg-blue-50 hover:text-lidl-blue hover:border-blue-200';

    html += `
      <button onclick="setCapacitySkillFilter('${encodeURIComponent('Alle')}')" class="px-3 py-1 rounded-full border text-xs font-bold transition ${allClass}">
        Alle Skills
      </button>
    `;

    safeSkills.forEach(skill => {
      const active = capacitySkillFilter === skill;
      const cls = active
        ? 'bg-lidl-blue text-white border-lidl-blue'
        : 'bg-white text-gray-500 border-gray-200 hover:bg-blue-50 hover:text-lidl-blue hover:border-blue-200';

      html += `
        <button onclick="setCapacitySkillFilter('${encodeURIComponent(skill)}')" class="px-3 py-1 rounded-full border text-xs font-bold transition ${cls}">
          ${escapeHtml(skill)}
        </button>
      `;
    });

    html += `
        </div>
      </div>
    `;

    return html;
  }

  function capacityDecimalToTime(decimalVal) {
    decimalVal = Number(decimalVal || 0);

    const isNegative = decimalVal < 0;
    decimalVal = Math.abs(decimalVal);

    let hours = Math.floor(decimalVal);
    let minutes = Math.round((decimalVal - hours) * 60);

    if (minutes === 60) {
      hours += 1;
      minutes = 0;
    }

    return `${isNegative ? '-' : ''}${String(hours).padStart(2, '0')}:${String(minutes).padStart(2, '0')}`;
  }

  function openModalById(modalId, contentId) {
    const modal = document.getElementById(modalId);
    const content = document.getElementById(contentId);

    if (!modal || !content) return;

    modal.classList.remove('hidden');

    setTimeout(() => {
      modal.classList.remove('opacity-0');
      content.classList.remove('scale-95');
      content.classList.add('scale-100');
    }, 10);
  }

  function closeModalById(modalId, contentId) {
    const modal = document.getElementById(modalId);
    const content = document.getElementById(contentId);

    if (!modal || !content) return;

    modal.classList.add('opacity-0');
    content.classList.remove('scale-100');
    content.classList.add('scale-95');

    setTimeout(() => {
      modal.classList.add('hidden');
    }, 300);
  }

  function escapeHtml(value) {
    return String(value || '')
      .replace(/&/g, '&amp;')
      .replace(/</g, '&lt;')
      .replace(/>/g, '&gt;')
      .replace(/"/g, '&quot;')
      .replace(/'/g, '&#039;');
  }
