function cwsNormalizeText_(value) {
  return String(value == null ? "" : value).normalize("NFC").trim();
}

function cwsFindLastDataRow_(sheet, startRow, startColumn, columnCount) {
  const maxRows = sheet.getMaxRows();
  if (maxRows < startRow) return 0;

  const values = sheet.getRange(startRow, startColumn, maxRows - startRow + 1, columnCount).getDisplayValues();
  for (let r = values.length - 1; r >= 0; r--) {
    if (values[r].some(cell => cwsNormalizeText_(cell) !== "")) {
      return startRow + r;
    }
  }

  return 0;
}

/**
 * Button function: importToChangeWishesSorting
 * Imports from "Rohdaten Änderungswünsche Sort." to "Änderungswünsche Sortierung".
 */
function importToChangeWishesSorting() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const ui = SpreadsheetApp.getUi();
  const sourceSheet = ss.getSheetByName("Rohdaten Änderungswünsche Sort.");
  const destSheet = ss.getSheetByName("Änderungswünsche Sortierung");

  Logger.log("--- Starting importToChangeWishesSorting ---");

  if (!sourceSheet || !destSheet) {
    ui.alert("Fehler", 'Quelle "Rohdaten Änderungswünsche Sort." oder Ziel "Änderungswünsche Sortierung" wurde nicht gefunden.', ui.ButtonSet.OK);
    return;
  }

  const sourceLastRow = cwsFindLastDataRow_(sourceSheet, 8, 1, 8);
  if (sourceLastRow < 8) {
    ui.alert("Keine Daten", 'Das Blatt "Rohdaten Änderungswünsche Sort." enthält ab Zeile 8 keine importierbaren Daten.', ui.ButtonSet.OK);
    return;
  }

  try {
    const destLastRow = destSheet.getLastRow();
    if (destLastRow >= 6) {
      destSheet.getRange("A6:R" + destLastRow).clearContent();
    }

    const sourceValues = sourceSheet.getRange("A8:H" + sourceLastRow).getDisplayValues();
    const uniqueKeys = new Set();
    const processedData = [];

    for (const row of sourceValues) {
      const artikelNummer = cwsNormalizeText_(row[4]); // Source E -> destination F
      const artikel = cwsNormalizeText_(row[7]);       // Source H -> destination G
      const uniqueKey = `${artikelNummer}|||${artikel}`;

      if (!artikelNummer && !artikel) continue;
      if (uniqueKeys.has(uniqueKey)) continue;

      processedData.push([
        cwsNormalizeText_(row[0]), // A -> A
        cwsNormalizeText_(row[1]), // B -> B
        cwsNormalizeText_(row[2]), // C -> C
        cwsNormalizeText_(row[3]), // D -> D
        cwsNormalizeText_(row[5]), // F -> E
        artikelNummer,             // E -> F
        artikel                    // H -> G
      ]);

      uniqueKeys.add(uniqueKey);
    }

    if (!processedData.length) {
      ui.alert("Keine Daten", "Es wurden keine eindeutigen Datenzeilen zum Import gefunden.", ui.ButtonSet.OK);
      return;
    }

    const outputRange = destSheet.getRange(6, 1, processedData.length, 7);
    outputRange.setNumberFormat("@");
    outputRange.setValues(processedData);

    const newLastRow = 5 + processedData.length;
    const formulaTemplateRange = destSheet.getRange("H3:R3");
    const templateFormulas = formulaTemplateRange.getFormulasR1C1()[0];

    if (templateFormulas.join("").length > 0) {
      const formulaRows = [];
      for (let i = 0; i < processedData.length; i++) {
        formulaRows.push(templateFormulas);
      }

      const formulaDestinationRange = destSheet.getRange("H6:R" + newLastRow);
      formulaDestinationRange.setFormulasR1C1(formulaRows);

      SpreadsheetApp.flush();
      Utilities.sleep(3000);

      const calculatedValues = formulaDestinationRange.getDisplayValues();
      formulaDestinationRange.setValues(calculatedValues);
    } else {
      Logger.log("Keine Formeln in H3:R3 gefunden.");
    }

    const fullDataRange = destSheet.getRange("A6:R" + newLastRow);
    fullDataRange.sort({ column: 7, ascending: true });
    fullDataRange.sort([
      { column: 1, ascending: true },
      { column: 2, ascending: true },
      { column: 3, ascending: true }
    ]);

    const formatTemplateRange = destSheet.getRange("A6:R6");
    if (newLastRow > 6) {
      formatTemplateRange.copyTo(destSheet.getRange("A7:R" + newLastRow), SpreadsheetApp.CopyPasteType.PASTE_FORMAT, false);
    }

    const maxRows = destSheet.getMaxRows();
    if (maxRows > newLastRow) {
      destSheet.getRange(newLastRow + 1, 1, maxRows - newLastRow, 18).clearContent();
    }

    destSheet.getRange("A1").activate();

    if (typeof navigateOnSuccess === "function") {
      navigateOnSuccess();
    } else {
      ss.toast(`${processedData.length} Zeilen importiert.`, "Änderungswünsche Sortierung", 8);
    }

    Logger.log("--- Finished importToChangeWishesSorting ---");
  } catch (e) {
    ui.alert("Importfehler", `${e.message}\n\n${e.stack}`, ui.ButtonSet.OK);
    Logger.log(`ERROR: ${e.stack}`);
  }
}
