/**
 * ZENTRALE BERECHTIGUNG
 * Prüft differenziert nach Zugriff, Admin-Status und Dienstplaner-Status.
 */
function getUserPermissions(email) {
  if (!email) return { hasAccess: false, isAdmin: false, isDienstplaner: false };
  
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const searchEmail = email.toLowerCase().trim();
  
  let hasAccess = false;
  let isAdmin = false;
  let isDienstplaner = false; // Neuer Flag

  // --- CHECK 1: LISTE TAGESSTEUERUNG ---
  const sheetTS = ss.getSheetByName('Tagessteuerung');
  if (sheetTS && sheetTS.getLastRow() > 1) {
    const data = sheetTS.getRange(2, 1, sheetTS.getLastRow() - 1, 2).getValues();
    for (let i = 0; i < data.length; i++) {
      const rowEmail = String(data[i][0]).toLowerCase().trim();
      if (rowEmail === searchEmail) {
        hasAccess = true; // Zugriff erlaubt
        const rowRole = String(data[i][1] || "").toLowerCase().trim();
        
        // Prüfen auf generellen Admin
        if (rowRole.includes('tagessteuerung') || rowRole.includes('dienstplanung')) {
          isAdmin = true;
        }
        // Spezifische Prüfung für den neuen Tab
        if (rowRole.includes('dienstplanung')) {
          isDienstplaner = true;
        }
      }
    }
  }

  // --- CHECK 2: DATA_DB ROLLE (Fallback für Admins, falls nicht in TS Liste) ---
  if (!isAdmin) {
    const sheetDB = ss.getSheetByName('Data_DB');
    if (sheetDB && sheetDB.getLastRow() > 1) {
      const lastRow = sheetDB.getLastRow();
      const dataDB = sheetDB.getRange(2, 1, lastRow - 1, 14).getValues();

      const isDienstplanerInDB = dataDB.some(row => {
        const dbEmail = String(row[13] || "").trim().toLowerCase();
        const skill = String(row[4] || "").trim().toLowerCase();
        return dbEmail === searchEmail && skill === 'dienstplanung';
      });

      if (isDienstplanerInDB) {
        hasAccess = true;
        isAdmin = true;
        isDienstplaner = true; // Auch hier Flag setzen
      }
    }
  }
  
  return { hasAccess, isAdmin, isDienstplaner };
}

/**
 * Hauptfunktion: Daten laden.
 */
function getData() {
  try {
    const userEmail = Session.getActiveUser().getEmail().toLowerCase();
    const permissions = getUserPermissions(userEmail);

    if (!permissions.hasAccess) {
       throw new Error("⛔ Zugriff verweigert: Ihre E-Mail (" + userEmail + ") ist nicht berechtigt.");
    }

    const isAdmin = permissions.isAdmin;
    const isDienstplaner = permissions.isDienstplaner;

    const ss = SpreadsheetApp.getActiveSpreadsheet();
    let sheetDB = ss.getSheetByName('Data_DB');
    let sheetAuftrag = ss.getSheetByName('Auftrag');
    let sheetTS = ss.getSheetByName('Tagessteuerung');
    let sheetSchicht = ss.getSheetByName('Schichtmodelle');
    // NEU: Blatt für Statistik
    let sheetAbbau = ss.getSheetByName('AbbauKW');
    
    if (!sheetDB) throw new Error('Blatt "Data_DB" fehlt.');
    if (!sheetAuftrag) {
       sheetAuftrag = ss.insertSheet('Auftrag');
       sheetAuftrag.appendRow(['Name', 'Vorname', 'Datum', 'Stunden', 'Notizen', 'Erstellt von']);
    }
    if (!sheetSchicht) {
       sheetSchicht = ss.insertSheet('Schichtmodelle');
       sheetSchicht.appendRow(['Nachname', 'Vorname', 'Schichtzeit', 'Typ', 'Pause', 'Taetigkeiten']);
    }

    // CLEANUP - Veraltete Aufträge verrechnen
    processAuftragExpiry(sheetDB, sheetAuftrag);

    const lastRow = sheetDB.getLastRow();
    if (lastRow < 2) return { employees: [], stats: {}, currentWeek: 0, currentYear: 0, isAdmin: isAdmin, isDienstplaner: isDienstplaner, abbauHistory: [] };

    // Daten laden
    const data = sheetDB.getRange(2, 1, lastRow - 1, 14).getValues();
    const rawAuftraege = getAuftraege(sheetAuftrag);

    // TS Namen sammeln
    const tsNames = new Set();
    if (sheetTS && sheetTS.getLastRow() > 1) {
      const tsData = sheetTS.getRange(2, 3, sheetTS.getLastRow() - 1, 1).getValues();
      tsData.forEach(r => {
         const full = String(r[0]).trim(); 
         if(full) tsNames.add(full.split(' ')[0].toLowerCase());
      });
    }

    // SCHICHTEN (Angepasst: Dynamische Spaltenanzahl, um Abstürze bei alten Daten zu verhindern)
    const shiftMap = new Map(); 
    if (sheetSchicht.getLastRow() > 1) {
       const lastColS = sheetSchicht.getLastColumn();
       const colsToLoad = lastColS < 6 ? lastColS : 6;
       
       if (colsToLoad >= 1) {
           const sData = sheetSchicht.getRange(2, 1, sheetSchicht.getLastRow()-1, colsToLoad).getValues();
           sData.forEach(r => {
             const key = (String(r[1] || "").trim() + " " + String(r[0] || "").trim()).toLowerCase();
             let activitiesArr = [];
             if (colsToLoad >= 6 && r[5]) {
                activitiesArr = String(r[5]).split(',').map(a => a.trim()).filter(a => a);
             }
             
             let rawActivityList = [];
             if (colsToLoad >= 6 && r[5]) {
                 let rawText = String(r[5] || "");
                 let parts = rawText.split(/(?=\d{2}:\d{2}-\d{2}:\d{2})/);
                 parts.forEach(p => {
                     let match = p.match(/(\d{2}:\d{2}-\d{2}:\d{2})\s+([^\d]+)/);
                     if (match) {
                         rawActivityList.push({
                             time: match[1].trim(),
                             activity: match[2].replace(/\n/g, '').trim()
                         });
                     }
                 });
             }
             
             shiftMap.set(key, { 
                time: colsToLoad >= 3 ? String(r[2] || "") : "", 
                type: colsToLoad >= 4 ? String(r[3] || "") : "",
                pause: colsToLoad >= 5 ? String(r[4] || "") : "",
                activities: activitiesArr,
                details: rawActivityList
             });
           });
       }
    }

    // --- NEU: ABBAU HISTORIE LADEN (Matrix -> Liste) ---
    let abbauRaw = [];
    if (sheetAbbau && sheetAbbau.getLastRow() > 1) {
       const rows = sheetAbbau.getDataRange().getValues();
       const headers = rows[0]; 
       
       for (let r = 1; r < rows.length; r++) {
          const rowData = rows[r];
          const rYear = rowData[0];
          const rKw = rowData[1];
          
          for (let c = 3; c < rowData.length; c++) {
             const val = Number(rowData[c]);
             const teamName = String(headers[c]); 
             
             if (val !== 0 && teamName) {
                abbauRaw.push({ 
                  y: rYear, 
                  kw: rKw, 
                  val: val, 
                  team: teamName,
                  tma: "" 
                });
             }
          }
       }
    }
    // ----------------------------------

    const formattedData = data.map((row, index) => {
      if (!row[0]) return null;
      
      const austrittRaw = row[12];
      let isAustritt = false;
      if (austrittRaw) {
         let aDate = null;
         if (austrittRaw instanceof Date) {
            aDate = austrittRaw;
         } else if (typeof austrittRaw === 'string' && austrittRaw.trim() !== "") {
            const parts = extractDates(austrittRaw);
            if (parts.length > 0) aDate = parts[0];
         }
         if (aDate) {
            aDate.setHours(0,0,0,0);
            const today = new Date();
            today.setHours(0,0,0,0);
            if (aDate <= today) isAustritt = true;
         }
      }
      if (isAustritt) return null;

      const nachname = String(row[0] || "").trim();
      const vorname = String(row[1] || "").trim();
      const dbKey = (vorname + " " + nachname).toLowerCase();
      
      const myAuftraege = rawAuftraege.filter(a => 
        a.name.toLowerCase() === nachname.toLowerCase() && 
        a.vorname.toLowerCase() === vorname.toLowerCase()
      ).map(a => {
         let isTs = false;
         if (a.erstelltVon) {
            const creatorFirst = a.erstelltVon.split(' ')[0].toLowerCase();
            if (tsNames.has(creatorFirst)) isTs = true;
         }
         return { ...a, isTsRelevant: isTs };
      });

      // SALDO BERECHNUNG
      const rawEss = row[5]; 
      const essDecimal = timeToDecimal(rawEss);
      const plannedDecimal = myAuftraege.reduce((sum, order) => sum + (order.stunden || 0), 0);
      let saldoDecimal = essDecimal - plannedDecimal;
      saldoDecimal = Math.round(saldoDecimal * 100) / 100;
      
      const shiftInfo = shiftMap.get(dbKey) || { time: "", type: "", pause: "", activities: [], details: [] };
      const austrittDisplay = (row[12] instanceof Date) ? formatDateCompat(row[12]) : "";

      return {
        id: index + 2,
        nachname: nachname,
        vorname: vorname,
        fullName: `${vorname} ${nachname}`,
        tmaVorname: String(row[3] || "Ohne TMA"), 
        skill: String(row[4] || "-"),
        stundenEss: rawEss, 
        stundenSaldo: saldoDecimal, 
        letzteAktualisierung: row[6] ? formatDateCompat(row[6]) : "-",
        notizen: String(row[11] || ""),
        austrittDatum: austrittDisplay,
        auftraege: myAuftraege,
        schicht: shiftInfo 
      };
    }).filter(item => item !== null);

    return { 
      employees: formattedData,
      currentWeek: getWeekNumber(new Date()),
      currentYear: new Date().getFullYear(),
      isAdmin: isAdmin,
      isDienstplaner: isDienstplaner,
      abbauHistory: abbauRaw 
    };
    
  } catch (e) {
    Logger.log("Fehler in getData: " + e.toString());
    throw e;
  }
}

/**
 * Cleanup Logic - Aktualisiert auf HH:MM Berechnung
 */
function processAuftragExpiry(sheetDB, sheetAuftrag) {
  const lastRowA = sheetAuftrag.getLastRow();
  if (lastRowA < 2) return; 

  const dbDataRange = sheetDB.getDataRange();
  const dbData = dbDataRange.getValues();
  const today = new Date();
  today.setHours(0, 0, 0, 0);

  const dbMap = new Map();
  for (let i = 1; i < dbData.length; i++) {
    const key = (String(dbData[i][1]).trim() + " " + String(dbData[i][0]).trim()).toLowerCase();
    dbMap.set(key, i); 
  }

  const auftragData = sheetAuftrag.getRange(2, 1, lastRowA - 1, 6).getValues();
  let dbDirty = false;
  let auftragDirty = false;
  const validAuftraege = [];

  auftragData.forEach((row) => {
    const n = String(row[0]);
    const v = String(row[1]);
    const rawDate = row[2]; 
    const hoursVal = row[3]; 
    const notiz = row[4];
    const creator = row[5];

    let dates = [];
    if (rawDate instanceof Date) {
        dates.push(rawDate);
    } else {
        dates = extractDates(String(rawDate));
    }
    
    let isExpired = false;
    
    if (dates.length > 0) {
      const lastDate = dates[dates.length - 1];
      lastDate.setHours(0,0,0,0);
      if (lastDate < today) isExpired = true;
    }
    
    let processed = false;

    if (isExpired) {
      const dbKey = (v.trim() + " " + n.trim()).toLowerCase();
      if (dbMap.has(dbKey)) {
        const dbIdx = dbMap.get(dbKey);
        
        let currentEssStr = dbData[dbIdx][5];
        let currentEssDec = timeToDecimal(currentEssStr);
        let orderHoursDec = timeToDecimal(hoursVal);
        let newEssDec = currentEssDec - orderHoursDec;

        dbData[dbIdx][5] = decimalToTime(newEssDec); 
        dbData[dbIdx][6] = today; 
        dbDirty = true;
        processed = true;
        auftragDirty = true;
      }
    }
    
    if (!processed) {
       validAuftraege.push([n, v, rawDate, hoursVal, notiz, creator]);
    }
  });

  if (dbDirty) {
     const dataToWrite = dbData.slice(1);
     const rangeFG = sheetDB.getRange(2, 6, dataToWrite.length, 2); 
     rangeFG.setNumberFormat("@"); 
     
     const valuesFG = rangeFG.getValues();
     let rangeDirty = false;
     for(let i=0; i<dataToWrite.length; i++) {
        const newF = dataToWrite[i][5];
        const newG = dataToWrite[i][6];
        if (valuesFG[i][0] !== newF) {
           valuesFG[i][0] = newF; 
           valuesFG[i][1] = newG;
           rangeDirty = true;
        }
     }
     if (rangeDirty) rangeFG.setValues(valuesFG);
  }
  
  if (auftragDirty) {
     sheetAuftrag.getRange(2, 1, lastRowA - 1, 6).clearContent();
     if (validAuftraege.length > 0) {
        sheetAuftrag.getRange(2, 1, validAuftraege.length, 6).setValues(validAuftraege);
     }
     SpreadsheetApp.flush();
  }
}

function getAuftraege(sheet) {
  const lastRow = sheet.getLastRow();
  if (lastRow < 2) return [];
  const data = sheet.getRange(2, 1, lastRow - 1, 6).getValues();
  return data.map((row, idx) => ({
    rowId: idx + 2,
    name: String(row[0]),
    vorname: String(row[1]),
    datum: formatDateCompat(row[2]), 
    stunden: timeToDecimal(row[3]), 
    stundenDisplay: row[3], 
    notizen: String(row[4]),
    erstelltVon: String(row[5] || "")
  }));
}

function addAuftrag(data) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheetAuftrag = ss.getSheetByName('Auftrag');
  if (!sheetAuftrag) sheetAuftrag = ss.insertSheet('Auftrag');
  
  const userEmail = Session.getActiveUser().getEmail();
  const editorName = getEditorNameFromEmail(userEmail);
  const lastRow = sheetAuftrag.getLastRow() + 1;
  
  sheetAuftrag.getRange(lastRow, 4).setNumberFormat("@");
  sheetAuftrag.getRange(lastRow, 1, 1, 6).setValues([[
    data.nachname.trim(), data.vorname.trim(), data.datum, data.stunden, data.notizen, editorName
  ]]);
  
  const team = getEmployeeTeam(data.nachname, data.vorname);
  updateAbbauStats(data.datum, timeToDecimal(data.stunden), 'add', team);
  
  SpreadsheetApp.flush();
  return getData(); 
}

function deleteAuftrag(rowId) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName('Auftrag');
  
  if (sheet) {
     const r = Number(rowId);
     const val = sheet.getRange(r, 1, 1, 6).getValues()[0];
     const team = getEmployeeTeam(val[0], val[1]);
     updateAbbauStats(val[2], timeToDecimal(val[3]), 'remove', team);
     sheet.deleteRow(r);
  }
  return getData();
}

function updateEmployeeBase(formData) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName('Data_DB');
  const row = parseInt(formData.id);
  
  const essCell = sheet.getRange(row, 6);
  essCell.setNumberFormat("@"); 
  essCell.setValue(formData.stundenEss); 

  sheet.getRange(row, 7).setValue(new Date()).setNumberFormat("dd.MM.yyyy");
  sheet.getRange(row, 12).setValue(formData.notizen); 
  
  if (formData.austrittDatum) {
      const parts = formData.austrittDatum.split('-'); 
      const d = new Date(parts[0], parts[1]-1, parts[2]);
      sheet.getRange(row, 13).setValue(d).setNumberFormat("dd.MM.yyyy");
  } else {
      sheet.getRange(row, 13).setValue(""); 
  }
  
  SpreadsheetApp.flush();
  return getData();
}

function getEditorNameFromEmail(email) {
  if (!email) return "";
  try {
    const localPart = email.split('@')[0];
    const parts = localPart.split('.');
    if (parts.length > 0) {
      let firstNamePart = parts[0].replace(/_/g, ' ');
      return firstNamePart.replace(/\b\w/g, l => l.toUpperCase());
    }
    return email;
  } catch (e) { return email; }
}

function extractDates(str) {
  const regex = /(\d{1,2})\.(\d{1,2})\.(\d{4})/g;
  const matches = [];
  let match;
  while ((match = regex.exec(str)) !== null) {
    matches.push(new Date(match[3], match[2] - 1, match[1]));
  }
  return matches.sort((a, b) => a - b);
}

function formatDateCompat(dateObj) {
  if (!dateObj) return "-";
  let d = null;
  if (dateObj instanceof Date) d = dateObj;
  else if (typeof dateObj === 'string') {
    if (dateObj.match(/[A-Za-z]{3}\s[A-Za-z]{3}\s\d{2}/) || dateObj.includes("GMT")) {
          const tryD = new Date(dateObj);
          if (!isNaN(tryD.getTime())) d = tryD; else return dateObj;
    } else return dateObj; 
  }
  if (d && !isNaN(d.getTime())) return Utilities.formatDate(d, Session.getScriptTimeZone(), "dd.MM.yyyy");
  return String(dateObj);
}

function getWeekNumber(d) {
    d = new Date(Date.UTC(d.getFullYear(), d.getMonth(), d.getDate()));
    d.setUTCDate(d.getUTCDate() + 4 - (d.getUTCDay()||7));
    var yearStart = new Date(Date.UTC(d.getUTCFullYear(),0,1));
    return Math.ceil((((d - yearStart) / 86400000) + 1)/7);
}

/* =========================================================================
   HILFSFUNKTIONEN FÜR ZEIT-UMRECHNUNG (HH:MM <-> DEZIMAL)
   ========================================================================= */

function timeToDecimal(timeStr) {
  if (timeStr === null || timeStr === undefined || timeStr === "") return 0;
  if (typeof timeStr === 'number') return timeStr; 
  if (timeStr instanceof Date) {
      const h = timeStr.getHours();
      const m = timeStr.getMinutes();
      return h + (m / 60);
  }
  timeStr = String(timeStr).trim();
  let isNegative = timeStr.startsWith("-");
  if (isNegative) timeStr = timeStr.substring(1);
  const parts = timeStr.split(':');
  if (parts.length < 2) return parseFloat(timeStr) || 0; 
  const hours = parseInt(parts[0], 10);
  const minutes = parseInt(parts[1], 10);
  let decimal = hours + (minutes / 60);
  return isNegative ? -decimal : decimal;
}

function decimalToTime(decimalVal) {
  if (decimalVal === null || decimalVal === undefined) return "00:00";
  const isNegative = decimalVal < 0;
  decimalVal = Math.abs(decimalVal);
  const hours = Math.floor(decimalVal);
  const minutes = Math.round((decimalVal - hours) * 60);
  if (minutes === 60) {
      const displayHours = String(hours + 1).padStart(2, '0');
      return (isNegative ? "-" : "") + displayHours + ":00";
  }
  const hStr = String(hours).padStart(2, '0');
  const mStr = String(minutes).padStart(2, '0');
  return (isNegative ? "-" : "") + hStr + ":" + mStr;
}

function updateAbbauStats(dateStr, hoursDecimal, operation, teamName) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName('AbbauKW');
  if (!sheet) {
    sheet = ss.insertSheet('AbbauKW');
    sheet.appendRow(['Jahr', 'KW', 'Gesamt']); 
    sheet.setFrozenRows(1);
    sheet.setFrozenColumns(2);
  }

  let dateObj = (dateStr instanceof Date) ? dateStr : extractDates(String(dateStr))[0];
  if (!dateObj) return;

  const year = dateObj.getFullYear();
  const kw = getWeekNumber(dateObj);
  const team = teamName || "Sonstiges"; 

  const lastCol = sheet.getLastColumn();
  const headers = sheet.getRange(1, 1, 1, lastCol).getValues()[0];
  let colIndex = -1; 
  
  for (let c = 0; c < headers.length; c++) {
    if (String(headers[c]).toLowerCase() === String(team).toLowerCase()) {
      colIndex = c + 1;
      break;
    }
  }
  if (colIndex === -1) {
    colIndex = lastCol + 1;
    sheet.getRange(1, colIndex).setValue(team).setFontWeight("bold");
  }

  const lastRow = sheet.getLastRow();
  let rowIndex = -1;
  if (lastRow > 1) {
    const timeData = sheet.getRange(2, 1, lastRow - 1, 2).getValues();
    for (let r = 0; r < timeData.length; r++) {
      if (timeData[r][0] == year && timeData[r][1] == kw) {
        rowIndex = r + 2; 
        break;
      }
    }
  }
  if (rowIndex === -1) {
    sheet.appendRow([year, kw, 0]); 
    rowIndex = sheet.getLastRow();
  }

  const cell = sheet.getRange(rowIndex, colIndex);
  const oldVal = Number(cell.getValue()) || 0;
  let newVal = oldVal;
  if (operation === 'add') newVal += hoursDecimal;
  else newVal -= hoursDecimal;
  
  newVal = Math.round(newVal * 100) / 100;
  cell.setValue(newVal);
}

function getEmployeeTeam(nachname, vorname) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheetDB = ss.getSheetByName('Data_DB');
  if (!sheetDB) return "Unbekannt";
  const data = sheetDB.getDataRange().getValues();
  for(let i=1; i<data.length; i++) {
     if(String(data[i][0]).toLowerCase().trim() == String(nachname).toLowerCase().trim() && 
        String(data[i][1]).toLowerCase().trim() == String(vorname).toLowerCase().trim()) {
        return String(data[i][4]); 
     }
  }
  return "Unbekannt";
}

        allData = response.employees || [];
        globalStats = response.stats || {};
        
        currentWeekNum = response.currentWeek || getWeekNumber(new Date());
        currentYearNum = response.currentYear || new Date().getFullYear();
        abbauHistoryData = response.abbauHistory || [];

        isDienstplaner = response.isDienstplaner;

        else document.getElementById('nav-control').style.display = 'none';
        if(isDienstplaner) document.getElementById('nav-dienstplanung').style.display = 'flex';
        else document.getElementById('nav-dienstplanung').style.display = 'none';
        
        availableTeams = [...new Set(allData.map(d => d.skill).filter(Boolean))].sort();
        availableTmas = [...new Set(allData.map(d => d.tmaVorname).filter(Boolean))].sort();

        document.getElementById('loader').classList.add('hidden');
        
        applyFilters(); 
        
        if(typeof vacEmployeeList !== 'undefined') {
            google.script.run.withSuccessHandler(list => { 
                vacEmployeeList = list;
                const searchEl = document.getElementById('vac-search');
                if(searchEl && searchEl.value && searchEl.value.length >= 2) handleEmployeeSearch(searchEl.value);

        }

        if (currentPage === 'control' && !isAdmin) currentPage = 'dashboard';
        if (currentPage === 'dienstplanung' && !isDienstplaner) currentPage = 'dashboard';

        switchPage(currentPage);
        
        const btnRemind = document.getElementById('btn-send-reminder');
        if (btnRemind) {
            if (isAdmin || isDienstplaner) btnRemind.classList.remove('hidden');
            else btnRemind.classList.add('hidden');
        }
      }

      function switchPage(pageId) {
        currentPage = pageId;
        const navs = ['dashboard', 'directory', 'control', 'dienstplanung'];
        navs.forEach(n => {
           const btn = document.getElementById('nav-' + n);
           if (!btn) return;
           if (n === pageId) {
             btn.className = "px-6 py-2 rounded-md text-base font-bold bg-white text-lidl-blue shadow-sm transition-all flex items-center gap-2 cursor-pointer relative z-50";
             document.getElementById('page-' + n).classList.remove('hidden');
           } else {
             btn.className = "px-6 py-2 rounded-md text-base font-bold text-gray-500 hover:text-lidl-blue hover:bg-white transition-all flex items-center gap-2 cursor-pointer relative z-50";
             const pg = document.getElementById('page-' + n);
             if(pg) pg.classList.add('hidden');
           }
        });
        if (pageId === 'dienstplanung') {
            if (!dienstplanungLoaded) loadDienstplanungContent();
            else if (!dienstplanInitDone) initDienstplanView();
        } else {
            applyFilters();
        }

        if (pageId === 'control') {
            updateRadar();
            if (!radarInterval) radarInterval = setInterval(updateRadar, 60000);
        } else {
            if (radarInterval) { clearInterval(radarInterval); radarInterval = null; }
        }
      }

      function initFilters() {
        const teamC = document.getElementById('filter-teams-container');
        const tmaC = document.getElementById('filter-tma-container');
        teamC.innerHTML = ''; tmaC.innerHTML = '';
        availableTeams.forEach(t => {
           const chip = document.createElement('button');
           chip.className = "filter-chip default px-3 py-1 rounded-full";
           chip.innerText = t;
           chip.dataset.value = t;
           chip.onclick = () => toggleFilter(t, 'team', chip);
           teamC.appendChild(chip);
        });
        availableTmas.forEach(t => {
           const chip = document.createElement('button');
           chip.dataset.value = t;
           chip.onclick = () => toggleFilter(t, 'tma', chip);
           const colorClass = getTmaColorClass(t);
           chip.className = `filter-chip px-3 py-1 rounded-full ${colorClass}`;
           chip.innerText = t;
           tmaC.appendChild(chip);
        });
      }

      function getTmaColorClass(tma) {
         const emps = allData.filter(d => d.tmaVorname === tma && !d.austritt);
         if (emps.length === 0) return "bg-gray-50 border-gray-200 text-gray-400"; 
         let maxDays = 0;
         emps.forEach(e => {
            if (e.notizen && e.notizen.trim() !== "") return;
            if (e.letzteAktualisierung && e.letzteAktualisierung !== '-') {
               const parts = e.letzteAktualisierung.split('.');
               if(parts.length === 3) {
                  const d = new Date(parts[2], parts[1]-1, parts[0]);
                  const diff = (new Date() - d) / (1000 * 60 * 60 * 24);
                  if (diff > maxDays) maxDays = diff;
               }
            }
         });
         if (maxDays > 14) return "bg-red-600 text-white border-red-600 hover:opacity-90"; 
         if (maxDays > 7) return "bg-orange-100 text-orange-800 border-orange-300 hover:bg-orange-200"; 
         return "bg-emerald-50 text-emerald-700 border-emerald-200 hover:bg-emerald-100"; 
      }

      function toggleFilter(val, type, element) {
        if (element.classList.contains('disabled')) return;
        const set = type === 'team' ? selectedTeams : selectedTmas;
        if (set.has(val)) { 
           set.delete(val); 
           if (type === 'team') { element.classList.remove('active'); element.classList.add('default'); } 
           else { element.classList.remove('ring-2', 'ring-blue-700', 'ring-offset-1'); } 
        } else { 
           set.add(val); 
           if (type === 'team') { element.classList.remove('default'); element.classList.add('active'); } 
           else { element.classList.add('ring-2', 'ring-blue-700', 'ring-offset-1'); } 
        } 
        applyFilters(); 
      }

      function resetFilters() {
         selectedTeams.clear(); selectedTmas.clear(); 
         document.querySelectorAll('#filter-teams-container .filter-chip').forEach(el => { el.classList.remove('active', 'disabled'); el.classList.add('default'); }); 
         document.querySelectorAll('#filter-tma-container .filter-chip').forEach(el => { el.classList.remove('disabled', 'ring-2', 'ring-blue-700', 'ring-offset-1'); }); 
         applyFilters(); 
      }

      function applyFilters() {
        let filtered = allData.filter(d => {
          if (d.austritt) return false;
          const teamMatch = selectedTeams.size === 0 || selectedTeams.has(d.skill);
          const tmaMatch = selectedTmas.size === 0 || selectedTmas.has(d.tmaVorname);
          return teamMatch && tmaMatch;
        });
        updateTmaAvailability();
        renderStatsAndLists(filtered);
        if (dienstplanInitDone && typeof window.filterAndRenderDienstplan === 'function') {
            window.filterAndRenderDienstplan();
        }
      }
      
      function updateTmaAvailability() {
         const tmaChips = document.querySelectorAll('#filter-tma-container .filter-chip');
         if (selectedTeams.size === 0) { tmaChips.forEach(c => c.classList.remove('disabled')); return; }
         const validTmas = new Set();
         allData.forEach(d => { if (!d.austritt && selectedTeams.has(d.skill) && d.tmaVorname) validTmas.add(d.tmaVorname); });
         tmaChips.forEach(chip => {
            if (validTmas.has(chip.dataset.value)) { chip.classList.remove('disabled'); } 
            else { chip.classList.add('disabled'); if(selectedTmas.has(chip.dataset.value)) { selectedTmas.delete(chip.dataset.value); chip.classList.remove('ring-2', 'ring-blue-700', 'ring-offset-1'); } } 
         });
      }

      function renderStatsAndLists(data) {
        const currKW = currentWeekNum;
        const currY = currentYearNum;
        
        let lastKW = currKW - 1; let lastY = currY;
        if(lastKW < 1) { lastKW = 52; lastY--; }
        
        let nextKW = currKW + 1; let nextY = currY;
        if(nextKW > 52) { nextKW = 1; nextY++; }

        let sumLast = 0; let sumCurr = 0; let sumNext = 0;

        abbauHistoryData.forEach(h => {
             const teamMatch = selectedTeams.size === 0 || selectedTeams.has(h.team);
             if (teamMatch) {
                 if (h.y == lastY && h.kw == lastKW) sumLast += h.val;
                 if (h.y == currY && h.kw == currKW) sumCurr += h.val;
                 if (h.y == nextY && h.kw == nextKW) sumNext += h.val;
             }
        });

        const myKwStats = { lastKW: lastKW, lastVal: sumLast, currKW: currKW, currVal: sumCurr, nextKW: nextKW, nextVal: sumNext };

        const saldos = data.map(d => parseFloat(d.stundenSaldo) || 0);
        const critTotal = saldos.filter(h => h > 40).length;
        const critMid = saldos.filter(h => h >= 10 && h <= 40).length;
        const critLow = saldos.filter(h => h > 0 && h < 10).length;
        const minusArr = saldos.filter(h => h < 0);
        const minusSum = minusArr.reduce((a,b) => a+b, 0);
        const plusArr = saldos.filter(h => h > 0);
        const plusSum = plusArr.reduce((a,b) => a+b, 0);
        const avgPlus = plusArr.length > 0 ? plusSum / plusArr.length : 0;
        
        let totalAbbau = 0; let tsAbbau = 0;
        data.forEach(emp => { if (emp.auftraege) { emp.auftraege.forEach(o => { totalAbbau += o.stunden; if (o.isTsRelevant && isDateInWeek(o.datum, currentWeekNum, currentYearNum)) tsAbbau += o.stunden; }); } });

        let targetId = 'dash-stats-container';
        if (currentPage === 'directory') targetId = 'dir-stats-container';
        
        if (currentPage === 'dashboard' || currentPage === 'directory') {
            updateStatValues(targetId, {
                critTotal, critMid, critLow, minusSum, minusCount: minusArr.length, 
                plusSum, countPlus: plusArr.length, avgPlus, totalAbbau, tsAbbau,
                kwStats: myKwStats 
            });
        }

        if (currentPage === 'dashboard') renderDashboardLists(data);
        else if (currentPage === 'directory') renderDirectoryList(data);
        else if (currentPage === 'control') renderControl(data);
      }

      function updateStatValues(containerId, stats) {
         const container = document.getElementById(containerId);
         if (container && !container.innerHTML.trim()) container.innerHTML = document.getElementById('stats-row-template').innerHTML;
         if (!container) return;

         setText('crit-total', stats.critTotal);
         setText('crit-mid', stats.critMid);

         setText('avg-plus', decimalToTime(stats.avgPlus));
         setText('total-minus', decimalToTime(stats.minusSum));
         setText('count-minus', `(${stats.minusCount} Kol.)`);
         setText('total-plus', "+" + decimalToTime(stats.plusSum));
         setText('count-plus', `(${stats.countPlus} Kol.)`);
         setText('total-abbau', decimalToTime(stats.totalAbbau));
         
         if (stats.kwStats) {

             setText('kw-curr-num', stats.kwStats.currKW);

             setText('kw-next-num', stats.kwStats.nextKW);
             setText('kw-next-val', decimalToTime(stats.kwStats.nextVal));
         }
      }

      function renderDashboardLists(data) {
        renderTable('top15-plus-body', [...data].filter(d => d.stundenSaldo > 0).sort((a, b) => b.stundenSaldo - a.stundenSaldo).slice(0, 15));
        renderTable('top15-minus-body', [...data].filter(d => d.stundenSaldo < 0).sort((a, b) => a.stundenSaldo - b.stundenSaldo).slice(0, 15));

        const grp = (prop) => { 
            const g = {}; 
            data.forEach(d => { 
                const n = d[prop] || "Ohne"; 
                if(!g[n]) g[n] = { m: 0, p: 0, cP: 0 }; 
                const v = parseFloat(d.stundenSaldo) || 0; 
                if(v < 0) { g[n].m += v; } else if (v > 0) { g[n].p += v; g[n].cP++; }
            }); 
            return Object.keys(g).map(k => ({name: k, ...g[k]})).sort((a, b) => b.p - a.p); 
        };

        const rGrp = (id, arr) => { 
            const tb = document.getElementById(id); 
            tb.innerHTML = ''; 
            arr.forEach(x => { 
                let avgStr = "-";
                if (x.p > 0 && x.cP > 0) avgStr = decimalToTime(x.p / x.cP);

                tb.innerHTML += `
                <tr class="hover:bg-gray-50 border-b border-gray-50 last:border-0 transition-colors">
                    <td class="px-4 py-2 font-medium text-slate-700 truncate max-w-[120px] text-xs">${x.name}</td>
                    <td class="px-4 py-2 text-right text-red-600 font-mono text-xs">${x.m ? decimalToTime(x.m) : '-'}</td>
                    <td class="px-4 py-2 text-right text-modern-green font-mono font-semibold text-xs">${x.p ? '+' + decimalToTime(x.p) : '-'}</td>
                    <td class="px-4 py-2 text-right text-gray-500 font-mono text-xs bg-gray-50/50">${avgStr}</td>
                </tr>`; 
            }); 
        };

        rGrp('team-stats-body', grp('skill'));
        rGrp('tma-stats-body', grp('tmaVorname'));
      }

      function renderControl(data) {
         const listFrueh = document.getElementById('list-frueh');
         const listMittel = document.getElementById('list-mittel');
         const listSpaet = document.getElementById('list-spaet');
         
         let allActivities = new Set();
         
         const sorted = [...data].sort((a, b) => {
            if (activeControlSort === 'plus') return b.stundenSaldo - a.stundenSaldo;
            if (activeControlSort === 'minus') return a.stundenSaldo - b.stundenSaldo;
            if (activeControlSort === 'time') {
                const timeA = parseStartTimeMinutes(getRelevantTimeStr(a, activeActivityFilter));
                const timeB = parseStartTimeMinutes(getRelevantTimeStr(b, activeActivityFilter));
                return timeA - timeB;
            }
            return 0;
         });

         sorted.forEach(e => {
            if(e.schicht && e.schicht.activities && e.schicht.activities.length > 0) {
               e.schicht.activities.forEach(act => allActivities.add(act));
            }

      function getRelevantTimeStr(emp, filter) {
         if (filter === 'Alle' || !emp.schicht || !emp.schicht.details || emp.schicht.details.length === 0) {
             return emp.schicht.time;
         }
         const matches = emp.schicht.details.filter(d => d.activity === filter);
         if (matches.length > 0) {
            if (matches.length === 1) return matches[0].time;
            
            let minStart = null;
            let maxEnd = null;
            let minStartMins = 99999;
            let maxEndMins = -1;

            matches.forEach(m => {
               const tMatch = m.time.match(/(\d{2}:\d{2})\s*-\s*(\d{2}:\d{2})/);
               if (tMatch) {
                   const sStr = tMatch[1];
                   const eStr = tMatch[2];
                   
                   const sMins = parseInt(sStr.split(':')[0], 10) * 60 + parseInt(sStr.split(':')[1], 10);
                   let eMins = parseInt(eStr.split(':')[0], 10) * 60 + parseInt(eStr.split(':')[1], 10);
                   
                   if (eMins <= sMins) eMins += 24 * 60;

                   if (sMins < minStartMins) {
                       minStartMins = sMins;
                       minStart = sStr;
                   }
                   if (eMins > maxEndMins) {
                       maxEndMins = eMins;
                       maxEnd = eStr;
                   }
               }
            });

            if (minStart && maxEnd) {
               return `${minStart}-${maxEnd}`;
            }
            return matches.map(m => m.time).join(', ');
         }
         return emp.schicht.time;
      }

      function parseStartTimeMinutes(timeStr) {
         if (!timeStr) return 9999;
         const match = timeStr.match(/(\d{2}):(\d{2})/);
         if (match) return parseInt(match[1], 10) * 60 + parseInt(match[2], 10);
         return 9999;
      }

      function getTmaColorClass(tma) {

               if(parts.length === 3) {

      // HILFSFUNKTIONEN FÜR ZEIT-SORTIERUNG
      function getRelevantTimeStr(emp, filter) {
         if (filter === 'Alle' || !emp.schicht || !emp.schicht.details || emp.schicht.details.length === 0) {
             return emp.schicht.time;
         }
         const matches = emp.schicht.details.filter(d => d.activity === filter);
         if (matches.length > 0) {
            // Wenn es nur einen Block gibt, direkt zurückgeben
            if (matches.length === 1) return matches[0].time;
            
            // Bei mehreren Blöcken: Finde absolute Start- und Endzeit
            let minStart = null;
            let maxEnd = null;
            let minStartMins = 99999;
            let maxEndMins = -1;

            matches.forEach(m => {
               const tMatch = m.time.match(/(\d{2}:\d{2})\s*-\s*(\d{2}:\d{2})/);
               if (tMatch) {
                   const sStr = tMatch[1];
                   const eStr = tMatch[2];
                   
                   const sMins = parseInt(sStr.split(':')[0], 10) * 60 + parseInt(sStr.split(':')[1], 10);
                   let eMins = parseInt(eStr.split(':')[0], 10) * 60 + parseInt(eStr.split(':')[1], 10);
                   
                   // Nachtschicht-Korrektur (Ende am nächsten Tag)
                   if (eMins <= sMins) eMins += 24 * 60;

                   if (sMins < minStartMins) {
                       minStartMins = sMins;
                       minStart = sStr;
                   }
                   if (eMins > maxEndMins) {
                       maxEndMins = eMins;
                       maxEnd = eStr;
                   }
               }
            });

            if (minStart && maxEnd) {
               return `${minStart}-${maxEnd}`;
            }
            
            // Fallback, falls die Regex unerwartet fehlschlägt
            return matches.map(m => m.time).join(', ');
         }
         return emp.schicht.time;
      }

      function parseStartTimeMinutes(timeStr) {
         if (!timeStr) return 9999;
         // Greift die erste Startzeit im String ab, z.B. aus "10:15-11:00" -> 10 und 15
         const match = timeStr.match(/(\d{2}):(\d{2})/);
         if (match) return parseInt(match[1], 10) * 60 + parseInt(match[2], 10);
         return 9999;
      }

                          actLower.includes('meeting') || 
                          actLower.includes('andere tätigkeiten') || 
                          actLower.includes('andere taetigkeiten') ||
                          actLower.includes('quality mng') ||
                          actLower.includes('wissen & training') ||
                          actLower.includes('bot')) {
                          return;
                      }
                      
                      const match = det.time.match(/(\d{2}):(\d{2})/);
                      if (match) {
                          const startMins = parseInt(match[1], 10) * 60 + parseInt(match[2], 10);
                          validDetails.push({
                              activity: det.activity,
                              timeStr: det.time,
                              startMins: startMins
                          });
                      }
                  });

                  // Nach Startzeit sortieren
                  validDetails.sort((a, b) => a.startMins - b.startMins);

                  // Nächsten echten Wechsel finden
                  for (let i = 0; i < validDetails.length; i++) {
                      let det = validDetails[i];
                      
                      // Ist in der Zukunft? (max 2 Stunden = 120 Min entfernt)
                      if (det.startMins > currentMins && det.startMins <= currentMins + 120) {
                          
                          // WICHTIG: Es MUSS eine vorherige operative Tätigkeit geben (i > 0),
                          // um als echter "Wechsel" zu gelten. Startet jemand mit einer 
                          // gefilterten Tätigkeit (z.B. "Andere Tätigkeiten"), ist der erste operative Skill KEIN Wechsel.
                          if (i > 0) {
                              let prevAct = validDetails[i-1].activity;
                              
                              // Leerzeichen und Groß-/Kleinschreibung bereinigen
                              let cleanCurr = det.activity.trim().toLowerCase();
                              let cleanPrev = prevAct.trim().toLowerCase();
                              
                              if (cleanCurr !== cleanPrev) {
                                  upcoming.push({
                                      empId: emp.id,
                                      empName: emp.fullName,
                                      activity: det.activity,
                                      timeStr: det.timeStr,
                                      startMins: det.startMins,
                                      diffMins: det.startMins - currentMins
                                  });
                                  break; // Nur den allerersten anstehenden Wechsel der Person pushen!
                              }
                          }
                      }
                  }
              }
          });

          // Sortieren nach dem am nächsten liegenden Wechsel
          upcoming.sort((a, b) => a.startMins - b.startMins);

          const list = document.getElementById('radar-list');
          const badge = document.getElementById('radar-badge');

          if (upcoming.length === 0) {
              list.innerHTML = '<div class="p-6 text-center text-sm text-gray-400 italic"><i class="fa-solid fa-mug-hot text-2xl text-gray-300 mb-2 block"></i>Keine anstehenden Wechsel in den nächsten 2 Stunden.</div>';
              badge.classList.add('hidden');
              return;
          }

          // Badge (Rote Blase auf dem Button) für sehr dringende Wechsel (<= 15 Min)
          const urgentCount = upcoming.filter(u => u.diffMins <= 15).length;
          if (urgentCount > 0) {
              badge.innerText = urgentCount;
              badge.classList.remove('hidden');
              badge.classList.add('animate-bounce'); // Wackelt kurz
              setTimeout(() => badge.classList.remove('animate-bounce'), 3000);
          } else {
              badge.classList.add('hidden');
          }

          let html = '';
          // Max. 10 Einträge im Radar anzeigen
          upcoming.slice(0, 10).forEach(u => {
              let colorClass = "text-lidl-blue bg-blue-50 border-blue-100";
              let textCol = "text-blue-800";
              let inText = `in ${u.diffMins} Min`;
              
              if (u.diffMins <= 5) {
                   colorClass = "text-red-600 bg-red-50 border-red-200 animate-pulse";
                   textCol = "text-red-800";
              } else if (u.diffMins <= 15) {
                   colorClass = "text-orange-600 bg-orange-50 border-orange-200";
                   textCol = "text-orange-800";
              }

              html += `
              <div onclick="openShiftDetailsModal(${u.empId})" class="p-3 border-b border-gray-100 last:border-0 hover:bg-gray-100 cursor-pointer transition">
                  <div class="flex justify-between items-center mb-1">
                      <span class="text-xs font-bold text-slate-700 truncate">${u.empName}</span>
                      <span class="text-[10px] font-bold px-2 py-0.5 rounded-full border ${colorClass} shrink-0 ml-2">${inText}</span>
                  </div>
                  <div class="flex items-center text-[10px] text-gray-500 mt-1">
                      <i class="fa-solid fa-arrow-right-arrow-left mr-1.5 text-gray-400"></i>
                      <span class="truncate font-bold ${textCol}">${u.activity}</span>
                      <span class="ml-auto font-mono text-gray-400 font-bold shrink-0">${u.timeStr.split('-')[0]} Uhr</span>
                  </div>
              </div>`;
          });
          
          list.innerHTML = html;
      }

      function decimalToTime(decimalVal) {
          if (decimalVal === null || decimalVal === undefined) return "00:00";
          const isNegative = decimalVal < 0;
          decimalVal = Math.abs(decimalVal);
          const hours = Math.floor(decimalVal);
          const minutes = Math.round((decimalVal - hours) * 60);
          if (minutes === 60) {
              const displayHours = (hours + 1).toLocaleString('de-DE', { minimumIntegerDigits: 2 });
              return (isNegative ? "-" : "") + displayHours + ":00";
          }

/**
 * ZENTRALE BERECHTIGUNG
 * Prüft differenziert nach Zugriff, Admin-Status und Dienstplaner-Status.
 */
function getUserPermissions(email) {
  if (!email) return { hasAccess: false, isAdmin: false, isDienstplaner: false };
  
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const searchEmail = email.toLowerCase().trim();
  
  let hasAccess = false;
  let isAdmin = false;
  let isDienstplaner = false; 

  // --- CHECK 1: LISTE TAGESSTEUERUNG ---
  const sheetTS = ss.getSheetByName('Tagessteuerung');
  if (sheetTS && sheetTS.getLastRow() > 1) {
    const data = sheetTS.getRange(2, 1, sheetTS.getLastRow() - 1, 2).getValues();
    for (let i = 0; i < data.length; i++) {
      const rowEmail = String(data[i][0]).toLowerCase().trim();
      if (rowEmail === searchEmail) {
        hasAccess = true; 
        const rowRole = String(data[i][1] || "").toLowerCase().trim();
        if (rowRole.includes('tagessteuerung') || rowRole.includes('dienstplanung')) isAdmin = true;
        if (rowRole.includes('dienstplanung')) isDienstplaner = true;
      }
    }
  }

  // --- CHECK 2: DATA_DB ROLLE (Fallback für Admins, falls nicht in TS Liste) ---
  if (!isAdmin) {
    const sheetDB = ss.getSheetByName('Data_DB');
    if (sheetDB && sheetDB.getLastRow() > 1) {
      const lastRow = sheetDB.getLastRow();
      const dataDB = sheetDB.getRange(2, 1, lastRow - 1, 14).getValues();

      const isDienstplanerInDB = dataDB.some(row => {
        const dbEmail = String(row[13] || "").trim().toLowerCase();
        const skill = String(row[4] || "").trim().toLowerCase();
        return dbEmail === searchEmail && skill === 'dienstplanung';
      });

      if (isDienstplanerInDB) {
        hasAccess = true;
        isAdmin = true;
        isDienstplaner = true; 
      }
    }
  }
  
  return { hasAccess, isAdmin, isDienstplaner };
}

/**
 * Hauptfunktion: Daten laden.
 */
function getData() {
  try {
    const userEmail = Session.getActiveUser().getEmail().toLowerCase();
    const permissions = getUserPermissions(userEmail);

    if (!permissions.hasAccess) {
      throw new Error("⛔ Zugriff verweigert: Ihre E-Mail (" + userEmail + ") ist nicht berechtigt.");
    }

    const ss = SpreadsheetApp.getActiveSpreadsheet();
    let sheetDB = ss.getSheetByName('Data_DB');
    let sheetAuftrag = ss.getSheetByName('Auftrag');
    let sheetTS = ss.getSheetByName('Tagessteuerung');
    let sheetSchicht = ss.getSheetByName('Schichtmodelle');
    let sheetAbbau = ss.getSheetByName('AbbauKW');
    
    if (!sheetDB) throw new Error('Blatt "Data_DB" fehlt.');
    
    if (!sheetAuftrag) { 
      sheetAuftrag = ss.insertSheet('Auftrag'); 
      sheetAuftrag.appendRow(['Name', 'Vorname', 'Datum', 'Stunden', 'Notizen', 'Erstellt von']); 
    }
    
    if (!sheetSchicht) { 
      sheetSchicht = ss.insertSheet('Schichtmodelle'); 
      sheetSchicht.appendRow(['Nachname', 'Vorname', 'Schichtzeit', 'Typ', 'Pause', 'Taetigkeiten']); 
    }

    // Cleanup ausführen
    processAuftragExpiry(sheetDB, sheetAuftrag);

    const lastRow = sheetDB.getLastRow();
    if (lastRow < 2) {
      return { employees: [], stats: {}, currentWeek: 0, currentYear: 0, isAdmin: permissions.isAdmin, isDienstplaner: permissions.isDienstplaner, abbauHistory: [] };
    }

    const data = sheetDB.getRange(2, 1, lastRow - 1, 14).getValues();
    const rawAuftraege = getAuftraege(sheetAuftrag);

    const tsNames = new Set();
    if (sheetTS && sheetTS.getLastRow() > 1) {
      const tsData = sheetTS.getRange(2, 3, sheetTS.getLastRow() - 1, 1).getValues();
      tsData.forEach(r => { 
        const full = String(r[0]).trim(); 
        if(full) tsNames.add(full.split(' ')[0].toLowerCase()); 
      });
    }

    // --- SCHICHTEN (Liest JSON aus Spalte 6 für die Timeline) ---
    const shiftMap = new Map(); 
    if (sheetSchicht.getLastRow() > 1) {
       const sData = sheetSchicht.getRange(2, 1, sheetSchicht.getLastRow()-1, 6).getValues();
       
       sData.forEach(r => {
         const key = (String(r[1]).trim() + " " + String(r[0]).trim()).toLowerCase();
         let activitiesArr = [];
         let detailsArr = [];
         
         if (r[5]) {
            try {
               // Wir versuchen den JSON-String aus PlanoImport.gs zu parsen
               const parsed = JSON.parse(String(r[5]));
               if (Array.isArray(parsed)) {
                  detailsArr = parsed; // Das ist für das Timeline-Modal
                  
                  // Fülle das unique Array für die Filter-Buttons
                  parsed.forEach(item => {
                     let a = item.activity.trim();
                     if (a && !a.toLowerCase().includes('pause') && a !== '-') {
                        if (!activitiesArr.includes(a)) activitiesArr.push(a);
                     }
                  });
               }
            } catch(e) {
               // Fallback, falls doch mal normaler Text drinsteht
               activitiesArr = String(r[5]).split(',').map(a => a.trim()).filter(a => a);
            }
         }
         
         shiftMap.set(key, { 
            time: r[2], 
            type: r[3],
            pause: String(r[4] || ""),
            activities: activitiesArr,
            details: detailsArr // Wird ins Frontend für den Dialog übertragen
         });
       });
    }

    let abbauRaw = [];
    if (sheetAbbau && sheetAbbau.getLastRow() > 1) {
       const rows = sheetAbbau.getDataRange().getValues();
       const headers = rows[0]; 
       for (let r = 1; r < rows.length; r++) {
          for (let c = 3; c < rows[r].length; c++) {
             const val = Number(rows[r][c]);
             const teamName = String(headers[c]); 
             if (val !== 0 && teamName) {
               abbauRaw.push({ y: rows[r][0], kw: rows[r][1], val: val, team: teamName, tma: "" });
             }
          }
       }
    }

    const formattedData = data.map((row, index) => {
      if (!row[0]) return null;
      let isAustritt = false;
      
      if (row[12]) {
         let aDate = (row[12] instanceof Date) ? row[12] : (typeof row[12] === 'string' ? extractDates(row[12])[0] : null);
         if (aDate) {
            aDate.setHours(0,0,0,0);
            const today = new Date(); today.setHours(0,0,0,0);
            if (aDate <= today) isAustritt = true;
         }
      }
      if (isAustritt) return null;

      const nachname = String(row[0] || "").trim();
      const vorname = String(row[1] || "").trim();
      const dbKey = (vorname + " " + nachname).toLowerCase();
      
      const myAuftraege = rawAuftraege.filter(a => a.name.toLowerCase() === nachname.toLowerCase() && a.vorname.toLowerCase() === vorname.toLowerCase())
          .map(a => { 
            let isTs = false; 
            if (a.erstelltVon) { 
              const creatorFirst = a.erstelltVon.split(' ')[0].toLowerCase(); 
              if (tsNames.has(creatorFirst)) isTs = true; 
            } 
            return { ...a, isTsRelevant: isTs }; 
          });

      const essDecimal = timeToDecimal(row[5]);
      const plannedDecimal = myAuftraege.reduce((sum, order) => sum + (order.stunden || 0), 0);
      let saldoDecimal = Math.round((essDecimal - plannedDecimal) * 100) / 100;
      
      return {
        id: index + 2,
        nachname: nachname, 
        vorname: vorname, 
        fullName: `${vorname} ${nachname}`,
        tmaVorname: String(row[3] || "Ohne TMA"), 
        skill: String(row[4] || "-"),
        stundenEss: row[5], 
        stundenSaldo: saldoDecimal, 
        letzteAktualisierung: row[6] ? formatDateCompat(row[6]) : "-",
        notizen: String(row[11] || ""), 
        austrittDatum: (row[12] instanceof Date) ? formatDateCompat(row[12]) : "",
        auftraege: myAuftraege, 
        schicht: shiftMap.get(dbKey) || { time: "", type: "", pause: "", activities: [], details: [] } 
      };
    }).filter(item => item !== null);

    return { 
      employees: formattedData, 
      currentWeek: getWeekNumber(new Date()), 
      currentYear: new Date().getFullYear(), 
      isAdmin: permissions.isAdmin, 
      isDienstplaner: permissions.isDienstplaner, 
      abbauHistory: abbauRaw 
    };
    
  } catch (e) { 
    throw e; 
  }
}

function processAuftragExpiry(sheetDB, sheetAuftrag) {
  const lastRowA = sheetAuftrag.getLastRow();
  if (lastRowA < 2) return; 

  const dbDataRange = sheetDB.getDataRange();
  const dbData = dbDataRange.getValues();
  const today = new Date(); today.setHours(0, 0, 0, 0);

  const dbMap = new Map();
  for (let i = 1; i < dbData.length; i++) { 
    dbMap.set((String(dbData[i][1]).trim() + " " + String(dbData[i][0]).trim()).toLowerCase(), i); 
  }

  const auftragData = sheetAuftrag.getRange(2, 1, lastRowA - 1, 6).getValues();
  let dbDirty = false, auftragDirty = false;
  const validAuftraege = [];

  auftragData.forEach((row) => {
    let dates = (row[2] instanceof Date) ? [row[2]] : extractDates(String(row[2]));
    let isExpired = false;
    
    if (dates.length > 0) { 
      const lastDate = dates[dates.length - 1]; 
      lastDate.setHours(0,0,0,0); 
      if (lastDate < today) isExpired = true; 
    }
    
    let processed = false;
    if (isExpired) {
      const dbKey = (String(row[1]).trim() + " " + String(row[0]).trim()).toLowerCase();
      if (dbMap.has(dbKey)) {
        const dbIdx = dbMap.get(dbKey);
        let newEssDec = timeToDecimal(dbData[dbIdx][5]) - timeToDecimal(row[3]);
        dbData[dbIdx][5] = decimalToTime(newEssDec); 
        dbData[dbIdx][6] = today; 
        dbDirty = true; 
        processed = true; 
        auftragDirty = true;
      }
    }
    if (!processed) {
      validAuftraege.push([row[0], row[1], row[2], row[3], row[4], row[5]]);
    }
  });

  if (dbDirty) {
     const dataToWrite = dbData.slice(1);
     const rangeFG = sheetDB.getRange(2, 6, dataToWrite.length, 2); 
     rangeFG.setNumberFormat("@"); 
     const valuesFG = rangeFG.getValues();
     let rangeDirty = false;
     for(let i=0; i<dataToWrite.length; i++) {
        if (valuesFG[i][0] !== dataToWrite[i][5]) { 
          valuesFG[i][0] = dataToWrite[i][5]; 
          valuesFG[i][1] = dataToWrite[i][6]; 
          rangeDirty = true; 
        }
     }
     if (rangeDirty) rangeFG.setValues(valuesFG);
  }
  
  if (auftragDirty) {
     sheetAuftrag.getRange(2, 1, lastRowA - 1, 6).clearContent();
     if (validAuftraege.length > 0) {
       sheetAuftrag.getRange(2, 1, validAuftraege.length, 6).setValues(validAuftraege);
     }
     SpreadsheetApp.flush();
  }
}

  return sheet.getRange(2, 1, lastRow - 1, 6).getValues().map((row, idx) => ({

    vorname: String(row[1]), 
    datum: formatDateCompat(row[2]), 
    stunden: timeToDecimal(row[3]), 
    stundenDisplay: row[3], 
    notizen: String(row[4]), 
    erstelltVon: String(row[5] || "") 
  }));
}

function getEditorNameFromEmail(email) {
  if (!email) return "";
  try { 
    const localPart = email.split('@')[0].split('.'); 
    if (localPart.length > 0) return localPart[0].replace(/_/g, ' ').replace(/\b\w/g, l => l.toUpperCase()); 
    return email; 
  } catch (e) { 
    return email; 
  }
}

function extractDates(str) {
  const matches = []; 
  let match; 
  const regex = /(\d{1,2})\.(\d{1,2})\.(\d{4})/g;
  while ((match = regex.exec(str)) !== null) { 
    matches.push(new Date(match[3], match[2] - 1, match[1])); 
  }
  return matches.sort((a, b) => a - b);
}

function formatDateCompat(dateObj) {
  if (!dateObj) return "-"; 
  let d = null;
  if (dateObj instanceof Date) d = dateObj;
  else if (typeof dateObj === 'string') { 
    if (dateObj.match(/[A-Za-z]{3}\s[A-Za-z]{3}\s\d{2}/) || dateObj.includes("GMT")) { 
      const tryD = new Date(dateObj); 
      if (!isNaN(tryD.getTime())) d = tryD; else return dateObj; 
    } else return dateObj; 
  }
  if (d && !isNaN(d.getTime())) return Utilities.formatDate(d, Session.getScriptTimeZone(), "dd.MM.yyyy");
  return String(dateObj);
}

function getWeekNumber(d) {
    d = new Date(Date.UTC(d.getFullYear(), d.getMonth(), d.getDate())); 
    d.setUTCDate(d.getUTCDate() + 4 - (d.getUTCDay()||7));
    return Math.ceil((((d - new Date(Date.UTC(d.getUTCFullYear(),0,1))) / 86400000) + 1)/7);
}

function timeToDecimal(timeStr) {
  if (!timeStr) return 0; 
  if (typeof timeStr === 'number') return timeStr; 
  if (timeStr instanceof Date) return timeStr.getHours() + (timeStr.getMinutes() / 60);
  
  timeStr = String(timeStr).trim();
  let isNegative = timeStr.startsWith("-"); 
  if (isNegative) timeStr = timeStr.substring(1);
  
  const parts = timeStr.split(':'); 
  if (parts.length < 2) return parseFloat(timeStr) || 0; 
  
  let decimal = parseInt(parts[0], 10) + (parseInt(parts[1], 10) / 60);
  return isNegative ? -decimal : decimal;
}

function decimalToTime(decimalVal) {
  if (decimalVal === null || decimalVal === undefined) return "00:00";
  const isNegative = decimalVal < 0; 
  decimalVal = Math.abs(decimalVal);
  
  const hours = Math.floor(decimalVal); 
  const minutes = Math.round((decimalVal - hours) * 60);
  
  if (minutes === 60) return (isNegative ? "-" : "") + String(hours + 1).padStart(2, '0') + ":00";
  return (isNegative ? "-" : "") + String(hours).padStart(2, '0') + ":" + String(minutes).padStart(2, '0');
}

function updateAbbauStats(dateStr, hoursDecimal, operation, teamName) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName('AbbauKW');
  
  if (!sheet) { 
    sheet = ss.insertSheet('AbbauKW'); 
    sheet.appendRow(['Jahr', 'KW', 'Gesamt']); 
    sheet.setFrozenRows(1); 
    sheet.setFrozenColumns(2); 
  }
  
  let dateObj = (dateStr instanceof Date) ? dateStr : extractDates(String(dateStr))[0]; 
  if (!dateObj) return;
  
  const year = dateObj.getFullYear(); 
  const kw = getWeekNumber(dateObj); 
  const team = teamName || "Sonstiges"; 
  
  const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
  let colIndex = headers.findIndex(h => String(h).toLowerCase() === String(team).toLowerCase()) + 1;
  
  if (colIndex === 0) { 
    colIndex = sheet.getLastColumn() + 1; 
    sheet.getRange(1, colIndex).setValue(team).setFontWeight("bold"); 
  }
  
  const lastRow = sheet.getLastRow(); 
  let rowIndex = -1;
  
  if (lastRow > 1) { 
    const timeData = sheet.getRange(2, 1, lastRow - 1, 2).getValues(); 
    for (let r = 0; r < timeData.length; r++) { 
      if (timeData[r][0] == year && timeData[r][1] == kw) { 
        rowIndex = r + 2; 
        break; 
      } 
    } 
  }
  
  if (rowIndex === -1) { 
    sheet.appendRow([year, kw, 0]); 
    rowIndex = sheet.getLastRow(); 
  }
  
  const cell = sheet.getRange(rowIndex, colIndex);
  const oldVal = Number(cell.getValue()) || 0;
  const change = operation === 'add' ? hoursDecimal : -hoursDecimal;
  cell.setValue(Math.round((oldVal + change) * 100) / 100);
}

function getEmployeeTeam(nachname, vorname) {
  const sheetDB = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Data_DB'); 
  if (!sheetDB) return "Unbekannt";
  
  const data = sheetDB.getDataRange().getValues();
  for(let i=1; i<data.length; i++) { 
    if(String(data[i][0]).toLowerCase().trim() == String(nachname).toLowerCase().trim() && 
       String(data[i][1]).toLowerCase().trim() == String(vorname).toLowerCase().trim()) {
       return String(data[i][4]); 
    } 
  }
  return "Unbekannt";
}
