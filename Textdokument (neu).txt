  function openHolidayModal() {
     const modal = document.getElementById('holidayModal');

     } else {
         sortedDates.forEach(dateIso => {
             const parts = dateIso.split('-');
             const d = new Date(parts[0], parts[1] - 1, parts[2]);
             const weekday = d.toLocaleDateString('de-DE', { weekday: 'short' });
             
             const deDate = `${parts[2]}.${parts[1]}.`; 
             const name = globalHolidays[dateIso];
             
             container.innerHTML += `
               <div class="flex justify-between border-b border-gray-50 pb-1 last:border-0">
                  <span>${name}</span>
                  <span class="font-bold text-gray-500 flex gap-2">
                    <span class="text-gray-400 font-normal">${weekday}</span>
                    ${deDate}
                  </span>
               </div>`;
         });
     }

         modal.classList.remove('opacity-0');
         document.getElementById('holidayModalContent').classList.remove('scale-95');

  function handleSlotTypeChange(selectEl) {

  function loadEmpDetailsForSlot(slotRow) {
      if (!slotRow || !vacSelectedEmp) return;

      const hoursInput = slotRow.querySelector('.vac-daily-hours');
      const saldoSpan = slotRow.querySelector('.vac-saldo-display');

      if (!hoursInput || !saldoSpan) return;

      const applyDetails = res => {
        if (!hoursInput.value || hoursInput.value === "Lade...") {
          hoursInput.value = res.hours;
        }

        const saldoVal = res.saldo;
        const formattedSaldo = decimalToTime(saldoVal);

        let colorClass = "text-modern-green";

        if (saldoVal < 0) {
          colorClass = "text-red-600";
        } else if (saldoVal > 40) {
          colorClass = "text-red-500";
        }

        saldoSpan.className = "vac-saldo-display ml-1 font-mono font-bold text-sm " + colorClass;
        saldoSpan.innerText = formattedSaldo + " Std.";
      };

      if (vacSelectedEmpDetails) {
        applyDetails(vacSelectedEmpDetails);
        return;
      }

      if (!hoursInput.value) {
        hoursInput.value = "Lade...";
      }

      saldoSpan.innerText = "...";

      google.script.run.withSuccessHandler(res => {
        vacSelectedEmpDetails = res;
        applyDetails(res);
      }).getEmployeeDetails(vacSelectedEmp.label);
  }

  }

  function getSlotData() {
      const rows = document.querySelectorAll('.vac-slot-row');
      const ranges = [];

      rows.forEach(r => {
          const s = r.querySelector('.vac-start-input').value;
          const e = r.querySelector('.vac-end-input').value;
          const type = r.querySelector('.vac-type-select') ? r.querySelector('.vac-type-select').value : 'Urlaub (genehmigt)';
          const dailyHoursInput = r.querySelector('.vac-daily-hours');
          const dailyHours = dailyHoursInput ? dailyHoursInput.value : "";

          if (s && e) {
            ranges.push({
              start: s,
              end: e,
              type: type,
              dailyHours: dailyHours,
              domRow: r
            });
          }
      });

      return ranges;
  }

                  const feedbackDiv = slot.domRow.querySelector('.slot-quota-info');

                      const match = res.details.find(d => d.kw == wk.kw && d.year == wk.year);

                           let colorClass = "text-green-700 bg-green-50 border-green-200";
                           let icon = "fa-circle-check";

                           } else if (match.status === 'unknown') {
                               colorClass = "text-gray-600 bg-gray-50 border-gray-200";

                           }

                  });

                  feedbackDiv.innerHTML = html;
              });
          }
      }).checkQuotaBatch(vacSelectedEmp.tma, vacSelectedEmp.name, ranges);
  }

  function getWeeksInRangeClient(d1, d2) {
      const weeks = [];
      const current = new Date(d1);
      const day = current.getDay() || 7; 

      if (day !== 1) {
        current.setHours(-24 * (day - 1));
      }

      while (current <= d2) {
          weeks.push(getKW(current));
          current.setDate(current.getDate() + 7);
      }

      const endKw = getKW(d2);

      if (!weeks.some(w => w.kw === endKw.kw && w.year === endKw.year)) {
          weeks.push(endKw);
      }

      return weeks;
  }

  function getKW(date) {
      const d = new Date(Date.UTC(date.getFullYear(), date.getMonth(), date.getDate()));

      d.setUTCDate(d.getUTCDate() + 4 - (d.getUTCDay() || 7));

      const yearStart = new Date(Date.UTC(d.getUTCFullYear(), 0, 1));
      const kw = Math.ceil((((d - yearStart) / 86400000) + 1) / 7);

      return { kw: kw, year: d.getUTCFullYear() };
  }

  function submitVacation() {
      const slotObjects = getSlotData();

      if (!vacSelectedEmp || slotObjects.length === 0) {
          alert("Bitte Mitarbeiter wählen und mindestens einen vollständigen Zeitraum angeben.");
          return;
      }

      for (const slot of slotObjects) {
        if (slot.type === 'Freizeitausgleich') {
          if (!slot.dailyHours || slot.dailyHours === "Lade..." || !slot.dailyHours.includes(":")) {
            alert("Bitte bei jedem Freizeitausgleich gültige Vertragsstunden im Format hh:mm angeben.");
            return;
          }
        }
      }

      const btn = document.getElementById('btn-vac-save');
      const oldText = btn.innerText;

      btn.innerText = "Speichere...";
      btn.disabled = true;

      const entries = slotObjects.map(r => ({
        start: r.start,
        end: r.end,
        type: r.type,
        dailyHours: r.dailyHours || "00:00"
      }));

      const payload = {
          name: vacSelectedEmp.label,
          tma: vacSelectedEmp.tma,
          team: vacSelectedEmp.team,
          entries: entries,
          submitted: document.getElementById('vac-submitted').value
      };

      google.script.run.withSuccessHandler(() => {
          btn.innerText = "Gespeichert!";

          setTimeout(() => {
              closeVacationModal();

              btn.innerText = oldText;
              btn.disabled = false;

              initDienstplanView(); 

              if (typeof loadData === 'function') {
                loadData(); 
              }
          }, 800);
      }).withFailureHandler(err => {
          alert("Fehler: " + err.message);

          btn.innerText = oldText;
          btn.disabled = false;
      }).saveVacationBatch(payload);
  }

  function updateBudget(empId, val) {
     if (!val || val < 0) return;

     val = parseInt(val);

     if (emp) {
         emp.vacationBudget = val; 
     }

     const feedback = document.getElementById(`save-feedback-${empId}`);

     if (feedback) {
         feedback.innerHTML = '<i class="fa-solid fa-spinner fa-spin"></i> Speichere...';
         feedback.classList.remove('opacity-0', 'text-green-600');
         feedback.classList.add('text-gray-400');
     }
     
     const display = document.getElementById(`budget-display-${empId}`);
     const bar = document.getElementById(`budget-bar-${empId}`);
     
     if (emp) {
         const usedDays = emp.vacations
              .filter(v => v.type === 'Urlaub (genehmigt)')
              .reduce((sum, v) => sum + (v.days || 0), 0);
         
         const percent = Math.min((usedDays / val) * 100, 100);
         
         if (display) {
          display.innerText = val;
         }
         
         const counterSpan = display.closest('.text-xl'); 

         if (counterSpan) {
             if (usedDays > val) {
                 counterSpan.classList.remove('text-modern-green');
                 counterSpan.classList.add('text-red-500');

                 if (bar) {
                  bar.classList.remove('bg-lidl-blue');
                  bar.classList.add('bg-red-500');
                 }
             } else {
                 counterSpan.classList.add('text-modern-green');
                 counterSpan.classList.remove('text-red-500');

                 if (bar) {
                  bar.classList.add('bg-lidl-blue');
                  bar.classList.remove('bg-red-500');
                 }
             }
         }

         if (bar) {
          bar.style.width = percent + "%";
         }
     }
     
     google.script.run.withSuccessHandler(() => {
         if (feedback) {
             feedback.innerHTML = '<i class="fa-solid fa-check"></i> Gespeichert';
             feedback.classList.remove('text-gray-400');
             feedback.classList.add('text-green-600');

             setTimeout(() => feedback.classList.add('opacity-0'), 2000);
         }
     }).updateEmployeeBudget(empId, val);
  }

  window.filterAndRenderDienstplan = function(onlyBody = false) {
     if (!rawDienstplanData.length && !dienstplanInitDone) return;

     const startFilter = document.getElementById('filter-start') ? document.getElementById('filter-start').value : null;
     const endFilter = document.getElementById('filter-end') ? document.getElementById('filter-end').value : null;

     let filtered = rawDienstplanData.filter(d => {
         const teamMatch = selectedTeams.size === 0 || selectedTeams.has(d.team);
         const tmaMatch = selectedTmas.size === 0 || selectedTmas.has(d.tma);
         
         let nameMatch = true;

         if (currentNameFilter) {
             nameMatch = d.fullName.toLowerCase().includes(currentNameFilter);
         }

         let dateMatch = true;

         if (startFilter && endFilter) {
             if (d.vacations.length === 0) {
                 dateMatch = false; 
             } else {
                 const fS = new Date(startFilter);
                 const fE = new Date(endFilter);

                 const hasOverlap = d.vacations.some(v => {
                     const vS = new Date(v.start);
                     const vE = new Date(v.end);

                     return vS <= fE && vE >= fS;
                 });

                 dateMatch = hasOverlap;
             }
         }

         return teamMatch && tmaMatch && dateMatch && nameMatch;
     });
     
     if (currentDaySort === 'asc') {
         filtered.sort((a, b) => getApprovedDays(a) - getApprovedDays(b));
     } else if (currentDaySort === 'desc') {
         filtered.sort((a, b) => getApprovedDays(b) - getApprovedDays(a));
     }
     
     updateGlobalStats(filtered);
     renderCalendarBody(filtered);
  };

  function updateGlobalStats(filteredData) {
      const statPlanned = document.getElementById('stat-total-planned');
      const statCapacity = document.getElementById('stat-total-capacity');
      const statBar = document.getElementById('stat-bar-capacity');
      const statPast = document.getElementById('stat-past');
      const statFuture = document.getElementById('stat-future');
      
      if (!statPlanned) return; 

      const totalCapacity = filteredData.reduce((sum, emp) => sum + (emp.vacationBudget || 30), 0);
      
      let totalPlanned = 0;
      let pastDays = 0;
      let futureDays = 0;

      const today = new Date();

      today.setHours(0, 0, 0, 0);

      filteredData.forEach(emp => {
          emp.vacations.forEach(v => {
              if (v.type === 'Urlaub (genehmigt)') {
                  totalPlanned += (v.days || 0);
                  
                  let current = new Date(v.start);
                  const end = new Date(v.end);
                  
                  while (current <= end) {
                      const d = current.getDay();

                      if (d !== 0 && d !== 6) { 
                         if (current < today) {
                             pastDays++;
                         } else {
                             futureDays++;
                         }
                      }

                      current.setDate(current.getDate() + 1);
                  }
              }
          });
      });

      statPlanned.innerText = totalPlanned;
      statCapacity.innerText = totalCapacity;
      statPast.innerText = pastDays;
      statFuture.innerText = futureDays;

      const percent = totalCapacity > 0 ? (totalPlanned / totalCapacity) * 100 : 0;

      statBar.style.width = Math.min(percent, 100) + "%";
      
      if (percent > 100) {
        statBar.classList.replace('bg-lidl-blue', 'bg-red-500');
      } else {
        statBar.classList.replace('bg-red-500', 'bg-lidl-blue');
      }
  }

                ${MONTH_NAMES[currentViewMonth]} ${YEAR_VIEW}
              </th>

          const dayNum = d.getDate();

      }).join('');

      thead.innerHTML = monthRowHTML + dayRowHTML;
  }

  function renderCalendarBody(data) {
      const body = document.getElementById('calendar-body');

      if (!body) return;
      
      if (data.length === 0) {
          body.innerHTML = `<tr><td class="p-8 text-center text-gray-400" colspan="100%">Keine Daten verfügbar.</td></tr>`;
          return;
      }

      const daysInMonth = getDaysInMonth(YEAR_VIEW, currentViewMonth);

      const formatDate = (dateObj) => {
             const y = dateObj.getFullYear();
             const m = String(dateObj.getMonth() + 1).padStart(2, '0');
             const d = String(dateObj.getDate()).padStart(2, '0');

             return `${y}-${m}-${d}`;
      };

      const rowsHTML = data.map(emp => {
             const vacMap = new Map(); 

             if (emp.vacations && emp.vacations.length > 0) {
                 emp.vacations.forEach(range => {
                     let current = new Date(range.start);
                     const end = new Date(range.end);
                     const viewStart = daysInMonth[0];
                     const viewEnd = daysInMonth[daysInMonth.length - 1];

                     if (end < viewStart || current > viewEnd) return; 

                     while (current <= end) {
                         if (current >= viewStart && current <= viewEnd) {
                             vacMap.set(formatDate(current), range.type);
                         }

                         current.setDate(current.getDate() + 1);
                     }
                 });
             }

             const cellsHTML = daysInMonth.map(d => {
                 const dateStr = formatDate(d);
                 const isWeekend = d.getDay() === 0 || d.getDay() === 6;
                 const isoDate = formatDateIsoSimple(d);
                 const isHoliday = !!globalHolidays[isoDate];

                 let bgClass = isWeekend ? 'bg-gray-100 text-gray-400' : 'bg-white';

                 if (isHoliday) {
                  bgClass = 'bg-purple-50';
                 }

                 const borderClass = d.getDate() === 1 ? 'border-l-2 border-l-gray-300' : 'border-l border-gray-100';
                 const vacType = vacMap.get(dateStr);
                 const isVacation = !!vacType && !isWeekend && !isHoliday; 

                 let content = '';
                 let cellBg = bgClass;

                 if (isVacation) {
                     if (vacType === 'Freizeitausgleich') {
                         content = '<i class="fa-solid fa-business-time text-blue-500"></i>';
                         cellBg = 'bg-blue-50';
                     } else if (vacType === 'Sonderurlaub') {
                         content = '<i class="fa-solid fa-star text-purple-500"></i>';
                         cellBg = 'bg-purple-50';
                     } else {
                         content = '<i class="fa-solid fa-umbrella-beach text-orange-500"></i>';
                         cellBg = 'bg-orange-50';
                     }
                 }

                 return `<td class="${cellBg} ${borderClass} border-b border-gray-50 text-center p-0 relative group/cell cursor-pointer hover:bg-blue-200 transition-colors">${content}</td>`;
             }).join('');

             const totalUrlaub = emp.vacations
                .filter(v => v.type === 'Urlaub (genehmigt)')
                .reduce((sum, v) => sum + (v.days || 0), 0);

             const budget = emp.vacationBudget || 30;

             let antraegeClass = "text-modern-green";

             if (totalUrlaub > budget || (totalUrlaub > 0 && totalUrlaub < 5)) {
              antraegeClass = "text-red-600";
             }
             
             const displayCount = (totalUrlaub > 0 ? totalUrlaub : 0) + " / " + budget;

             return `<tr onclick="openPersonModal(${emp.id})" class="hover:bg-blue-50/30 transition-colors h-10 group/row cursor-pointer">
                 <td class="sticky left-0 z-10 bg-white border-r border-gray-200 px-4 font-medium text-slate-700 truncate max-w-[200px] border-b border-gray-50 group-hover/row:bg-blue-50">${emp.nachname}, ${emp.vorname}</td>
                 <td class="sticky left-[200px] z-10 bg-white border-r border-gray-200 px-4 text-gray-500 truncate max-w-[150px] border-b border-gray-50 text-xs group-hover/row:bg-blue-50">${emp.tma}</td>
                 <td class="sticky left-[350px] z-10 bg-white border-r border-gray-200 px-4 text-gray-500 truncate max-w-[120px] border-b border-gray-50 text-xs group-hover/row:bg-blue-50">${emp.team}</td>
                 <td class="sticky left-[470px] z-10 bg-white border-r border-gray-200 px-2 text-center border-b border-gray-50 text-xs font-bold ${antraegeClass} group-hover/row:bg-blue-50">${displayCount}</td>
                 ${cellsHTML}
             </tr>`;
      }).join('');

      body.innerHTML = rowsHTML;
      
      const loader = document.getElementById('calendar-loader');

      if (loader) {
        loader.classList.add('hidden');
      }
  }

  function getDaysInMonth(year, monthIndex) {
      const date = new Date(year, monthIndex, 1);
      const days = [];

      while (date.getMonth() === monthIndex) {
          days.push(new Date(date));
          date.setDate(date.getDate() + 1);
      }

      return days;
  }
  
  function formatDateIsoSimple(d) {
     const y = d.getFullYear();
     const m = String(d.getMonth() + 1).padStart(2, '0');
     const day = String(d.getDate()).padStart(2, '0');

     return `${y}-${m}-${day}`;
  }
