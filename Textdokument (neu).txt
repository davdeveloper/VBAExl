function openPlanoStundenerfassungModal() {
    const modal = document.getElementById('planoStundenerfassungModal');
    const list = document.getElementById('plano-st-files-list');

    list.innerHTML = '<div class="text-center text-gray-400 py-4"><i class="fa-solid fa-spinner fa-spin"></i> Lade Dateien...</div>';

    modal.classList.remove('hidden');

    setTimeout(() => {
      modal.classList.remove('opacity-0');
      document.getElementById('planoStundenerfassungModalContent').classList.remove('scale-95');
      document.getElementById('planoStundenerfassungModalContent').classList.add('scale-100');
    }, 10);

    google.script.run
      .withSuccessHandler(files => {
        list.innerHTML = '';

        if (!files || files.length === 0) {
          list.innerHTML = '<div class="text-sm text-gray-500 italic text-center py-4">Keine Backup-Dateien im Ordner gefunden.</div>';
          return;
        }

        files.forEach(f => {
          const d = new Date(f.dateCreated);
          const dateStr = d.toLocaleDateString('de-DE') + ' ' + d.toLocaleTimeString('de-DE', { hour: '2-digit', minute: '2-digit' });

          list.innerHTML += `
            <div class="flex justify-between items-center p-3 border border-gray-200 rounded-lg hover:bg-green-50 hover:border-green-200 transition group">
              <div>
                <div class="font-bold text-slate-700 text-sm flex items-center gap-2">
                  <i class="fa-solid fa-database text-modern-green"></i> ${f.name}
                </div>
                <div class="text-xs text-gray-400 mt-1">Erstellt am: ${dateStr}</div>
              </div>
              <button onclick="runPlanoStundenerfassungImport('${f.id}', this)" class="bg-white border border-gray-300 text-modern-green px-3 py-1.5 rounded text-xs font-bold hover:bg-modern-green hover:text-white transition shadow-sm">
                Datenbank Import
              </button>
            </div>`;
        });
      })
      .withFailureHandler(err => {
        list.innerHTML = `<div class="text-red-500 text-sm text-center py-4">Fehler: ${err.message}</div>`;
      })
      .getPlanoFilesList();
  }

  function runPlanoStundenerfassungImport(fileId, btnEl) {
    const oldContent = btnEl.innerHTML;

    btnEl.innerHTML = '<i class="fa-solid fa-spinner fa-spin"></i> Importiere...';
    btnEl.disabled = true;
    btnEl.classList.add('opacity-75', 'cursor-not-allowed');

    const allBtns = document.getElementById('plano-st-files-list').querySelectorAll('button');
    allBtns.forEach(b => b.disabled = true);

    google.script.run
      .withSuccessHandler(function(response) {
        closePlanoStundenerfassungModal();
        alert(response);
      })
      .withFailureHandler(function(error) {
        btnEl.innerHTML = oldContent;
        btnEl.disabled = false;
        btnEl.classList.remove('opacity-75', 'cursor-not-allowed');
        allBtns.forEach(b => b.disabled = false);
        alert("Kritischer Fehler: " + error.message);
      })
      .libProcessSelectedFile(fileId);
  }

  function startTransfer() {

      return;

    const oldHtml = btn.innerHTML;

    btn.innerHTML = '<i class="fa-solid fa-spinner fa-spin"></i> Übermittlung läuft...';

      .withSuccessHandler(res => {

        alert("FEHLER: " + err.message);
        btn.innerHTML = oldHtml;

      })
      .libProcessStundenerfassung(dateVal);
  }

  function runPlanoImport(fileId, btnEl) {
    const oldContent = btnEl.innerHTML;

    btnEl.innerHTML = '<i class="fa-solid fa-spinner fa-spin"></i> Lade...';
    btnEl.disabled = true;
    btnEl.classList.add('opacity-75', 'cursor-not-allowed');

    const allBtns = document.getElementById('plano-files-list').querySelectorAll('button');

    allBtns.forEach(b => b.disabled = true);

    google.script.run
      .withSuccessHandler(function(response) {
        closePlanoModal();

        if (response.success) {
          alert("Import erfolgreich! " + response.count + " Schichten importiert.\nDie Seite wird nun aktualisiert.");
          document.getElementById('loader').classList.remove('hidden');

          google.script.run
            .withSuccessHandler(onDataLoaded)
            .withFailureHandler(onDataError)
            .getData();
        } else {
          alert("Import fehlgeschlagen: " + response.message);
        }
      })
      .withFailureHandler(function(error) {
        btnEl.innerHTML = oldContent;
        btnEl.disabled = false;
        btnEl.classList.remove('opacity-75', 'cursor-not-allowed');

        allBtns.forEach(b => b.disabled = false);

        alert("Kritischer Fehler: " + error.message);
      })
      .importSelectedPlano(fileId);
  }

  function renderControl(data) {
    const listFrueh = document.getElementById('list-frueh');
    const listMittel = document.getElementById('list-mittel');
    const listSpaet = document.getElementById('list-spaet');

    let allActivities = new Set();

    const sorted = [...data].sort((a, b) => {
      if (activeControlSort === 'plus') return b.stundenSaldo - a.stundenSaldo;
      if (activeControlSort === 'minus') return a.stundenSaldo - b.stundenSaldo;

      if (activeControlSort === 'time') {
        const timeA = parseStartTimeMinutes(getRelevantTimeStr(a, activeActivityFilter));
        const timeB = parseStartTimeMinutes(getRelevantTimeStr(b, activeActivityFilter));

        return timeA - timeB;
      }

      return 0;
    });

    sorted.forEach(e => {
      if (e.schicht && e.schicht.activities && e.schicht.activities.length > 0) {
        e.schicht.activities.forEach(act => allActivities.add(act));
      }
    });

    buildActivityFilters(Array.from(allActivities).sort());

    listFrueh.innerHTML = '';
    listMittel.innerHTML = '';
    listSpaet.innerHTML = '';

    let cF = 0;
    let cM = 0;
    let cS = 0;

    let sF = 0;
    let sM = 0;
    let sS = 0;

    sorted.forEach(e => {
      if (e.schicht && e.schicht.type) {
        let showItem = true;

        if (activeActivityFilter !== 'Alle') {
          if (!e.schicht.activities || !e.schicht.activities.includes(activeActivityFilter)) {
            showItem = false;
          }
        }

        if (showItem) {
          const col = getBalanceColor(e.stundenSaldo);

          let pauseBadge = "";

          if (e.schicht.pause) {
            pauseBadge = `<span class="bg-gray-100 text-gray-500 text-[9px] px-1.5 py-0.5 rounded mt-1 whitespace-nowrap inline-block" title="Pause"><i class="fa-solid fa-mug-hot mr-1"></i>${e.schicht.pause}</span>`;
          }

          let actHtml = "";

          if (e.schicht.activities && e.schicht.activities.length > 0) {
            const actsStr = e.schicht.activities.join(' • ');
            actHtml = `<div class="text-[9px] text-gray-400 mt-1 truncate" title="${actsStr}"><i class="fa-solid fa-briefcase mr-1"></i>${actsStr}</div>`;
          }

          const displayTime = getRelevantTimeStr(e, activeActivityFilter);

          let timeClass = "text-lidl-blue bg-blue-50 border border-blue-100";

          if (activeActivityFilter !== 'Alle' && displayTime !== e.schicht.time) {
            timeClass = "text-white bg-lidl-blue shadow-sm border-transparent";
          }

          const item = `
            <div onclick="openShiftDetailsModal(${e.id})" class="p-3 border border-gray-100 rounded-xl bg-white hover:border-blue-300 hover:shadow-md cursor-pointer flex flex-col gap-1 text-sm transition-all group">
              <div class="flex justify-between items-start">
                <div class="overflow-hidden pr-2">
                  <div class="font-bold text-slate-700 truncate">${e.fullName}</div>
                  <div class="flex items-center text-[10px] text-gray-500 mt-0.5">
                    <span class="truncate max-w-[80px]" title="${e.skill}">${e.skill}</span>
                    <span class="mx-1 text-gray-300">|</span>
                    <span class="font-mono font-bold ${col}">${decimalToTime(e.stundenSaldo)}</span>
                  </div>
                </div>
                <div class="text-right flex flex-col items-end shrink-0">
                  <div class="font-mono text-[10px] font-bold px-2 py-0.5 rounded ${timeClass} whitespace-nowrap">${displayTime}</div>
                  ${pauseBadge}
                </div>
              </div>
              ${actHtml}
            </div>`;

          const saldoVal = parseFloat(e.stundenSaldo) || 0;

          if (e.schicht.type === "Früh") {
            listFrueh.innerHTML += item;
            cF++;
            if (saldoVal > 0) sF += saldoVal;
          } else if (e.schicht.type === "Mittel") {
            listMittel.innerHTML += item;
            cM++;
            if (saldoVal > 0) sM += saldoVal;
          } else if (e.schicht.type === "Spät") {
            listSpaet.innerHTML += item;
            cS++;
            if (saldoVal > 0) sS += saldoVal;
          }
        }
      }
    });

    document.getElementById('count-frueh').innerText = cF;
    document.getElementById('sum-frueh').innerText = "+" + decimalToTime(sF) + " Std";

    document.getElementById('count-mittel').innerText = cM;
    document.getElementById('sum-mittel').innerText = "+" + decimalToTime(sM) + " Std";

    document.getElementById('count-spaet').innerText = cS;
    document.getElementById('sum-spaet').innerText = "+" + decimalToTime(sS) + " Std";
  }

  function getRelevantTimeStr(emp, filter) {
    if (filter === 'Alle' || !emp.schicht || !emp.schicht.details || emp.schicht.details.length === 0) {
      return emp.schicht ? emp.schicht.time : "";
    }

    const matches = emp.schicht.details.filter(d => d.activity === filter);

    if (matches.length > 0) {
      if (matches.length === 1) {
        return matches[0].time;
      }

      let minStart = null;
      let maxEnd = null;
      let minStartMins = 99999;
      let maxEndMins = -1;

      matches.forEach(m => {
        const tMatch = m.time.match(/(\d{2}:\d{2})\s*-\s*(\d{2}:\d{2})/);

        if (tMatch) {
          const sStr = tMatch[1];
          const eStr = tMatch[2];

          const sMins = parseInt(sStr.split(':')[0], 10) * 60 + parseInt(sStr.split(':')[1], 10);
          let eMins = parseInt(eStr.split(':')[0], 10) * 60 + parseInt(eStr.split(':')[1], 10);

          if (eMins <= sMins) {
            eMins += 24 * 60;
          }

          if (sMins < minStartMins) {
            minStartMins = sMins;
            minStart = sStr;
          }

          if (eMins > maxEndMins) {
            maxEndMins = eMins;
            maxEnd = eStr;
          }
        }
      });

      if (minStart && maxEnd) {
        return `${minStart}-${maxEnd}`;
      }

      return matches.map(m => m.time).join(', ');
    }

    return emp.schicht.time;
  }

  function parseStartTimeMinutes(timeStr) {
    if (!timeStr) return 9999;

    const match = String(timeStr).match(/(\d{2}):(\d{2})/);

    if (match) {
      return parseInt(match[1], 10) * 60 + parseInt(match[2], 10);
    }

    return 9999;
  }

  function buildActivityFilters(activities) {
    const container = document.getElementById('activity-filter-container');

    if (!container) return;

    container.innerHTML = '';

    const btnAll = document.createElement('button');

    btnAll.innerText = 'Alle anzeigen';
    btnAll.className = `act-filter-chip px-3 py-1 rounded-full border shadow-sm ${activeActivityFilter === 'Alle' ? 'active' : 'default'}`;
    btnAll.onclick = () => {
      activeActivityFilter = 'Alle';
      applyFilters();
    };

    container.appendChild(btnAll);

    activities.forEach(act => {
      const btn = document.createElement('button');

      btn.innerText = act;
      btn.className = `act-filter-chip px-3 py-1 rounded-full border shadow-sm ${activeActivityFilter === act ? 'active' : 'default'}`;
      btn.onclick = () => {
        activeActivityFilter = act;
        applyFilters();
      };

      container.appendChild(btn);
    });
  }
