<script>
  let dayPlanningState = null;
  let dayPlanningSaveTimer = null;
  let dayPlanningIsSaving = false;
  let dayPlanningBacklogLevel = '1st';

  function openDayPlanningModal() {
    const modal = document.getElementById('dayPlanningModal');
    const content = document.getElementById('day-planning-content');

    if (!modal || !content) return;

    modal.classList.remove('hidden');

    setTimeout(() => {
      modal.classList.remove('opacity-0');
      document.getElementById('dayPlanningModalContent').classList.remove('scale-95');
      document.getElementById('dayPlanningModalContent').classList.add('scale-100');
    }, 10);

    content.innerHTML = `
      <div class="flex flex-col items-center justify-center py-16 text-gray-400">
        <i class="fa-solid fa-circle-notch fa-spin text-4xl mb-3 text-lidl-blue"></i>
        <p class="font-medium">Tagesplanung wird geladen...</p>
        <p class="text-xs mt-1">Plano Produktivstunden und gespeicherte Planung werden zusammengeführt</p>
      </div>
    `;

    google.script.run
      .withSuccessHandler(data => {
        dayPlanningState = data;
        renderDayPlanning(data);
        setDayPlanningSaveStatus(data.savedAt ? `Letzter Speicherstand: ${data.savedAt}` : 'Noch kein Speicherstand für heute.');
      })
      .withFailureHandler(err => {
        content.innerHTML = `
          <div class="bg-red-50 border border-red-200 text-red-700 rounded-xl p-6 text-center">
            <i class="fa-solid fa-triangle-exclamation text-3xl mb-3"></i>
            <h4 class="font-bold text-lg">Fehler beim Laden</h4>
            <p class="text-sm mt-2">${err.message || err}</p>
          </div>
        `;
      })
      .getDayPlanningData();
  }

  function closeDayPlanningModal() {
    if (dayPlanningState) {
      collectDayPlanningInputs();
      saveDayPlanningNow(true);
    }

    const modal = document.getElementById('dayPlanningModal');
    const modalContent = document.getElementById('dayPlanningModalContent');

    if (!modal || !modalContent) return;

    modal.classList.add('opacity-0');
    modalContent.classList.remove('scale-100');
    modalContent.classList.add('scale-95');

    setTimeout(() => {
      modal.classList.add('hidden');
    }, 300);
  }

  function reloadDayPlanningFromServer() {
    openDayPlanningModal();
  }

  function renderDayPlanning(data) {
    const content = document.getElementById('day-planning-content');

    if (!content) return;

    if (!data || !data.success) {
      content.innerHTML = `
        <div class="bg-white border border-gray-200 rounded-xl p-8 text-center text-gray-400">
          <i class="fa-regular fa-folder-open text-4xl mb-3"></i>
          <h4 class="font-bold text-slate-600 text-lg">Keine Tagesplanung geladen</h4>
          <p class="text-sm mt-2">Bitte importiere zuerst das heutige Plano Backup.</p>
        </div>
      `;
      return;
    }

    const totals = data.totals || {};
    const gap = Number(totals.gap || 0);
    const remainingGap = Number(totals.remainingGap || 0);
    const gapClass = gap >= 0 ? 'text-modern-green' : 'text-red-600';
    const remainingGapClass = remainingGap >= 0 ? 'text-modern-green' : 'text-red-600';

    let html = `
      <div class="space-y-5">
        <div class="bg-white border border-gray-100 rounded-2xl shadow-sm overflow-hidden">
          <div class="bg-gradient-to-r from-blue-50 via-white to-yellow-50 p-4 border-b border-gray-100 flex flex-col xl:flex-row xl:items-center xl:justify-between gap-4">
            <div>
              <div class="text-[10px] uppercase font-bold text-gray-400">Planungstag</div>
              <div class="text-2xl font-bold text-slate-700 flex items-center gap-3">
                ${data.displayDate}
                <span class="text-xs bg-blue-50 text-lidl-blue border border-blue-100 rounded-full px-2 py-1">KW ${data.kw}</span>
              </div>
              <p class="text-xs text-gray-400 mt-1">Produktivität kommt aus dem Blatt <strong>Produktivität</strong>. Fehlt ein Wert, wird mit 5 Cases pro Stunde gerechnet.</p>
            </div>

            <div class="grid grid-cols-2 md:grid-cols-5 gap-3 w-full xl:w-auto">
              <div class="bg-white rounded-xl border border-gray-100 p-3 text-right shadow-sm">
                <div class="text-[9px] uppercase font-bold text-gray-400">Plano Std.</div>
                <div class="font-mono text-xl font-bold text-slate-700">${planningDecimalToTime(totals.planoHours || 0)}</div>
              </div>
              <div class="bg-white rounded-xl border border-gray-100 p-3 text-right shadow-sm">
                <div class="text-[9px] uppercase font-bold text-gray-400">Rest Std.</div>
                <div class="font-mono text-xl font-bold text-modern-green">${planningDecimalToTime(totals.remainingHours || 0)}</div>
              </div>
              <div class="bg-white rounded-xl border border-gray-100 p-3 text-right shadow-sm">
                <div class="text-[9px] uppercase font-bold text-gray-400">Backlog</div>
                <div class="font-mono text-xl font-bold text-slate-700">${Math.round(Number(totals.backlog || 0))}</div>
              </div>
              <div class="bg-white rounded-xl border border-gray-100 p-3 text-right shadow-sm">
                <div class="text-[9px] uppercase font-bold text-gray-400">Schaffbar</div>
                <div class="font-mono text-xl font-bold text-lidl-blue">${Math.round(Number(totals.possibleCases || 0))}</div>
              </div>
              <div class="bg-white rounded-xl border border-gray-100 p-3 text-right shadow-sm">
                <div class="text-[9px] uppercase font-bold text-gray-400">Gap Gesamt</div>
                <div class="font-mono text-xl font-bold ${gapClass}">${gap >= 0 ? '+' : ''}${Math.round(gap)}</div>
              </div>
            </div>
          </div>

          <div class="p-4 bg-white flex flex-col md:flex-row md:items-center md:justify-between gap-3">
            <div class="text-xs text-gray-500 flex items-center gap-2">
              <i class="fa-solid fa-circle-info text-lidl-blue"></i>
              Werte werden automatisch in <strong>Tagessteuerung_Planung</strong> zwischengespeichert und in <strong>Datenbank_Tagessteuerung</strong> historisiert.
            </div>
            <div class="text-xs font-bold ${remainingGapClass} bg-gray-50 border border-gray-100 rounded-full px-3 py-1">
              Rest-Gap ab jetzt: ${remainingGap >= 0 ? '+' : ''}${Math.round(remainingGap)} Cases
            </div>
          </div>
        </div>

        <div class="grid grid-cols-1 xl:grid-cols-2 2xl:grid-cols-4 gap-4" id="day-planning-team-grid">
    `;

    (data.teams || []).forEach((team, index) => {
      html += renderDayPlanningTeamCard(team, index);
    });

    html += `
        </div>
      </div>
    `;

    content.innerHTML = html;
  }

  function renderDayPlanningTeamCard(team, index) {
    const backlog = Number(team.backlog || 0);
    const productivity = Number(team.productivity || 5);
    const planoHours = Number(team.planoHours || 0);
    const remainingHours = Number(team.remainingHours || 0);
    const possibleCases = Math.round(planoHours * productivity);
    const possibleRemainingCases = Math.round(remainingHours * productivity);
    const gap = possibleCases - backlog;
    const remainingGap = possibleRemainingCases - backlog;
    const isOk = gap >= 0;
    const status = team.status || 'offen';
    const note = team.note || '';
    const details = Array.isArray(team.backlogDetails) ? team.backlogDetails : [];

    const statusClass = isOk
      ? 'bg-emerald-50 text-modern-green border-emerald-100'
      : 'bg-red-50 text-red-600 border-red-100';

    const detailHtml = details.length
      ? `<div class="mt-2 text-[10px] text-gray-400 line-clamp-2" title="${planningEscape(details.map(d => `${d.label}: ${d.value}`).join(' | '))}">${planningEscape(details.map(d => `${d.label}: ${d.value}`).join(' • '))}</div>`
      : '';

    return `
      <div class="day-planning-card bg-white border border-gray-100 rounded-2xl shadow-sm hover:shadow-md transition overflow-hidden" data-team-index="${index}">
        <div class="p-4 border-b border-gray-100 bg-gradient-to-r from-white to-blue-50">
          <div class="flex items-start justify-between gap-3">
            <div class="min-w-0">
              <h4 class="font-bold text-slate-700 flex items-center gap-2 truncate">
                <i class="fa-solid fa-circle text-[7px] text-lidl-blue"></i>
                <span title="${planningEscape(team.team)}">${planningEscape(team.team)}</span>
              </h4>
              <div class="text-[11px] text-gray-400 mt-1 truncate" title="${planningEscape((team.originalSkills || []).join(', '))}">
                ${(team.originalSkills || []).length ? planningEscape((team.originalSkills || []).join(' • ')) : 'Plano Skillgruppe'}
              </div>
              ${detailHtml}
            </div>
            <div class="text-[10px] font-bold border rounded-full px-2 py-1 ${statusClass}">
              ${isOk ? 'Schaffbar' : 'Risiko'}
            </div>
          </div>
        </div>

        <div class="p-4 space-y-4">
          <div class="grid grid-cols-2 gap-3">
            <div class="rounded-xl bg-gray-50 border border-gray-100 p-3">
              <div class="text-[9px] uppercase font-bold text-gray-400">Plano Produktivstunden</div>
              <div class="font-mono text-xl font-bold text-slate-700">${planningDecimalToTime(planoHours)}</div>
            </div>
            <div class="rounded-xl bg-emerald-50 border border-emerald-100 p-3">
              <div class="text-[9px] uppercase font-bold text-gray-400">Reststunden</div>
              <div class="font-mono text-xl font-bold text-modern-green">${planningDecimalToTime(remainingHours)}</div>
            </div>
          </div>

          <div class="grid grid-cols-2 gap-3">
            <label class="block">
              <span class="text-[10px] uppercase font-bold text-gray-400">Aktueller Backlog</span>
              <input type="number" min="0" step="1" value="${backlog}" data-planning-field="backlog" data-team-index="${index}" oninput="handleDayPlanningInput(${index})" class="mt-1 w-full border border-gray-200 rounded-xl px-3 py-2 font-mono font-bold text-slate-700 focus:ring-2 focus:ring-lidl-blue outline-none">
            </label>
            <label class="block">
              <span class="text-[10px] uppercase font-bold text-gray-400">Produktivität</span>
              <input type="number" min="0" step="0.1" value="${productivity}" data-planning-field="productivity" data-team-index="${index}" oninput="handleDayPlanningInput(${index})" class="mt-1 w-full border border-gray-200 rounded-xl px-3 py-2 font-mono font-bold text-slate-700 focus:ring-2 focus:ring-lidl-blue outline-none">
            </label>
          </div>

          <div class="grid grid-cols-3 gap-2 text-right">
            <div class="rounded-xl border border-blue-100 bg-blue-50 p-2">
              <div class="text-[9px] uppercase font-bold text-gray-400">Schaffbar</div>
              <div class="font-mono font-bold text-lidl-blue text-lg" data-planning-output="possible" data-team-index="${index}">${possibleCases}</div>
            </div>
            <div class="rounded-xl border border-emerald-100 bg-emerald-50 p-2">
              <div class="text-[9px] uppercase font-bold text-gray-400">Rest schaffbar</div>
              <div class="font-mono font-bold text-modern-green text-lg" data-planning-output="possibleRemaining" data-team-index="${index}">${possibleRemainingCases}</div>
            </div>
            <div class="rounded-xl border ${isOk ? 'border-emerald-100 bg-emerald-50' : 'border-red-100 bg-red-50'} p-2">
              <div class="text-[9px] uppercase font-bold text-gray-400">Gap</div>
              <div class="font-mono font-bold ${isOk ? 'text-modern-green' : 'text-red-600'} text-lg" data-planning-output="gap" data-team-index="${index}">${gap >= 0 ? '+' : ''}${gap}</div>
