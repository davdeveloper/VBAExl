  <div class="wrap">
    <h1>Master Knowledge</h1>
    <p class="subtitle">Google Docs zentral importieren, anspringen und wieder entfernen.</p>

    <div class="card">
      <label for="docUrl">Google Docs Link</label>
      <textarea id="docUrl" placeholder="https://docs.google.com/document/d/..."></textarea>
      <button id="importBtn" class="btn-primary" onclick="importDocument()">Importieren</button>
      <div id="status" class="status"></div>
    </div>

    <div class="card">
      <div class="list-head">
        <div class="list-title">Importierte Dokumente</div>
        <div id="count" class="count">0</div>
      </div>

      <input id="search" type="search" placeholder="Liste durchsuchen..." oninput="renderList()">
    </div>

    <div id="docList" class="doc-list"></div>
  </div>

  <script>
    let importedDocuments = [];
    let busy = false;

    document.addEventListener('DOMContentLoaded', function () {
      loadImportedDocuments();
    });

    function loadImportedDocuments() {
      setBusy(true);
      showStatus('Importierte Dokumente werden geladen...', 'info');

      google.script.run
        .withSuccessHandler(function (docs) {
          importedDocuments = docs || [];
          renderList();
          hideStatus();
          setBusy(false);
        })
        .withFailureHandler(function (err) {
          showStatus(getErrorMessage(err), 'error');
          setBusy(false);
        })
        .getImportedDocuments();
    }

    function importDocument() {
      const input = document.getElementById('docUrl');
      const value = input.value.trim();

      if (!value) {
        showStatus('Bitte zuerst einen Google Docs Link einfügen.', 'error');
        return;
      }

      setBusy(true);
      showStatus('Import läuft. Bei großen Dokumenten kann das kurz dauern...', 'info');

      google.script.run
        .withSuccessHandler(function (result) {
          input.value = '';

          let message = result && result.message ? result.message : 'Dokument wurde importiert.';

          if (result && result.warnings && result.warnings.length) {
            message += '<br><br><strong>Hinweis:</strong><br>' + result.warnings.map(escapeHtml).join('<br>');
            showStatus(message, 'warning');
          } else {
            showStatus(message, 'success');
          }

          loadImportedDocuments();
        })
        .withFailureHandler(function (err) {
          showStatus(getErrorMessage(err), 'error');
          setBusy(false);
        })
        .importDocumentFromLink(value);
    }

    function jumpToDocument(sourceId) {
      setBusy(true);
      showStatus('Bereich wird im Master-Dokument angesprungen...', 'info');

      google.script.run
        .withSuccessHandler(function () {
          hideStatus();
          setBusy(false);
        })
        .withFailureHandler(function (err) {
          showStatus(getErrorMessage(err), 'error');
          setBusy(false);
        })
        .jumpToImportedDocument(sourceId);
    }

    function removeDocument(sourceId, title) {
      const confirmed = confirm('Soll dieser importierte Bereich wirklich vollständig aus dem Master-Dokument entfernt werden?\n\n' + title);

      if (!confirmed) {
        return;
      }

      setBusy(true);
      showStatus('Bereich wird entfernt...', 'info');

      google.script.run
        .withSuccessHandler(function (docs) {
          importedDocuments = docs || [];
          renderList();
          showStatus('Bereich wurde entfernt.', 'success');
          setBusy(false);
        })
        .withFailureHandler(function (err) {
          showStatus(getErrorMessage(err), 'error');
          setBusy(false);
        })
        .removeImportedDocument(sourceId);
    }

    function renderList() {
      const list = document.getElementById('docList');
      const count = document.getElementById('count');
      const search = document.getElementById('search').value.trim().toLowerCase();

      const filtered = importedDocuments.filter(function (doc) {
        const haystack = [
          doc.title || '',
          doc.sourceId || '',
          doc.url || ''
        ].join(' ').toLowerCase();

        return !search || haystack.indexOf(search) !== -1;
      });

      count.textContent = filtered.length + ' von ' + importedDocuments.length;

      if (!filtered.length) {
        list.innerHTML = '<div class="empty">Keine passenden importierten Dokumente gefunden.</div>';
        return;
      }

      list.innerHTML = filtered.map(function (doc) {
        const title = doc.title || doc.sourceId;
        const importedAt = formatDate(doc.importedAt);
        const safeId = escapeAttr(doc.sourceId);
        const safeTitle = escapeHtml(title);
        const safeTitleAttr = escapeAttr(title);

        return `
          <div class="doc-item">
            <div class="doc-title">${safeTitle}</div>
            <div class="doc-meta">
              ID: ${escapeHtml(doc.sourceId || '')}
              ${importedAt ? '<br>Importiert: ' + escapeHtml(importedAt) : ''}
            </div>
            <div class="doc-actions">
              <button class="btn-light" onclick="jumpToDocument('${safeId}')">Anspringen</button>
              <button class="btn-danger" onclick="removeDocument('${safeId}', '${safeTitleAttr}')">Entfernen</button>
            </div>
          </div>
        `;
      }).join('');
    }

    function setBusy(value) {
      busy = value;
      document.getElementById('importBtn').disabled = value;

      const buttons = document.querySelectorAll('button');
      buttons.forEach(function (button) {
        button.disabled = value;
      });
    }

    function showStatus(message, type) {
      const status = document.getElementById('status');
      status.className = 'status show ' + (type || 'info');
      status.innerHTML = message;
    }

    function hideStatus() {
      const status = document.getElementById('status');
      status.className = 'status';
      status.innerHTML = '';
    }

    function getErrorMessage(err) {
      if (!err) {
        return 'Unbekannter Fehler.';
      }

      return escapeHtml(err.message || String(err));
    }

    function formatDate(value) {
      if (!value) {
        return '';
      }

      const date = new Date(value);

      if (isNaN(date.getTime())) {
        return value;
      }

      return date.toLocaleString('de-DE');
    }

    function escapeHtml(value) {
      return String(value || '')
        .replace(/&/g, '&amp;')
        .replace(/</g, '&lt;')
        .replace(/>/g, '&gt;')
        .replace(/"/g, '&quot;')
        .replace(/'/g, '&#039;');
    }

    function escapeAttr(value) {
      return escapeHtml(value).replace(/`/g, '&#096;');
    }
