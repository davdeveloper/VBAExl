/**
 * DIENSTPLANUNG MODULE
 * Enthält die Logik für den Dienstplanungs-Tab, Quoten-Checks, Speicherung und Löschung.

function getDienstplanData() {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const sheetDB = ss.getSheetByName('Data_DB');
    let employees = [];
    
    // 1. Mitarbeiter laden
    if (sheetDB && sheetDB.getLastRow() > 1) {
      const lastCol = sheetDB.getLastColumn();
      // Wir müssen sicherstellen, dass wir bis mindestens Spalte N (14) laden
      const colsToLoad = lastCol < 14 ? 14 : lastCol;
      
      const data = sheetDB.getRange(2, 1, sheetDB.getLastRow() - 1, colsToLoad).getValues();
      
      employees = data.map((row, index) => {
        const nachname = String(row[0] || "").trim();
        const vorname = String(row[1] || "").trim();
        if (!nachname) return null;
        
        // Spalte N ist Index 13 (0-basiert)
        // Standard auf 30 setzen, wenn leer oder 0
        let budget = 30;
        if (row[13] && !isNaN(row[13]) && Number(row[13]) > 0) {
            budget = Number(row[13]);
        }

        return {
          id: index + 2, // Zeilennummer in Excel
          nachname: nachname,
          vorname: vorname,
          fullName: `${nachname}, ${vorname}`, 
          tma: String(row[3] || ""), 
          team: String(row[4] || ""),
          vacationBudget: budget, // <--- NEU: Individueller Anspruch
          vacationCount: 0,
          vacations: []
        };
      }).filter(e => e !== null);
    }
    
    employees.sort((a, b) => a.nachname.localeCompare(b.nachname));
    
    // 2. Urlaubsdaten laden (Bleibt unverändert)
    const sheetPlan = ss.getSheetByName('Dienstplanung');
    if (sheetPlan && sheetPlan.getLastRow() > 1) {
      const lastColP = sheetPlan.getLastColumn();
      if (lastColP >= 1) {
         const colsToLoad = lastColP < 10 ? lastColP : 10; 
         const planData = sheetPlan.getRange(2, 1, sheetPlan.getLastRow() - 1, colsToLoad).getValues();
         
         const vacationCounts = {};
         const vacationRanges = {}; 
         
         planData.forEach((row, rowIdx) => {
            const rowRealIndex = rowIdx + 2; 
            const nameKey = String(row[0] || "").toLowerCase().trim();
            
            if (nameKey) {
               const daysCount = (row[7] && !isNaN(parseFloat(row[7]))) ? parseFloat(row[7]) : 0;
               if (!vacationCounts[nameKey]) vacationCounts[nameKey] = 0;
               vacationCounts[nameKey] += daysCount;

               const startRaw = row[3]; 
               const endRaw = row[4];   
               const typeRaw = String(row[5] || "Urlaub"); 
               const createdRaw = row[9]; 
               
               if (startRaw instanceof Date && endRaw instanceof Date) {
                   if (!vacationRanges[nameKey]) vacationRanges[nameKey] = [];
                   vacationRanges[nameKey].push({
                       rowId: rowRealIndex,
                       start: formatDateIso(startRaw),
                       end: formatDateIso(endRaw),
                       type: typeRaw,
                       days: daysCount, 
                       created: formatDateIso(createdRaw) 
                   });
               }
            }
         });
         
         employees.forEach(emp => {
            const keys = [
                emp.fullName.toLowerCase(),
                (emp.nachname + " " + emp.vorname).toLowerCase(),
                (emp.vorname + " " + emp.nachname).toLowerCase()
            ];
            for (const key of keys) {
                if (vacationCounts[key] !== undefined) emp.vacationCount = vacationCounts[key];
                if (vacationRanges[key] !== undefined) emp.vacations = emp.vacations.concat(vacationRanges[key]);
                if (vacationCounts[key] !== undefined) break;
            }
         });
      }
    }
    return employees;
  } catch (e) {
    throw new Error("Server Fehler: " + e.message);
  }
}

function getHolidaysFromSheet() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName('Betriebsferien');
  const holidays = {};
  if (!sheet) return holidays;
  const lastRow = sheet.getLastRow();
  if (lastRow < 4) return holidays;
  const data = sheet.getRange(4, 2, lastRow - 3, 2).getValues();
  data.forEach(row => {
     const dateRaw = row[0];
     const name = String(row[1]);
     if (dateRaw instanceof Date && name) {
        const iso = formatDateIso(dateRaw);
        if (iso) holidays[iso] = name;
     }
  });
  return holidays;
}

function getEmployeeDetails(fullNameLabel) {
   const ss = SpreadsheetApp.getActiveSpreadsheet();
   const hours = getContractHours(fullNameLabel);
   let saldo = 0;
   try {
     const sheetDB = ss.getSheetByName('Data_DB');
     const sheetAuftrag = ss.getSheetByName('Auftrag');
     if (sheetDB) {
        const dbData = sheetDB.getDataRange().getValues();
        const parts = fullNameLabel.split(',');
        const sNachname = parts[0].trim().toLowerCase();
        const sVorname = parts.length > 1 ? parts[1].trim().toLowerCase() : "";
        let essDecimal = 0;
        for(let i=1; i<dbData.length; i++) {
           const n = String(dbData[i][0]).trim().toLowerCase();
           const v = String(dbData[i][1]).trim().toLowerCase();
           if (n === sNachname && v === sVorname) {
              essDecimal = timeToDecimal(dbData[i][5]);
              break;
           }
        }
        let plannedDecimal = 0;
        if (sheetAuftrag && sheetAuftrag.getLastRow() > 1) {
           const aData = sheetAuftrag.getRange(2, 1, sheetAuftrag.getLastRow()-1, 6).getValues();
           aData.forEach(row => {
              const aN = String(row[0]).trim().toLowerCase();
              const aV = String(row[1]).trim().toLowerCase();
              if (aN === sNachname && aV === sVorname) {
                 plannedDecimal += timeToDecimal(row[3]);
              }
           });
        }
        saldo = essDecimal - plannedDecimal;
        saldo = Math.round(saldo * 100) / 100;
     }
   } catch(e) { saldo = 0; }
   return { hours: hours, saldo: saldo };
}

function getContractHours(fullName) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName('Schichtstunden');
  if (!sheet) {
      sheet = ss.insertSheet('Schichtstunden');
      sheet.appendRow(['Mitarbeiter', 'Stunden']);
      return "08:00";
  }
  const data = sheet.getDataRange().getValues();
  const search = fullName.toLowerCase().replace(',', '').replace(/\s+/g, '');
  for (let i = 1; i < data.length; i++) {
     const rowName = String(data[i][0]).toLowerCase().replace(',', '').replace(/\s+/g, '');
     if (rowName === search || rowName.includes(search) || search.includes(rowName)) {
         let val = data[i][1];
         if (val instanceof Date) {
             const h = String(val.getHours()).padStart(2,'0');
             const m = String(val.getMinutes()).padStart(2,'0');
             val = `${h}:${m}`;
         }
         return val ? String(val) : "08:00";
     }
  }
  return "08:00"; 
}

  if (sheetKont) { quotaData = sheetKont.getDataRange().getValues(); headerRow = quotaData[1]; }
  let plannedVacations = [];
  if (sheetPlan && sheetPlan.getLastRow() > 1) plannedVacations = sheetPlan.getRange(2, 1, sheetPlan.getLastRow()-1, 5).getValues();
  let targetRowIndex = -1, groupTmas = [tmaName.toLowerCase().trim()]; 
  if (quotaData.length > 2) {
      const searchTma = tmaName.toLowerCase().trim();
      for(let r=2; r<quotaData.length; r++) {
         const cellContent = String(quotaData[r][0]); 
         const parts = cellContent.split('/').map(s => s.trim());
         const partsLower = parts.map(s => s.toLowerCase());
         if (partsLower.some(p => p === searchTma || p.includes(searchTma) || searchTma.includes(p))) { targetRowIndex = r; groupTmas = partsLower; break; }
      }
  }
  const empLower = employeeName ? employeeName.toLowerCase() : "";
  weeksToCheck.forEach(weekInfo => {
      const kw = weekInfo.kw, year = weekInfo.year;
      let allowed = 0, limitFound = false, colIndex = -1;
      for(let c=0; c<headerRow.length; c++) {
         const val = String(headerRow[c]);
         if (val == kw || val === `KW ${kw}` || val === `KW${kw}`) { colIndex = c; break; }
      }
      if (colIndex > -1 && targetRowIndex > -1) { allowed = Number(quotaData[targetRowIndex][colIndex]); if(isNaN(allowed)) allowed = 0; limitFound = true; } else { allowed = 99; }
      let taken = 0;
      const kwStartEnd = getDateRangeOfWeek(kw, year);
      plannedVacations.forEach(row => {
          const pName = String(row[0]).toLowerCase(), pTma = String(row[1]).toLowerCase().trim(); 
          const pStart = row[3], pEnd = row[4];
          if (pStart instanceof Date && pEnd instanceof Date) {
              if (pStart <= kwStartEnd.end && pEnd >= kwStartEnd.start) {
                  if (empLower && (pName === empLower || pName.includes(empLower) || empLower.includes(pName))) {
                      ranges.forEach(newR => {
                          const nStart = new Date(newR.start), nEnd = new Date(newR.end);
                          if (nStart <= pEnd && nEnd >= pStart) result.userCollision = true;
                      });
                  }
                  if (pTma) {
                      const belongsToGroup = groupTmas.some(gTma => pTma.includes(gTma) || gTma.includes(pTma));
                      if (belongsToGroup) taken++;
                  }
              }
          }
      });
      let status = 'ok';
      if (limitFound && taken >= allowed) status = 'critical';
      else if (!limitFound) status = 'unknown';
      if (status === 'critical') result.overallStatus = 'critical';
      result.details.push({ kw: kw, year: year, allowed: limitFound ? allowed : '-', taken: taken, status: status });
  });
  result.details.sort((a,b) => (a.year - b.year) || (a.kw - b.kw));
  return result;
}

function saveVacationBatch(payload) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName('Dienstplanung');
  if (!sheet) sheet = ss.insertSheet('Dienstplanung');
  let sheetAuftrag = ss.getSheetByName('Auftrag');
  if (!sheetAuftrag && payload.entries.some(e => e.type === 'Freizeitausgleich')) {
      sheetAuftrag = ss.insertSheet('Auftrag');
      sheetAuftrag.appendRow(['Name', 'Vorname', 'Datum', 'Stunden', 'Notizen', 'Erstellt von']);
  }
  if (sheet.getLastRow() === 0) sheet.appendRow(['Mitarbeiter', 'TMA', 'Bereich', 'Start', 'Ende', 'Art', 'KW', 'Tage', 'Eingereicht', 'Zeitstempel']);
  const entries = payload.entries; 
  const submitted = new Date(payload.submitted);
  const now = new Date();
  const dailyHoursStr = payload.dailyHours || "00:00"; 
  const rowsToAdd = [];
  entries.forEach(entry => {
      const sParts = entry.start.split('-');
      const eParts = entry.end.split('-');
      const start = new Date(sParts[0], sParts[1]-1, sParts[2], 12, 0, 0);
      const end = new Date(eParts[0], eParts[1]-1, eParts[2], 12, 0, 0);
      let workDays = 0;
      let cur = new Date(start.getTime());
      while (cur <= end) {
        const day = cur.getDay();
        if (entry.type === 'Freizeitausgleich' || entry.type === 'Sonderurlaub') { workDays++; } 
        else { if (day !== 0 && day !== 6) workDays++; }
        cur.setDate(cur.getDate() + 1);
      }
      const kw = getWeekNumber(start);
      if (entry.type === 'Freizeitausgleich' && workDays > 0) {
          const decimalDaily = timeToDecimal(dailyHoursStr); 
          const totalDecimal = decimalDaily * workDays;
          const totalTimeStr = decimalToTime(totalDecimal); 
          const sStr = formatDateDe(start);
          const eStr = formatDateDe(end);
          const dateDisplay = (sStr === eStr) ? sStr : `${sStr} - ${eStr}`;
          const nameParts = payload.name.split(',');
          const nachname = nameParts[0].trim();
          const vorname = nameParts.length > 1 ? nameParts[1].trim() : "";
          if(sheetAuftrag) {
             sheetAuftrag.appendRow([nachname, vorname, dateDisplay, totalTimeStr, "FZA", "Dienstplanung"]);
             sheetAuftrag.getRange(sheetAuftrag.getLastRow(), 4).setNumberFormat("@");
          }
          updateAbbauStats(start, totalDecimal, 'add', payload.team); 
      }
      rowsToAdd.push([payload.name, payload.tma, payload.team, start, end, entry.type, kw, workDays, submitted, now]);
  });
  if (rowsToAdd.length > 0) {
      const startRow = sheet.getLastRow() + 1;
      sheet.getRange(startRow, 1, rowsToAdd.length, 10).setValues(rowsToAdd);
      sheet.getRange(startRow, 4, rowsToAdd.length, 2).setNumberFormat("dd.MM.yyyy"); 
      sheet.getRange(startRow, 9, rowsToAdd.length, 2).setNumberFormat("dd.MM.yyyy"); 
  }
  return true;
}

function deleteVacationEntry(rowId) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName('Dienstplanung');

      const sheetAuftrag = ss.getSheetByName('Auftrag');
      if (sheetAuftrag) {
          const sStr = formatDateDe(start);
          const eStr = formatDateDe(end);
          const dateSearch = (sStr === eStr) ? sStr : `${sStr} - ${eStr}`;
          const nameParts = nameFull.split(',');
          const nachname = nameParts[0].trim().toLowerCase();
          const vorname = nameParts.length > 1 ? nameParts[1].trim().toLowerCase() : "";
          const dataA = sheetAuftrag.getDataRange().getValues();
          for (let i = dataA.length - 1; i >= 1; i--) {
              const rowA = dataA[i];
              const nA = String(rowA[0]).trim().toLowerCase();
              const vA = String(rowA[1]).trim().toLowerCase();
              let dateInAuftragRaw = rowA[2];
              let dateInAuftragStr = "";
              if (dateInAuftragRaw instanceof Date) { dateInAuftragStr = formatDateDe(dateInAuftragRaw); }

function formatDateIso(dateObj) {
    if (!dateObj) return null;
    let d = dateObj;
    if (!(d instanceof Date)) d = new Date(dateObj); 
    if (isNaN(d.getTime())) return null;
    const y = d.getFullYear();
    const m = String(d.getMonth() + 1).padStart(2, '0');
    const da = String(d.getDate()).padStart(2, '0');
    return `${y}-${m}-${da}`;
}
function formatDateDe(dateObj) {
    if (!dateObj) return "";
    let d = dateObj;
    if (!(d instanceof Date)) d = new Date(dateObj);
    if (isNaN(d.getTime())) return String(dateObj);
    const day = String(d.getDate()).padStart(2, '0');
    const month = String(d.getMonth() + 1).padStart(2, '0');
    const year = d.getFullYear();
    return `${day}.${month}.${year}`;
}
function getWeekNumber(d) {
    d = new Date(Date.UTC(d.getFullYear(), d.getMonth(), d.getDate()));
    d.setUTCDate(d.getUTCDate() + 4 - (d.getUTCDay()||7));
    var yearStart = new Date(Date.UTC(d.getUTCFullYear(),0,1));
    return Math.ceil((((d - yearStart) / 86400000) + 1)/7);
}
function getWeeksInRange(startDate, endDate) {
    let weeks = [];
    let current = new Date(startDate);
    const day = current.getDay() || 7; 
    if (day !== 1) current.setHours(-24 * (day - 1));
    while (current <= endDate) {
        const kw = getWeekNumber(current);
        const year = current.getFullYear();
        if (!weeks.some(w => w.kw === kw && w.year === year)) { weeks.push({ kw: kw, year: year }); }
        current.setDate(current.getDate() + 7);
    }
    const endKw = getWeekNumber(endDate);
    const endYear = endDate.getFullYear();
    if (!weeks.some(w => w.kw === endKw && w.year === endYear)) { weeks.push({ kw: endKw, year: endYear }); }
    return weeks;
}
function getDateRangeOfWeek(w, y) {
    var simple = new Date(y, 0, 1 + (w - 1) * 7);
    var dow = simple.getDay();
    var ISOweekStart = simple;
    if (dow <= 4) ISOweekStart.setDate(simple.getDate() - simple.getDay() + 1);
    else ISOweekStart.setDate(simple.getDate() + 8 - simple.getDay());
    const end = new Date(ISOweekStart);
    end.setDate(end.getDate() + 6);
    return { start: ISOweekStart, end: end };
}

          alert("Bitte Mitarbeiter wählen und mindestens einen vollständigen Zeitraum angeben.");
          return;
      }
      const type = document.getElementById('vac-type').value;
      let dailyHours = "00:00";
      if (type === 'Freizeitausgleich') {
          dailyHours = document.getElementById('vac-daily-hours').value;
          if (!dailyHours || dailyHours === "Lade..." || !dailyHours.includes(":")) {
               alert("Bitte gültige Vertragsstunden (hh:mm) angeben.");
               return;
          }
      }
      const btn = document.getElementById('btn-vac-save');
      const oldText = btn.innerText;
      btn.innerText = "Speichere...";
      btn.disabled = true;
      const entries = ranges.map(r => ({ start: r.start, end: r.end, type: type }));
      const payload = {
          name: vacSelectedEmp.label,
          tma: vacSelectedEmp.tma,
          team: vacSelectedEmp.team,
          entries: entries,
          submitted: document.getElementById('vac-submitted').value,
          dailyHours: dailyHours
      };
      google.script.run.withSuccessHandler(() => {
          btn.innerText = "Gespeichert!";
          setTimeout(() => {
              closeVacationModal();
              btn.innerText = oldText;
              btn.disabled = false;
              initDienstplanView(); 
              if(typeof loadData === 'function') loadData(); 
          }, 800);
      }).withFailureHandler(err => {
          alert("Fehler: " + err.message);
          btn.innerText = oldText;
          btn.disabled = false;
      }).saveVacationBatch(payload);
  }

  function updateBudget(empId, val) {
     if (!val || val < 0) return;
     val = parseInt(val);
     
     const emp = rawDienstplanData.find(e => e.id == empId);
     if (emp) {
         emp.vacationBudget = val; 
     }

     const feedback = document.getElementById(`save-feedback-${empId}`);
     if(feedback) {
         feedback.innerHTML = '<i class="fa-solid fa-spinner fa-spin"></i> Speichere...';
         feedback.classList.remove('opacity-0', 'text-green-600');
         feedback.classList.add('text-gray-400');
     }
     
     const display = document.getElementById(`budget-display-${empId}`);
     const bar = document.getElementById(`budget-bar-${empId}`);
     
     if (emp) {
         const usedDays = emp.vacations
              .filter(v => v.type === 'Urlaub (genehmigt)')
              .reduce((sum, v) => sum + (v.days || 0), 0);
         
         const percent = Math.min((usedDays / val) * 100, 100);
         
         if (display) display.innerText = val;
         
         const counterSpan = display.closest('.text-xl'); 
         if (counterSpan) {
             if (usedDays > val) {
                 counterSpan.classList.remove('text-modern-green');
                 counterSpan.classList.add('text-red-500');
                 if(bar) { bar.classList.remove('bg-lidl-blue'); bar.classList.add('bg-red-500'); }
             } else {
                 counterSpan.classList.add('text-modern-green');
                 counterSpan.classList.remove('text-red-500');
                 if(bar) { bar.classList.add('bg-lidl-blue'); bar.classList.remove('bg-red-500'); }
             }
         }
         if (bar) bar.style.width = percent + "%";
     }
     
     google.script.run.withSuccessHandler(() => {
         if(feedback) {
             feedback.innerHTML = '<i class="fa-solid fa-check"></i> Gespeichert';
             feedback.classList.remove('text-gray-400');
             feedback.classList.add('text-green-600');
             setTimeout(() => feedback.classList.add('opacity-0'), 2000);

     }).updateEmployeeBudget(empId, val);
  }

  window.filterAndRenderDienstplan = function(onlyBody = false) {
     if (!rawDienstplanData.length && !dienstplanInitDone) return;
     const startFilter = document.getElementById('filter-start') ? document.getElementById('filter-start').value : null;
     const endFilter = document.getElementById('filter-end') ? document.getElementById('filter-end').value : null;

     let filtered = rawDienstplanData.filter(d => {
         const teamMatch = selectedTeams.size === 0 || selectedTeams.has(d.team);
         const tmaMatch = selectedTmas.size === 0 || selectedTmas.has(d.tma);
         
         let nameMatch = true;
         if (currentNameFilter) {
             nameMatch = d.fullName.toLowerCase().includes(currentNameFilter);
         }

         let dateMatch = true;
         if (startFilter && endFilter) {
             if (d.vacations.length === 0) {
                 dateMatch = false; 
             } else {
                 const fS = new Date(startFilter);
                 const fE = new Date(endFilter);
                 const hasOverlap = d.vacations.some(v => {
                     const vS = new Date(v.start);
                     const vE = new Date(v.end);
                     return vS <= fE && vE >= fS;
                 });
                 dateMatch = hasOverlap;
             }
         }
         return teamMatch && tmaMatch && dateMatch && nameMatch;
     });
     
     if (currentDaySort === 'asc') {
         filtered.sort((a,b) => getApprovedDays(a) - getApprovedDays(b));
     } else if (currentDaySort === 'desc') {
         filtered.sort((a,b) => getApprovedDays(b) - getApprovedDays(a));
     }
     
     updateGlobalStats(filtered);
     
     if (onlyBody) {
         renderCalendarBody(filtered);
     } else {
         renderCalendarBody(filtered);
     }
  }

  function updateGlobalStats(filteredData) {
      const statPlanned = document.getElementById('stat-total-planned');
      const statCapacity = document.getElementById('stat-total-capacity');
      const statBar = document.getElementById('stat-bar-capacity');
      const statPast = document.getElementById('stat-past');
      const statFuture = document.getElementById('stat-future');
      
      if(!statPlanned) return; 

      // ÄNDERUNG: Berechnung der Gesamtkapazität basierend auf individuellem Budget
      const totalCapacity = filteredData.reduce((sum, emp) => sum + (emp.vacationBudget || 30), 0);
      
      let totalPlanned = 0;
      let pastDays = 0;
      let futureDays = 0;
      const today = new Date();
      today.setHours(0,0,0,0);

      filteredData.forEach(emp => {
          emp.vacations.forEach(v => {
              if (v.type === 'Urlaub (genehmigt)') {
                  totalPlanned += (v.days || 0);
                  
                  let current = new Date(v.start);
                  const end = new Date(v.end);
                  
                  while (current <= end) {
                      const d = current.getDay();
                      if (d !== 0 && d !== 6) { 
                         if (current < today) {
                             pastDays++;
                         } else {
                             futureDays++;
                         }
                      }
                      current.setDate(current.getDate() + 1);
                  }
              }
          });
      });

      statPlanned.innerText = totalPlanned;
      statCapacity.innerText = totalCapacity;
      statPast.innerText = pastDays;
      statFuture.innerText = futureDays;

      const percent = totalCapacity > 0 ? (totalPlanned / totalCapacity) * 100 : 0;
      statBar.style.width = Math.min(percent, 100) + "%";
      
      if(percent > 100) statBar.classList.replace('bg-lidl-blue', 'bg-red-500');
      else statBar.classList.replace('bg-red-500', 'bg-lidl-blue');
  }

  function renderCalendarBody(data) {
      const body = document.getElementById('calendar-body');
      if (!body) return;
      
      if (data.length === 0) {
          body.innerHTML = `<tr><td class="p-8 text-center text-gray-400" colspan="100%">Keine Daten verfügbar.</td></tr>`;
          return;
      }

      const daysInMonth = getDaysInMonth(YEAR_VIEW, currentViewMonth);
      const formatDate = (dateObj) => {
             const y = dateObj.getFullYear();
             const m = String(dateObj.getMonth() + 1).padStart(2, '0');
             const d = String(dateObj.getDate()).padStart(2, '0');
             return `${y}-${m}-${d}`;
      };

      const rowsHTML = data.map(emp => {
             const vacMap = new Map(); 
             if (emp.vacations && emp.vacations.length > 0) {
                 emp.vacations.forEach(range => {
                     let current = new Date(range.start);
                     const end = new Date(range.end);
                     const viewStart = daysInMonth[0];
                     const viewEnd = daysInMonth[daysInMonth.length - 1];
                     if (end < viewStart || current > viewEnd) return; 
                     while (current <= end) {
                         if (current >= viewStart && current <= viewEnd) {
                             vacMap.set(formatDate(current), range.type);
                         }
                         current.setDate(current.getDate() + 1);
                     }
                 });
             }

             const cellsHTML = daysInMonth.map(d => {
                 const dateStr = formatDate(d);
                 const isWeekend = d.getDay() === 0 || d.getDay() === 6;
                 const isoDate = formatDateIsoSimple(d);
                 const isHoliday = !!globalHolidays[isoDate];
                 let bgClass = isWeekend ? 'bg-gray-100 text-gray-400' : 'bg-white';
                 if (isHoliday) bgClass = 'bg-purple-50'; 
                 const borderClass = d.getDate() === 1 ? 'border-l-2 border-l-gray-300' : 'border-l border-gray-100';
                 const vacType = vacMap.get(dateStr);
                 const isVacation = !!vacType && !isWeekend && !isHoliday; 
                 let content = '';
                 let cellBg = bgClass;
                 if (isVacation) {
                     if (vacType === 'Freizeitausgleich') {
                         content = '<i class="fa-solid fa-business-time text-blue-500"></i>';
                         cellBg = 'bg-blue-50';
                     } else if (vacType === 'Sonderurlaub') {
                         content = '<i class="fa-solid fa-star text-purple-500"></i>';
                         cellBg = 'bg-purple-50';
                     } else {
                         content = '<i class="fa-solid fa-umbrella-beach text-orange-500"></i>';
                         cellBg = 'bg-orange-50';
                     }
                 }
                 return `<td class="${cellBg} ${borderClass} border-b border-gray-50 text-center p-0 relative group/cell cursor-pointer hover:bg-blue-200 transition-colors">${content}</td>`;
             }).join('');

             const totalUrlaub = emp.vacations.filter(v => v.type === 'Urlaub (genehmigt)').reduce((sum, v) => sum + (v.days||0), 0);
             const budget = emp.vacationBudget || 30;
             let antraegeClass = "text-modern-green";
             if (totalUrlaub > budget || (totalUrlaub > 0 && totalUrlaub < 5)) antraegeClass = "text-red-600";
             
             // ÄNDERUNG: Anzeige von IST / MAX (z.B. 27 / 30)
             const displayCount = (totalUrlaub > 0 ? totalUrlaub : 0) + " / " + budget;

             return `<tr onclick="openPersonModal(${emp.id})" class="hover:bg-blue-50/30 transition-colors h-10 group/row cursor-pointer">
                 <td class="sticky left-0 z-10 bg-white border-r border-gray-200 px-4 font-medium text-slate-700 truncate max-w-[200px] border-b border-gray-50 group-hover/row:bg-blue-50">${emp.nachname}, ${emp.vorname}</td>
                 <td class="sticky left-[200px] z-10 bg-white border-r border-gray-200 px-4 text-gray-500 truncate max-w-[150px] border-b border-gray-50 text-xs group-hover/row:bg-blue-50">${emp.tma}</td>
                 <td class="sticky left-[350px] z-10 bg-white border-r border-gray-200 px-4 text-gray-500 truncate max-w-[120px] border-b border-gray-50 text-xs group-hover/row:bg-blue-50">${emp.team}</td>
                 <td class="sticky left-[470px] z-10 bg-white border-r border-gray-200 px-2 text-center border-b border-gray-50 text-xs font-bold ${antraegeClass} group-hover/row:bg-blue-50">${displayCount}</td>
                 ${cellsHTML}
             </tr>`;
      }).join('');
      body.innerHTML = rowsHTML;
      
      const loader = document.getElementById('calendar-loader');
      if(loader) loader.classList.add('hidden');
  }

  function getDaysInMonth(year, monthIndex) {
      const date = new Date(year, monthIndex, 1);
      const days = [];
      while (date.getMonth() === monthIndex) {
          days.push(new Date(date));
          date.setDate(date.getDate() + 1);
      }
      return days;
  }

     const y = d.getFullYear();
     const m = String(d.getMonth() + 1).padStart(2, '0');
     const day = String(d.getDate()).padStart(2, '0');
     return `${y}-${m}-${day}`;
  }

/**
 * Ermittelt Empfänger (To) und Vertretung/CC (CC) basierend auf dem Mitarbeiter.
 */
function getAbsenceRecipients(empId) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheetDB = ss.getSheetByName('Data_DB');
  const sheetTS = ss.getSheetByName('Tagessteuerung');
  
  let result = { to: "", cc: "", tmaList: [] };
  
  // 1. TMA Liste und Bereiche laden
  const tsData = sheetTS.getRange(2, 1, sheetTS.getLastRow() - 1, 4).getValues(); 
  
  const allTmas = tsData.map(row => ({
    email: String(row[0]).trim(),
    name: String(row[2]).trim(),
    area: String(row[3]).trim()
  })).filter(t => t.email && t.name);

  result.tmaList = allTmas;

  // 2. Mitarbeiter suchen und dessen TMA finden
  const rowIndex = Number(empId);
  if (rowIndex > 1 && sheetDB) {
      const tmaNameOfEmp = sheetDB.getRange(rowIndex, 4).getValue();
      
      if (tmaNameOfEmp) {
         const cleanTmaName = String(tmaNameOfEmp).trim().toLowerCase();
         
         // 3. TMA in TS Liste finden (TO Empfänger)
         const recipient = allTmas.find(t => t.name.toLowerCase() === cleanTmaName);
         
         if (recipient) {
            result.to = recipient.email;
            
            // 4. CC finden (Bereichs-Vertretung)
            if (recipient.area) {
               const ccCandidate = allTmas.find(t => 
                  t.area.toLowerCase() === recipient.area.toLowerCase() && 
                  t.email !== recipient.email
               );
               if (ccCandidate) {
                  result.cc = ccCandidate.email;
               }
            }
         }
      }
  }
  
  return result;
}

/**
 * ReminderLogic.gs
 * Logik für den Versand von Erinnerungs-Mails an TMAs bei veralteten Daten.
 */

function getReminderDraft() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheetDB = ss.getSheetByName('Data_DB');
  const sheetTS = ss.getSheetByName('Tagessteuerung');

  // 1. TMA E-Mail Map erstellen (Name -> Email)
  const tsData = sheetTS.getRange(2, 1, sheetTS.getLastRow() - 1, 3).getValues();
  const tmaMap = new Map();
   
  tsData.forEach(row => {
    const email = String(row[0]).trim();
    const name = String(row[2]).trim().toLowerCase(); 
    if (email && name) {
      tmaMap.set(name, email);
    }
  });

  // 2. Veraltete Einträge finden (Austritt beachten!)
  const dbData = sheetDB.getRange(2, 1, sheetDB.getLastRow() - 1, 13).getValues(); 
  const today = new Date();
  today.setHours(0,0,0,0);
   
  const tmasToNotify = new Map(); // Speichert Email -> Name

  dbData.forEach(row => {
    const tmaName = String(row[3]).trim().toLowerCase();
    const lastUpdateRaw = row[6];
    const notiz = String(row[11] || "").trim(); // Spalte L (Index 11)
    const austrittRaw = row[12]; // Spalte M (Index 12)

    // --- 1. Austritts-Prüfung (Neu!) ---
    if (austrittRaw) {
       let aDate = null;
       if (austrittRaw instanceof Date) {
           aDate = austrittRaw;
       } else if (typeof austrittRaw === 'string' && austrittRaw.trim() !== "") {
           const parts = austrittRaw.trim().match(/(\d{1,2})\.(\d{1,2})\.(\d{4})/);
           if(parts) aDate = new Date(parts[3], parts[2]-1, parts[1]);
       }
       if (aDate) {
           aDate.setHours(0,0,0,0);
           if (aDate <= today) return; // Kollege hat das Unternehmen verlassen -> Überspringen!
       }
    }

    // --- 2. Notiz Prüfung ---
    if (notiz !== "") return; // Wenn etwas pausiert ist -> Überspringen
    
    // --- 3. Datums Prüfung ---
    let isOutdated = false;

    // NEU: Wenn komplett leer oder "-", dann ignorieren (Neu/Temporär)
    const lastUpdateStr = String(lastUpdateRaw).trim();
    if (!lastUpdateRaw || lastUpdateStr === "" || lastUpdateStr === "-") {
       return; 
    }

    if (lastUpdateRaw instanceof Date && !isNaN(lastUpdateRaw)) {
      const diffTime = Math.abs(today.getTime() - lastUpdateRaw.getTime());
      const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24)); 
      if (diffDays > 7) isOutdated = true;
    } 
    else if (typeof lastUpdateRaw === 'string' && lastUpdateStr !== "") {
      const parts = lastUpdateStr.split('.');
      if (parts.length === 3) {
        const parsedDate = new Date(parts[2], parts[1] - 1, parts[0]);
        if (!isNaN(parsedDate)) {
           const diffTime = today.getTime() - parsedDate.getTime();
           const diffDays = diffTime / (1000 * 60 * 60 * 24);
           if (diffDays > 7) isOutdated = true;
        }
      }
    } 

    // --- Sammeln ---
    if (isOutdated && tmaMap.has(tmaName)) {
       const email = tmaMap.get(tmaName);
       tmasToNotify.set(email, String(row[3]).trim()); // Schönen Namen merken
    }
  });

/**
 * Ermittelt Empfänger (To) und Vertretung/CC (CC) basierend auf dem Mitarbeiter.
 */
function getEarlyLeaveRecipients(empId) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheetDB = ss.getSheetByName('Data_DB');
  const sheetTS = ss.getSheetByName('Tagessteuerung');
  
  let result = { to: "", cc: "", tmaList: [] };
  
  // 1. TMA Liste und Bereiche laden
  const tsData = sheetTS.getRange(2, 1, sheetTS.getLastRow() - 1, 4).getValues(); 
  
  const allTmas = tsData.map(row => ({
    email: String(row[0]).trim(),
    name: String(row[2]).trim(),
    area: String(row[3]).trim()
  })).filter(t => t.email && t.name);

  result.tmaList = allTmas;

  // 2. Mitarbeiter suchen und dessen TMA finden
  const rowIndex = Number(empId);
  if (rowIndex > 1 && sheetDB) {
      const tmaNameOfEmp = sheetDB.getRange(rowIndex, 4).getValue();
      
      if (tmaNameOfEmp) {
         const cleanTmaName = String(tmaNameOfEmp).trim().toLowerCase();
         
         // 3. TMA in TS Liste finden (TO Empfänger)
         const recipient = allTmas.find(t => t.name.toLowerCase() === cleanTmaName);
         
         if (recipient) {
            result.to = recipient.email;
            
            // 4. CC finden (Bereichs-Vertretung)
            if (recipient.area) {
               const ccCandidate = allTmas.find(t => 
                  t.area.toLowerCase() === recipient.area.toLowerCase() && 
                  t.email !== recipient.email
               );
               if (ccCandidate) {
                  result.cc = ccCandidate.email;
               }
            }
         }
      }
  }
  
  return result;
}

/**
 * MissingColleagueLogic.gs
 * Logik für das Versenden von Meldungen über fehlende Mitarbeiter.
 */

/**
 * Ermittelt Empfänger (To) und Vertretung/CC (CC) basierend auf dem Mitarbeiter.
 */
function getMissingColleagueRecipients(empId) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheetDB = ss.getSheetByName('Data_DB');
  const sheetTS = ss.getSheetByName('Tagessteuerung');
  
  let result = { to: "", cc: "", tmaList: [] };
  
  // 1. TMA Liste und Bereiche laden
  const tsData = sheetTS.getRange(2, 1, sheetTS.getLastRow() - 1, 4).getValues(); 
  
  const allTmas = tsData.map(row => ({
    email: String(row[0]).trim(),
    name: String(row[2]).trim(),
    area: String(row[3]).trim()
  })).filter(t => t.email && t.name);

  result.tmaList = allTmas;

  // 2. Mitarbeiter suchen und dessen TMA finden
  const rowIndex = Number(empId);
  if (rowIndex > 1 && sheetDB) {
      const tmaNameOfEmp = sheetDB.getRange(rowIndex, 4).getValue();
      
      if (tmaNameOfEmp) {
         const cleanTmaName = String(tmaNameOfEmp).trim().toLowerCase();
         
         const recipient = allTmas.find(t => t.name.toLowerCase() === cleanTmaName);
         
         if (recipient) {
            result.to = recipient.email;
            
            if (recipient.area) {
               const ccCandidate = allTmas.find(t => 
                  t.area.toLowerCase() === recipient.area.toLowerCase() && 
                  t.email !== recipient.email
               );
               if (ccCandidate) {
                  result.cc = ccCandidate.email;
               }
            }
         }
      }
  }
  
  return result;
}

 * Sendet die "Fehlender Kollege" E-Mail.
 * PRODUKTIV-MODUS
 */
function sendMissingColleagueEmail(payload) {
  try {
    const { 
      empName,       
      recipientEmail, 
      ccEmail,         
      vertretungEmail
    } = payload;

    const subject = `Fehlende/-r Kollege-/in - ${empName}`;

    // --- HTML BODY AUFBAU ---
    let body = `<p style="font-family: sans-serif; font-size: 14px;">Hallo zusammen,</p>
    
    <p style="font-family: sans-serif; font-size: 14px;">Folgende/-r Kollege/-in ist bisher nicht zum Dienst erschienen - <strong>${empName}</strong></p>
    
    <p style="font-family: sans-serif; font-size: 14px;">Eine Meldung liegt nicht vor.</p>`;

    // --- CC LOGIK ---
    let ccList = [];
    if (ccEmail) ccList.push(ccEmail);             
    if (vertretungEmail) ccList.push(vertretungEmail); 
    
    // Statische CCs (Verteiler)
    const staticCCs = [
      "tagessteuerung.lks@lidl.de",
      "dienstplanung.lks@lidl.de",
      "julian.leder@lidl.de",
      "sven.liedtke@lidl.de",
      "diana.wurl@lidl.de",
      "marion.pietrowski@lidl.de",
      "ulrich-markus.focke@lidl.de",
      "gregor.kuehn@lidl.de",
      "sekretariat.lks@lidl.de"
    ];

    const allCCs = [...new Set([...ccList, ...staticCCs])];
    const finalCCString = allCCs.join(",");

    const senderAlias = 'tagessteuerung.lks@lidl.de';
    
    const mailOptions = {
      htmlBody: body,
      from: senderAlias,
      cc: finalCCString 
    };

    // SENDEN AN ECHTEN EMPFÄNGER
    GmailApp.sendEmail(recipientEmail, subject, "", mailOptions);
    
    return { success: true };

  } catch (e) {
    Logger.log("Fehler beim Mailversand (MissingColleague): " + e.toString());
    throw new Error("Fehler beim Senden: " + e.message);
  }
}
