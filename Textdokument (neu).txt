function getTagesplanungData() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const today = new Date();
  today.setHours(0, 0, 0, 0);

  ensureTagesplanungSheets_();

  const capacity = getCapacityOverviewData();
  const productivityMap = getTagesplanungProductivityMap_();
  const savedMap = getTagesplanungSavedMap_(today);
  const dbBacklogMap = getTagesplanungLatestBacklogMap_(today);

  const teams = TAGESPLANUNG_TEAM_ORDER.map(teamName => {
    const capacityArea = capacity.areas.find(a => a.area === teamName) || {};
    const saved = savedMap[teamName] || {};
    const dbBacklog = dbBacklogMap[teamName];

    const totalHours = Number(capacityArea.totalHours || 0);
    const remainingHours = Number(capacityArea.remainingHours || 0);
    const usedHours = Number(capacityArea.usedHours || 0);
    const productivity = saved.productivity !== undefined && saved.productivity !== ''
      ? Number(saved.productivity)
      : Number(productivityMap[teamName] || 5);

    const backlog = saved.backlog !== undefined && saved.backlog !== ''
      ? Number(saved.backlog)
      : Number(dbBacklog || 0);

    const manualInflow = Number(saved.manualInflow || 0);
    const manualOutput = Number(saved.manualOutput || 0);
    const note = String(saved.note || '');

    const possibleCases = Math.round(totalHours * productivity);
    const possibleRemainingCases = Math.round(remainingHours * productivity);
    const workload = backlog + manualInflow;
    const gap = possibleRemainingCases + manualOutput - workload;

    return {
      team: teamName,
      totalHours: round2_(totalHours),
      remainingHours: round2_(remainingHours),
      usedHours: round2_(usedHours),
      productivity: round2_(productivity),
      backlog: backlog,
      manualInflow: manualInflow,
      manualOutput: manualOutput,
      possibleCases: possibleCases,
      possibleRemainingCases: possibleRemainingCases,
      workload: workload,
      gap: gap,
      status: getTagesplanungStatus_(gap),
      note: note
    };
  });

  const payload = {
    success: true,
    date: formatDateIso_(today),
    dateDisplay: Utilities.formatDate(today, Session.getScriptTimeZone(), 'dd.MM.yyyy'),
    kw: getWeekNumber(today),
    updatedAt: Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'dd.MM.yyyy HH:mm'),
    teams: teams
  };

  writeTagesplanungCurrentState_(payload);

  return payload;
}

function saveTagesplanungData(payload) {
  ensureTagesplanungSheets_();

  if (!payload) throw new Error('Keine Tagesplanung-Daten übergeben.');

  const dateObj = payload.date ? parseIsoDate_(payload.date) : new Date();
  dateObj.setHours(0, 0, 0, 0);

  const normalized = {
    success: true,
    date: formatDateIso_(dateObj),
    dateDisplay: Utilities.formatDate(dateObj, Session.getScriptTimeZone(), 'dd.MM.yyyy'),
    kw: getWeekNumber(dateObj),
    updatedAt: Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'dd.MM.yyyy HH:mm'),
    teams: Array.isArray(payload.teams) ? payload.teams.map(team => normalizeTagesplanungTeam_(team)) : []
  };

  writeTagesplanungCurrentState_(normalized);
  appendTagesplanungDatabase_(normalized);

  return {
    success: true,
    message: 'Tagesplanung gespeichert.',
    updatedAt: normalized.updatedAt
  };
}

function importTagesplanungBacklogFromText(level, pastedText, refDate) {
  ensureTagesplanungSheets_();

  const dateObj = refDate ? parseIsoDate_(refDate) : new Date();
  dateObj.setHours(0, 0, 0, 0);

  const config = String(level) === '2nd'
    ? TAGESPLANUNG_BACKLOG_2ND_CONFIG
    : TAGESPLANUNG_BACKLOG_1ST_CONFIG;

  const teamBacklog = {};
  TAGESPLANUNG_TEAM_ORDER.forEach(t => teamBacklog[t] = 0);

  const lines = String(pastedText || '').split(/\r?\n/);
  let currentLabel = null;
  let processedCount = 0;

  lines.forEach(rawLine => {
    const line = String(rawLine || '').trim();

    if (!line) return;

    if (line.startsWith('DEU_')) {
      currentLabel = line;
      return;
    }

    if (currentLabel && /^\d+/.test(line)) {
      const match = line.match(/^(\d+)/);

      if (match) {
        const count = Number(match[1]) || 0;
        const mappedTeam = config[currentLabel] || null;

        if (mappedTeam) {
          teamBacklog[mappedTeam] = Number(teamBacklog[mappedTeam] || 0) + count;
          processedCount += 1;
        }
      }

      currentLabel = null;
    }
  });

  const existing = getTagesplanungData();
  existing.teams = existing.teams.map(team => {
    const imported = Number(teamBacklog[team.team] || 0);

    if (imported > 0) {
      team.backlog = imported;
    }

    return normalizeTagesplanungTeam_(team);
  });

  writeTagesplanungCurrentState_(existing);
  appendBacklogImportToDatabase_(dateObj, level, teamBacklog);

  return {
    success: true,
    message: `${String(level) === '2nd' ? '2nd' : '1st'} Level Backlog importiert.`,
    processedCount: processedCount,
    teamBacklog: teamBacklog,
    data: existing
  };
}

function getTagesplanungProductivityMap_() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(TAGESPLANUNG_PRODUCTIVITY_SHEET_NAME);
  const map = {};

  if (!sheet || sheet.getLastRow() < 2) {
    return map;
  }

  const data = sheet.getRange(2, 1, sheet.getLastRow() - 1, 2).getValues();

  data.forEach(row => {
    const team = String(row[0] || '').trim();
    const value = Number(row[1]);

    if (team && !isNaN(value) && value > 0) {
      map[team] = value;

      const mapped = mapCapacitySkillGroup(team);
      if (mapped) {
        map[mapped] = value;
      }
    }
  });

  return map;
}

function getTagesplanungSavedMap_(dateObj) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(TAGESPLANUNG_SHEET_NAME);
  const map = {};

  if (!sheet || sheet.getLastRow() < 2) return map;

  const values = sheet.getRange(2, 1, sheet.getLastRow() - 1, 17).getValues();

  values.forEach(row => {
    const rowDate = row[0];

    if (rowDate instanceof Date && isSameDay_(rowDate, dateObj)) {
      const team = String(row[2] || '').trim();

      if (team) {
        map[team] = {
          totalHours: row[3],
          remainingHours: row[4],
          usedHours: row[5],
          backlog: row[6],
          manualInflow: row[7],
          manualOutput: row[8],
          productivity: row[9],
          note: row[15]
        };
      }
    }
  });

  return map;
}

function getTagesplanungLatestBacklogMap_(dateObj) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(TAGESPLANUNG_DB_SHEET_NAME);
  const map = {};

  if (!sheet || sheet.getLastRow() < 2) return map;

  const values = sheet.getRange(2, 1, sheet.getLastRow() - 1, 18).getValues();

  values.forEach(row => {
    const rowDate = row[1];

    if (rowDate instanceof Date && isSameDay_(rowDate, dateObj)) {
      const typ = String(row[3] || '');

      if (typ.toLowerCase().includes('backlog')) {
        const team = String(row[4] || '').trim();
        const backlog = Number(row[8]) || 0;

        if (team) {
          map[team] = backlog;
        }
      }
    }
  });

  return map;
}

function writeTagesplanungCurrentState_(payload) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(TAGESPLANUNG_SHEET_NAME);

  const dateObj = parseIsoDate_(payload.date);
  const kw = payload.kw || getWeekNumber(dateObj);

  const rows = (payload.teams || []).map(team => {
    const normalized = normalizeTagesplanungTeam_(team);

    return [
      dateObj,
      kw,
      normalized.team,
      normalized.totalHours,
      normalized.remainingHours,
      normalized.usedHours,
      normalized.backlog,
      normalized.manualInflow,
      normalized.manualOutput,
      normalized.productivity,
      normalized.possibleCases,
      normalized.possibleRemainingCases,
      normalized.workload,
      normalized.gap,
      normalized.status,
      normalized.note,
      new Date()
    ];
  });

  if (sheet.getLastRow() > 1) {
    const all = sheet.getRange(2, 1, sheet.getLastRow() - 1, 17).getValues();
    const keepRows = all.filter(row => !(row[0] instanceof Date && isSameDay_(row[0], dateObj)));

    sheet.getRange(2, 1, sheet.getLastRow() - 1, 17).clearContent();

    if (keepRows.length > 0) {
      sheet.getRange(2, 1, keepRows.length, 17).setValues(keepRows);
    }

    if (rows.length > 0) {
      sheet.getRange(2 + keepRows.length, 1, rows.length, 17).setValues(rows);
    }
  } else if (rows.length > 0) {
    sheet.getRange(2, 1, rows.length, 17).setValues(rows);
  }

  sheet.getRange('A:A').setNumberFormat('dd.MM.yyyy');
  sheet.getRange('D:F').setNumberFormat('0.00');
  sheet.getRange('G:N').setNumberFormat('0.00');
  sheet.autoResizeColumns(1, Math.min(sheet.getLastColumn(), 17));
}

function appendTagesplanungDatabase_(payload) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(TAGESPLANUNG_DB_SHEET_NAME);
  const dateObj = parseIsoDate_(payload.date);
  const kw = payload.kw || getWeekNumber(dateObj);

  const rows = (payload.teams || []).map(team => {
    const normalized = normalizeTagesplanungTeam_(team);

    return [
      timestamp,
      dateObj,
      kw,
      'Planung gespeichert',
      normalized.team,
      normalized.totalHours,
      normalized.remainingHours,
      normalized.usedHours,
      normalized.backlog,
      normalized.manualInflow,

      normalized.possibleRemainingCases,
      normalized.workload,
      normalized.gap,

  }

  sheet.getRange('B:B').setNumberFormat('dd.MM.yyyy');
  sheet.autoResizeColumns(1, Math.min(sheet.getLastColumn(), 18));
}

  const sheet = ss.getSheetByName(TAGESPLANUNG_DB_SHEET_NAME);
  const timestamp = new Date();
  const kw = getWeekNumber(dateObj);

  const rows = Object.keys(teamBacklog).map(team => [
    timestamp,
    dateObj,
    kw,
    `Backlog Import ${level}`,
    team,
    '',

    '',
    '',

    '',
    '',

    '',
    ''
  ]);

  if (rows.length > 0) {
    sheet.getRange(sheet.getLastRow() + 1, 1, rows.length, 18).setValues(rows);
  }
}

function normalizeTagesplanungTeam_(team) {
  const teamName = String(team.team || '').trim();
  const totalHours = Number(team.totalHours || 0);
  const remainingHours = Number(team.remainingHours || 0);
  const usedHours = Number(team.usedHours || Math.max(0, totalHours - remainingHours));
  const productivity = Number(team.productivity || 5);
  const backlog = Number(team.backlog || 0);
  const manualInflow = Number(team.manualInflow || 0);
  const manualOutput = Number(team.manualOutput || 0);
  const possibleCases = Math.round(totalHours * productivity);
  const possibleRemainingCases = Math.round(remainingHours * productivity);
  const workload = backlog + manualInflow;
  const gap = possibleRemainingCases + manualOutput - workload;

  return {
    team: teamName,
    totalHours: round2_(totalHours),
    remainingHours: round2_(remainingHours),
    usedHours: round2_(usedHours),
    productivity: round2_(productivity),
    backlog: backlog,
    manualInflow: manualInflow,
    manualOutput: manualOutput,
    possibleCases: possibleCases,
    possibleRemainingCases: possibleRemainingCases,
    workload: workload,
    gap: gap,
    status: getTagesplanungStatus_(gap),
    note: String(team.note || '')
  };
}

function getTagesplanungStatus_(gap) {
  if (gap >= 25) return 'Grün';
  if (gap >= 0) return 'Gelb';
  return 'Rot';
}

function formatDateIso_(dateObj) {
  return Utilities.formatDate(dateObj, Session.getScriptTimeZone(), 'yyyy-MM-dd');
}

function parseIsoDate_(iso) {
  if (iso instanceof Date) {
    return iso;
  }

  const parts = String(iso || '').split('-');

  if (parts.length === 3) {
    return new Date(Number(parts[0]), Number(parts[1]) - 1, Number(parts[2]));
  }

  return new Date();
}

function round2_(value) {
  return Math.round((Number(value) || 0) * 100) / 100;
}

function isSameDay_(d1, d2) {
  return d1.getFullYear() === d2.getFullYear() &&
    d1.getMonth() === d2.getMonth() &&
    d1.getDate() === d2.getDate();
}
