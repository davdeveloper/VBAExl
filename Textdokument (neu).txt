/**
 * Monatsimport für Plano Backup.
 * Liest mehrere Wochenplan-Blöcke aus einer Datei.
 * Schreibt testweise nach Datenbank_KO_SKILL_Test und Datenbank_Gesamt_Test.
 */
function processSelectedMonthFile(fileId) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();

  const koSkillSheetName = "Datenbank_KO_SKILL_Test";
  const gesamtSheetName = "Datenbank_Gesamt_Test";
  const errorSheetName = "Fehlende_Zuordnungen_Monatsimport";

  const skillMap = getMapping(ss);
  const planoSkillsSorted = Object.keys(skillMap).sort((a, b) => b.length - a.length);
  const nameMap = getNameMapping(ss);
  const localTeamMap = getTeamMapping(ss);
  const ignoreSet = getIgnoreMapping(ss);
  const teamMap = getEmployeeTeams();

  const unmappedSkills = new Set();
  const unmappedEmployees = new Set();

  const abwDbName = (skillMap["ABW"] || "ABW").toUpperCase();

  const data = readPlanoBackupFile_(fileId);

  if (!data || data.length < 4) {
    throw new Error("Die Datei enthält nicht genug Daten.");
  }

  const koHeaders = prepareKoSkillHeaders_(ss, koSkillSheetName, skillMap);

  const records = parsePlanoMonthRecords_(data, {
    skillMap,
    planoSkillsSorted,
    nameMap,
    localTeamMap,
    teamMap,
    ignoreSet,
    unmappedSkills,
    unmappedEmployees
  });

  const recordKeys = Object.keys(records);

  if (recordKeys.length === 0) {
    throw new Error("Es wurden keine verwertbaren Monatsdaten gefunden. Prüfe, ob die Datei wirklich Monatsblöcke mit 'Wochenplan KW xx / Jahr' enthält.");
  }

  const koRows = buildKoSkillRows_(records, koHeaders, abwDbName);
  writeRowsToSheet_(ss, koSkillSheetName, koHeaders, koRows);

  const gesamtHeaders = getExistingHeadersOrCreateGesamtHeaders_(ss, gesamtSheetName, koHeaders);

  const teamRows = buildGesamtTeamRows_(records, gesamtHeaders, abwDbName);
  writeRowsToExistingHeaderSheet_(ss, gesamtSheetName, gesamtHeaders, teamRows);

  writeMonthImportErrors_(ss, errorSheetName, unmappedEmployees, unmappedSkills);

  const foundDates = new Set();

  recordKeys.forEach(key => {
    foundDates.add(records[key].date);
  });

  const foundWeeks = new Set();

  recordKeys.forEach(key => {
    foundWeeks.add(records[key].kw);
  });

  return `Monatsimport abgeschlossen.
${koRows.length} Mitarbeiter-Tagesdatensätze nach '${koSkillSheetName}' geschrieben.
${teamRows.length} Team-Tagesdatensätze nach '${gesamtSheetName}' geschrieben.
Gefundene Tage: ${foundDates.size}
Gefundene Kalenderwochen: ${Array.from(foundWeeks).filter(Boolean).join(", ")}
Fehlende Team-Zuordnungen: ${unmappedEmployees.size}
Fehlende Tätigkeiten: ${unmappedSkills.size}`;
}

/**
 * Liest Google Sheet, Excel oder CSV wie im bestehenden Tagesimport.
 */
function readPlanoBackupFile_(fileId) {
  const file = DriveApp.getFileById(fileId);
  const mimeType = file.getMimeType();

  let data = [];

  try {
    if (mimeType === MimeType.GOOGLE_SHEETS) {
      data = SpreadsheetApp.openById(fileId).getSheets()[0].getDataRange().getValues();
    } else {
      const blob = file.getBlob();
      const textData = blob.getDataAsString("UTF-8");

      if (textData.includes("PK\x03\x04")) {
        try {
          const resource = {
            name: "Temp_" + file.getName(),
            mimeType: MimeType.GOOGLE_SHEETS
          };

          let tempFile;

          if (typeof Drive !== "undefined") {
            tempFile = Drive.Files.create(resource, blob, { supportsAllDrives: true });

            const tempDoc = SpreadsheetApp.openById(tempFile.id);
            data = tempDoc.getSheets()[0].getDataRange().getValues();

            Drive.Files.remove(tempFile.id);
          } else {
            throw new Error("Drive API nicht aktiviert.");
          }
        } catch (convertErr) {
          throw new Error("Fehler bei der Excel-Konvertierung! Detail: " + convertErr.message);
        }
      } else {
        if (textData.includes(";") && textData.split(";").length > textData.split(",").length) {
          data = Utilities.parseCsv(textData, ";");
        } else {
          data = Utilities.parseCsv(textData);
        }
      }
    }
  } catch (e) {
    throw new Error("Fehler beim Lesen der Datei: " + e.message);
  }

  return data;
}

/**
 * Erstellt die Zielspalten analog zum bestehenden KO_SKILL Aufbau.
 */
function prepareKoSkillHeaders_(ss, sheetName, skillMap) {
  let sheet = ss.getSheetByName(sheetName);

  if (!sheet) {
    sheet = ss.insertSheet(sheetName);
    sheet.appendRow(["Datum", "KW", "Mitarbeitername"]);
    sheet.getRange("A1:C1").setFontWeight("bold");
  }

  const lastCol = Math.max(sheet.getLastColumn(), 3);
  let headers = sheet.getRange(1, 1, 1, lastCol).getValues()[0];

  const requiredHeaders = [...new Set(Object.values(skillMap))];

  ["Differenz", "Team", "Schicht", "Soll", "Ist", "Ausgleichstag"].forEach(headerName => {
    if (!requiredHeaders.some(x => String(x).trim().toUpperCase() === headerName.toUpperCase())) {
      requiredHeaders.push(headerName);
    }
  });

  requiredHeaders.forEach(headerName => {
    const exists = headers.some(h => String(h).trim().toUpperCase() === String(headerName).trim().toUpperCase());

    if (!exists) {
      headers.push(headerName);
      sheet.getRange(1, headers.length).setValue(headerName);
      sheet.getRange(1, headers.length).setFontWeight("bold");
    }
  });

  return headers;
}

/**
 * Erkennt alle Wochenplan-Blöcke untereinander.
 * Jeder Block startet mit "Wochenplan KW xx / yyyy".
 */
function parsePlanoMonthRecords_(data, ctx) {
  const records = {};

  let r = 0;

  while (r < data.length) {
    const rowStr = data[r].join(" ");

    if (!rowStr.includes("Wochenplan KW")) {
      r++;
      continue;
    }

    const titleStr = rowStr;

    const kwMatch = titleStr.match(/KW\s+(\d+)/i);
    const yearMatch = titleStr.match(/\/\s*(\d{4})/);

    const kw = kwMatch ? kwMatch[1] : "";
    const year = yearMatch ? yearMatch[1] : new Date().getFullYear().toString();

    const nextBlockRow = findNextPlanoBlockRow_(data, r + 1);
    const blockEndRow = nextBlockRow === -1 ? data.length : nextBlockRow;

    const headerRowIndex = findHeaderRowInBlock_(data, r + 1, blockEndRow);

    if (headerRowIndex === -1) {
      r = blockEndRow;
      continue;
    }

    parsePlanoBlockIntoRecords_(data, headerRowIndex, blockEndRow, kw, year, records, ctx);

    r = blockEndRow;
  }

  return records;
}

/**
 * Findet den nächsten Wochenplan-Block.
 */
function findNextPlanoBlockRow_(data, startRow) {
  for (let i = startRow; i < data.length; i++) {
    const rowStr = data[i].join(" ");

    if (rowStr.includes("Wochenplan KW")) {
      return i;
    }
  }

  return -1;
}

/**
 * Findet die Kopfzeile innerhalb eines Blocks.
 */
function findHeaderRowInBlock_(data, startRow, endRow) {
  for (let i = startRow; i < Math.min(endRow, startRow + 20); i++) {
    const rowStr = data[i].join(" ");
    const firstCell = String(data[i][0]).trim();

    const hasNameHeader = firstCell === "Name" || rowStr.includes("Name");
    const hasDate = rowStr.match(/\d{1,2}\.\d{1,2}\./);

    if (hasNameHeader && hasDate) {
      return i;
    }

    if (
      rowStr.includes("Montag,") ||
      rowStr.includes("Dienstag,") ||
      rowStr.includes("Mittwoch,") ||
      rowStr.includes("Donnerstag,") ||
      rowStr.includes("Freitag,") ||
      rowStr.includes("Samstag,") ||
      rowStr.includes("Sonntag,")
    ) {
      return i;
    }
  }

  return -1;
}

/**
 * Wertet einen Wochenplan-Block aus.
 *
 * Monatsbackup:
 * Isth und Vertrag h sind Wochenwerte.
 *
 * IST:
 * Wird aus den echten Tagesdetails ohne Pause berechnet.
 *
 * SOLL:
 * Wird aus Vertrag h auf Minutenbasis auf Vertragstage verteilt.
 * Das verhindert Rundungsfehler wie 195:14 statt 195:08.
 *
 * Ausgleichstag:
 * Wird als Vertragstag mitgerechnet, bekommt aber Soll = 0.
 * Die Stunden landen in der Ausgleichstag-Spalte.
 */
function parsePlanoBlockIntoRecords_(data, headerRowIndex, blockEndRow, kw, year, records, ctx) {
  const AUSGLEICHSTAG_ZAEHLT_INS_SOLL = false;

  const headerRow = data[headerRowIndex];
  const dateColumns = [];

  let differenzColIndex = headerRow.findIndex(h => String(h).trim() === "Differenz");
  if (differenzColIndex === -1) differenzColIndex = 10;

  let isthColIndex = headerRow.findIndex(h => String(h).trim() === "Isth");
  if (isthColIndex === -1) isthColIndex = 12;

  let vertragColIndex = headerRow.findIndex(h => String(h).trim() === "Vertrag h");
  if (vertragColIndex === -1) vertragColIndex = 14;

  for (let c = 0; c < headerRow.length; c++) {
    const dateInfo = extractDateFromHeaderCell_(headerRow[c], year);

    if (dateInfo) {
      dateColumns.push({
        colIndex: c,
        date: dateInfo.dateStr,
        dayOfWeek: dateInfo.dayOfWeek,
        isSaturday: dateInfo.isSaturday,
        isSunday: dateInfo.isSunday,
        isWeekend: dateInfo.isWeekend
      });
    }
  }

  if (dateColumns.length === 0) {
    return;
  }

  const startRow = headerRowIndex + 2;

  for (let r = startRow; r < blockEndRow; r++) {
    const nameCell = String(data[r][0]);

    if (!nameCell.includes("Lidl Kundenservice DE")) {
      continue;
    }

    const planoCleanName = normalizeStr(nameCell.split("\n")[0]);

    if (ctx.ignoreSet.has(planoCleanName)) {
      continue;
    }

    const mappedName = ctx.nameMap[planoCleanName] || planoCleanName;
    const assignedTeam = ctx.localTeamMap[mappedName] || ctx.localTeamMap[planoCleanName] || ctx.teamMap[mappedName];

    if (!assignedTeam) {
      ctx.unmappedEmployees.add(planoCleanName);
    }

    const differenzVal = parseGermanFloat(data[r][differenzColIndex]);
    const weeklyIstVal = parseGermanFloat(data[r][isthColIndex]);
    const weeklyVertragVal = parseGermanFloat(data[r][vertragColIndex]);

    const dayInfos = buildEmployeeDayInfos_(
      data,
      r,
      blockEndRow,
      dateColumns,
      weeklyIstVal,
      weeklyVertragVal
    );

    for (const info of dayInfos) {
      const recKey = `${info.date}|${kw}|${mappedName}`;

      if (!records[recKey]) {
        records[recKey] = {
          date: info.date,
          kw: kw,
          name: mappedName,
          skills: {},
          differenz: differenzVal,
          soll: 0,
          ist: 0,
          isAnw: false,
          schicht: info.dayCell,
          team: assignedTeam || "Unbekannt"
        };
      }

      const rec = records[recKey];

      if (info.isUrlaub || info.isAbw || info.isUeberstunden || info.isWunschfrei || info.isAusgleichstag) {
        let matchedPlanoSkill = "ABW";

        if (info.isUrlaub) matchedPlanoSkill = "Urlaub";
        if (info.isUeberstunden) matchedPlanoSkill = "Überstundenfrei";
        if (info.isWunschfrei) matchedPlanoSkill = "Wunschfrei";
        if (info.isAusgleichstag) matchedPlanoSkill = "Ausgleichstag";

        const dbSkill = ctx.skillMap[matchedPlanoSkill] || matchedPlanoSkill;

        if (!ctx.skillMap[matchedPlanoSkill] && !info.isAusgleichstag) {
          ctx.unmappedSkills.add(matchedPlanoSkill);
        }

        let absenceMinutes = info.dailyContractMinutes;

        if (!absenceMinutes && info.detailIstMinutes > 0) {
          absenceMinutes = info.detailIstMinutes;
        }

        if (!absenceMinutes && weeklyVertragVal > 0 && info.contractDayCount > 0) {
          absenceMinutes = Math.round((weeklyVertragVal * 60) / info.contractDayCount);
        }

        const absenceHours = absenceMinutes / 60;

        rec.skills[dbSkill] = (rec.skills[dbSkill] || 0) + absenceHours;

        if (info.isAusgleichstag) {
          rec.soll = AUSGLEICHSTAG_ZAEHLT_INS_SOLL ? absenceHours : 0;
          rec.ist = 0;
        } else if (info.isUeberstunden || info.isWunschfrei) {
          rec.soll = absenceHours;
          rec.ist = 0;
        } else {
          rec.soll = absenceHours;
          rec.ist = absenceHours;
        }

        continue;
      }

      if (info.upperDayCell.includes("ANW") || info.upperDayCell === "ANW") {
        rec.isAnw = true;
      }

      if (info.isWeekend && info.detailIstMinutes > 0 && info.dailyContractMinutes === 0) {
        rec.soll = info.detailIstMinutes / 60;
      } else {
        rec.soll = info.dailyContractMinutes / 60;
      }

      rec.ist = info.detailIstMinutes / 60;

      if (!info.detailCell || info.detailCell.trim() === "") {
        continue;
      }

      const lines = info.detailCell.split("\n").map(l => l.trim());
      let currentTimeRange = null;

      for (let i = 0; i < lines.length; i++) {
        const line = lines[i];

        if (line.match(/^\d{1,2}:\d{2}\s*-\s*\d{1,2}:\d{2}/)) {
          currentTimeRange = normalizeTimeRange_(line);
          continue;
        }

        if (!currentTimeRange) {
          continue;
        }

        const lineUpper = line.toUpperCase();

        if (
          !line ||
          lineUpper === "PAUSE" ||
          lineUpper.includes("PAUSE")
        ) {
          currentTimeRange = null;
          continue;
        }

        let matchedDbSkill = null;
        const normalizedLineUpper = normalizeStr(line).toUpperCase();

        for (const planoSkill of ctx.planoSkillsSorted) {
          if (normalizedLineUpper.includes(planoSkill.toUpperCase())) {
            matchedDbSkill = ctx.skillMap[planoSkill];
            break;
          }
        }

        if (matchedDbSkill) {
          const hrs = calculateHours(currentTimeRange);
          rec.skills[matchedDbSkill] = (rec.skills[matchedDbSkill] || 0) + hrs;
        } else {
          if (line && lineUpper !== "ANW") {
            ctx.unmappedSkills.add(line.trim());
          }
        }

        currentTimeRange = null;
      }
    }
  }
}

/**
 * Baut pro Mitarbeiter alle Tagesinfos eines Wochenblocks.
 *
 * Vertrag h wird zuerst in exakte Minuten umgewandelt.
 * Diese Minuten werden dann auf Vertragstage verteilt.
 *
 * Vertragstage sind:
 * Arbeitstage mit Stunden
 * Urlaub
 * ABW
 * Überstundenfrei
 * Wunschfrei
 * Ausgleichstage
 * sonstige Wochentage mit Plantext
 *
 * Keine Vertragstage sind:
 * Wochenende ohne Arbeitszeit
 * Wochenende mit "-"
 * komplett freie "-" Tage am Wochenende
 *
 * Ausgleichstage zählen für die Verteilung mit,
 * bekommen später aber Soll = 0.
 */
function buildEmployeeDayInfos_(data, employeeRowIndex, blockEndRow, dateColumns, weeklyIstVal, weeklyVertragVal) {
  const infos = [];

  for (const dc of dateColumns) {
    const dayCell = String(data[employeeRowIndex][dc.colIndex]).trim();
    const upperDayCell = dayCell.toUpperCase();

    const detailCell = employeeRowIndex + 1 < blockEndRow
      ? String(data[employeeRowIndex + 1][dc.colIndex])
      : "";

    const detailIstMinutes = calculateWorkMinutesFromDetailCell_(detailCell);
    const dayCellTimeMinutes = calculateMinutesFromAnyTimeRanges_(dayCell);

    const isUrlaub = upperDayCell.includes("URLAUB");
    const isAbw = upperDayCell.includes("ABW");
    const isUeberstunden = upperDayCell.includes("ÜBERSTUNDEN");
    const isWunschfrei = upperDayCell.includes("WUNSCHFREI");

    const hasAbsenceLabel = isUrlaub || isAbw || isUeberstunden || isWunschfrei;
    const hasHours = detailIstMinutes > 0 || dayCellTimeMinutes > 0;
    const hasPlanText = upperDayCell !== "" && upperDayCell !== "-";

    const isWeekendFreeDash = dc.isWeekend && upperDayCell === "-" && !hasHours && !hasAbsenceLabel;

    const isAusgleichstag =
      upperDayCell === "-" &&
      !dc.isWeekend &&
      !hasHours &&
      !hasAbsenceLabel &&
      weeklyVertragVal > 0;

    const isEmptyFreeDay =
      upperDayCell === "-" &&
      !hasHours &&
      !hasAbsenceLabel &&
      !isAusgleichstag;

    let countsForContractDistribution = false;

    if (isWeekendFreeDash) {
      countsForContractDistribution = false;
    } else if (isEmptyFreeDay && !isAusgleichstag) {
      countsForContractDistribution = false;
    } else if (isAusgleichstag) {
      countsForContractDistribution = true;
    } else if (hasHours) {
      countsForContractDistribution = !dc.isWeekend;
    } else if (hasAbsenceLabel) {
      countsForContractDistribution = !dc.isWeekend;
    } else if (!dc.isWeekend && hasPlanText) {
      countsForContractDistribution = true;
    }

    infos.push({
      colIndex: dc.colIndex,
      date: dc.date,
      dayOfWeek: dc.dayOfWeek,
      isWeekend: dc.isWeekend,
      isSaturday: dc.isSaturday,
      isSunday: dc.isSunday,
      dayCell: dayCell,
      upperDayCell: upperDayCell,
      detailCell: detailCell,
      detailIstMinutes: detailIstMinutes,
      detailIstHours: detailIstMinutes / 60,
      dayCellTimeMinutes: dayCellTimeMinutes,
      dayCellTimeHours: dayCellTimeMinutes / 60,
      isUrlaub: isUrlaub,
      isAbw: isAbw,
      isUeberstunden: isUeberstunden,
      isWunschfrei: isWunschfrei,
      isAusgleichstag: isAusgleichstag,
      isEmptyFreeDay: isEmptyFreeDay,
      hasHours: hasHours,
      hasAbsenceLabel: hasAbsenceLabel,
      hasPlanText: hasPlanText,
      countsForContractDistribution: countsForContractDistribution,
      dailyContractMinutes: 0,
      contractDayCount: 0
    });
  }

  const contractInfos = infos.filter(info => info.countsForContractDistribution);
  const contractDayCount = contractInfos.length;

  const weeklyContractMinutes = Math.round((weeklyVertragVal || 0) * 60);

  if (weeklyContractMinutes > 0 && contractDayCount > 0) {
    const baseMinutes = Math.floor(weeklyContractMinutes / contractDayCount);
    let restMinutes = weeklyContractMinutes - (baseMinutes * contractDayCount);

    contractInfos.forEach(info => {
      info.dailyContractMinutes = baseMinutes;

      if (restMinutes > 0) {
        info.dailyContractMinutes += 1;
        restMinutes--;
      }
    });
  }

  infos.forEach(info => {
    info.contractDayCount = contractDayCount;
  });

  return infos;
}

/**
 * Holt Datum aus einer Tageszelle.
 */
function extractDateFromHeaderCell_(cellValue, year) {
  let valStr = "";

  if (cellValue instanceof Date) {
    valStr = Utilities.formatDate(cellValue, Session.getScriptTimeZone(), "dd.MM.yyyy");
  } else {
    valStr = String(cellValue).trim();
  }

  const match = valStr.match(/(\d{1,2})\.(\d{1,2})\./);

  if (!match) {
    return null;
  }

  const day = ("0" + match[1]).slice(-2);
  const month = ("0" + match[2]).slice(-2);
  const dateStr = `${day}.${month}.${year}`;

  const dateObj = new Date(Number(year), Number(month) - 1, Number(day));
  const dayOfWeek = dateObj.getDay();
  const isSaturday = dayOfWeek === 6;
  const isSunday = dayOfWeek === 0;
  const isWeekend = isSaturday || isSunday;

  return {
    dateStr: dateStr,
    dayOfWeek: dayOfWeek,
    isSaturday: isSaturday,
    isSunday: isSunday,
    isWeekend: isWeekend
  };
}

/**
 * Berechnet die täglichen Arbeitsstunden aus Detailzelle oder Tageszelle.
 */
function calculateDailyWorkHoursFromPlanoDay_(dayCell, detailCell) {
  const detailMinutes = calculateWorkMinutesFromDetailCell_(detailCell);

  if (detailMinutes > 0) {
    return detailMinutes / 60;
  }

  const dayCellMinutes = calculateMinutesFromAnyTimeRanges_(dayCell);

  if (dayCellMinutes > 0) {
    return dayCellMinutes / 60;
  }

  return 0;
}

/**
 * Summiert alle Arbeitszeitblöcke aus der Detailzelle in Stunden.
 * Pausen werden nicht gezählt.
 */
function calculateWorkHoursFromDetailCell_(detailCell) {
  return calculateWorkMinutesFromDetailCell_(detailCell) / 60;
}

/**
 * Summiert alle Arbeitszeitblöcke aus der Detailzelle in Minuten.
 * Pausen werden nicht gezählt.
 */
function calculateWorkMinutesFromDetailCell_(detailCell) {
  if (!detailCell) {
    return 0;
  }

  const lines = String(detailCell).split("\n").map(l => l.trim());
  let currentTimeRange = null;
  let totalMinutes = 0;

  for (let i = 0; i < lines.length; i++) {
    const line = lines[i];

    if (line.match(/^\d{1,2}:\d{2}\s*-\s*\d{1,2}:\d{2}/)) {
      currentTimeRange = normalizeTimeRange_(line);
      continue;
    }

    if (!currentTimeRange) {
      continue;
    }

    const lineUpper = line.toUpperCase();

    if (
      line &&
      lineUpper !== "PAUSE" &&
      !lineUpper.includes("PAUSE")
    ) {
      totalMinutes += calculateMinutes(currentTimeRange);
    }

    currentTimeRange = null;
  }

  return totalMinutes;
}

/**
 * Erkennt Zeiträume in beliebigem Text und gibt Stunden zurück.
 */
function calculateHoursFromAnyTimeRanges_(text) {
  return calculateMinutesFromAnyTimeRanges_(text) / 60;
}

/**
 * Erkennt Zeiträume in beliebigem Text und gibt Minuten zurück.
 */
function calculateMinutesFromAnyTimeRanges_(text) {
  if (!text) {
    return 0;
  }

  const str = String(text);
  const regex = /(\d{1,2}:\d{2})\s*-\s*(\d{1,2}:\d{2})/g;

  let match;
  let totalMinutes = 0;

  while ((match = regex.exec(str)) !== null) {
    totalMinutes += calculateMinutes(`${match[1]}-${match[2]}`);
  }

  return totalMinutes;
}

/**
 * Normalisiert eine Zeitzeile auf "HH:MM-HH:MM".
 */
function normalizeTimeRange_(line) {
  const match = String(line).match(/(\d{1,2}:\d{2})\s*-\s*(\d{1,2}:\d{2})/);

  if (!match) {
    return "";
  }

  return `${match[1]}-${match[2]}`;
}

/**
 * Berechnet Minuten aus einem String wie "07:00-09:15".
 */
function calculateMinutes(timeRange) {
  try {
    const [start, end] = timeRange.split("-");
    const [h1, m1] = start.split(":").map(Number);
    const [h2, m2] = end.split(":").map(Number);

    let diffMinutes = (h2 * 60 + m2) - (h1 * 60 + m1);

    if (diffMinutes < 0) {
      diffMinutes += 24 * 60;
    }

    return diffMinutes;
  } catch (e) {
    return 0;
  }
}

/**
 * Fallback für Abwesenheiten.
 */
function calculateDailyContractFallback_(data, employeeRowIndex, dateColumns, weeklyVertragVal) {
  if (!weeklyVertragVal || weeklyVertragVal <= 0) {
    return 0;
  }

  let countedDays = 0;

  for (const dc of dateColumns) {
    const dayCell = String(data[employeeRowIndex][dc.colIndex]).trim();
    const detailCell = employeeRowIndex + 1 < data.length
      ? String(data[employeeRowIndex + 1][dc.colIndex])
      : "";

    const upperDayCell = dayCell.toUpperCase();
    const hasWorkHours = calculateDailyWorkHoursFromPlanoDay_(dayCell, detailCell) > 0;

    const hasAbsence =
      upperDayCell.includes("URLAUB") ||
      upperDayCell.includes("ABW") ||
      upperDayCell.includes("ÜBERSTUNDEN") ||
      upperDayCell.includes("WUNSCHFREI");

    if (!dc.isWeekend && (hasWorkHours || hasAbsence)) {
      countedDays++;
    }
  }

  if (countedDays > 0) {
    return weeklyVertragVal / countedDays;
  }

  return weeklyVertragVal / 5;
}

/**
 * Baut Mitarbeiterzeilen für KO_SKILL.
 */
function buildKoSkillRows_(records, headers, abwDbName) {
  const results = [];

  Object.keys(records).sort(sortRecordKeys_).forEach(key => {
    const rec = records[key];

    const rowData = new Array(headers.length).fill("");

    rowData[0] = rec.date;
    rowData[1] = rec.kw;
    rowData[2] = rec.name;

    let hasSkillData = false;

    for (const dbSkill in rec.skills) {
      const colIndex = headers.findIndex(h => String(h).trim().toUpperCase() === dbSkill.toUpperCase());

      if (colIndex !== -1) {
        if (shouldFormatSkillAsHHMM_(dbSkill, abwDbName)) {
          rowData[colIndex] = minutesToHHMM(Math.round(rec.skills[dbSkill] * 60));
        } else {
          rowData[colIndex] = rec.skills[dbSkill];
        }

        hasSkillData = true;
      }
    }

    if (hasSkillData || rec.isAnw) {
      setRowValueByHeader_(rowData, headers, "Differenz", rec.differenz);
      setRowValueByHeader_(rowData, headers, "Soll", minutesToHHMM(Math.round(rec.soll * 60)));
      setRowValueByHeader_(rowData, headers, "Ist", rec.ist);
      setRowValueByHeader_(rowData, headers, "Team", rec.team);
      setRowValueByHeader_(rowData, headers, "Schicht", rec.schicht);

      results.push(rowData);
    }
  });

  return results;
}

/**
 * Baut Teamaggregation für Datenbank_Gesamt_Test.
 */
function buildGesamtTeamRows_(records, headers, abwDbName) {
  const teamRecords = {};

  for (const key in records) {
    const rec = records[key];

    if (Object.keys(rec.skills).length === 0 && !rec.isAnw) {
      continue;
    }

    const teamName = rec.team || "Ohne Team";
    const teamKey = `${rec.date}|${rec.kw}|${teamName}`;

    if (!teamRecords[teamKey]) {
      teamRecords[teamKey] = {
        date: rec.date,
        kw: rec.kw,
        team: teamName,
        skills: {},
        differenz: 0,
        sollMins: 0,
        ist: 0
      };
    }

    for (const dbSkill in rec.skills) {
      teamRecords[teamKey].skills[dbSkill] = (teamRecords[teamKey].skills[dbSkill] || 0) + rec.skills[dbSkill];
    }

    teamRecords[teamKey].differenz += rec.differenz || 0;
    teamRecords[teamKey].sollMins += Math.round((rec.soll || 0) * 60);
    teamRecords[teamKey].ist += rec.ist || 0;
  }

  const results = [];

  Object.keys(teamRecords).sort(sortRecordKeys_).forEach(teamKey => {
    const tr = teamRecords[teamKey];

    const rowData = new Array(headers.length).fill("");

    rowData[0] = tr.date;
    rowData[1] = tr.kw;
    rowData[2] = tr.team;

    for (const dbSkill in tr.skills) {
      const colIndex = headers.findIndex(h => String(h).trim().toUpperCase() === dbSkill.toUpperCase());

      if (colIndex !== -1) {
        if (shouldFormatSkillAsHHMM_(dbSkill, abwDbName)) {
          rowData[colIndex] = minutesToHHMM(Math.round(tr.skills[dbSkill] * 60));
        } else {
          rowData[colIndex] = tr.skills[dbSkill];
        }
      }
    }

    setRowValueByHeader_(rowData, headers, "Differenz", tr.differenz);
    setRowValueByHeader_(rowData, headers, "Soll", minutesToHHMM(tr.sollMins));
    setRowValueByHeader_(rowData, headers, "Ist", tr.ist);

    results.push(rowData);
  });

  return results;
}

/**
 * Formatiert bestimmte Stundenfelder als HH:MM.
 */
function shouldFormatSkillAsHHMM_(dbSkill, abwDbName) {
  const upper = String(dbSkill).trim().toUpperCase();

  return (
    upper === String(abwDbName).trim().toUpperCase() ||
    upper === "AUSGLEICHSTAG"
  );
}

/**
 * Schreibt ein Sheet komplett neu.
 * Diese Funktion wird für Datenbank_KO_SKILL_Test genutzt.
 */
function writeRowsToSheet_(ss, sheetName, headers, rows) {
  let sheet = ss.getSheetByName(sheetName);

  if (!sheet) {
    sheet = ss.insertSheet(sheetName);
  }

  sheet.clearContents();

  sheet.getRange(1, 1, 1, headers.length).setValues([headers]);
  sheet.getRange(1, 1, 1, headers.length).setFontWeight("bold");

  if (rows.length > 0) {
    sheet.getRange(2, 1, rows.length, headers.length).setValues(rows);
  }

  sheet.autoResizeColumns(1, Math.min(headers.length, 20));
}

/**
 * Nutzt bei Datenbank_Gesamt_Test die bereits vorhandene Kopfzeile.
 * Nur wenn das Sheet leer ist, wird eine Kopfzeile aus KO_SKILL abgeleitet.
 */
function getExistingHeadersOrCreateGesamtHeaders_(ss, sheetName, fallbackKoHeaders) {
  let sheet = ss.getSheetByName(sheetName);

  if (!sheet) {
    sheet = ss.insertSheet(sheetName);

    const fallbackHeaders = [...fallbackKoHeaders];
    fallbackHeaders[2] = "Team";

    sheet.getRange(1, 1, 1, fallbackHeaders.length).setValues([fallbackHeaders]);
    sheet.getRange(1, 1, 1, fallbackHeaders.length).setFontWeight("bold");

    return fallbackHeaders;
  }

  const lastCol = sheet.getLastColumn();

  if (lastCol < 1) {
    const fallbackHeaders = [...fallbackKoHeaders];
    fallbackHeaders[2] = "Team";

    sheet.getRange(1, 1, 1, fallbackHeaders.length).setValues([fallbackHeaders]);
    sheet.getRange(1, 1, 1, fallbackHeaders.length).setFontWeight("bold");

    return fallbackHeaders;
  }

  const existingHeaders = sheet.getRange(1, 1, 1, lastCol).getValues()[0];
  const hasUsableHeaders = existingHeaders.some(h => String(h).trim() !== "");

  if (!hasUsableHeaders) {
    const fallbackHeaders = [...fallbackKoHeaders];
    fallbackHeaders[2] = "Team";

    sheet.getRange(1, 1, 1, fallbackHeaders.length).setValues([fallbackHeaders]);
    sheet.getRange(1, 1, 1, fallbackHeaders.length).setFontWeight("bold");

    return fallbackHeaders;
  }

  return existingHeaders;
}

/**
 * Schreibt nur Daten ab Zeile 2.
 * Zeile 1 bleibt komplett unverändert.
 * Diese Funktion wird für Datenbank_Gesamt_Test genutzt.
 */
function writeRowsToExistingHeaderSheet_(ss, sheetName, headers, rows) {
  let sheet = ss.getSheetByName(sheetName);

  if (!sheet) {
    sheet = ss.insertSheet(sheetName);
    sheet.getRange(1, 1, 1, headers.length).setValues([headers]);
    sheet.getRange(1, 1, 1, headers.length).setFontWeight("bold");
  }

  const lastRow = sheet.getLastRow();
  const lastCol = Math.max(sheet.getLastColumn(), headers.length);

  if (lastRow > 1) {
    sheet.getRange(2, 1, lastRow - 1, lastCol).clearContent();
  }

  if (rows.length > 0) {
    sheet.getRange(2, 1, rows.length, headers.length).setValues(rows);
  }
}

/**
 * Schreibt fehlende Zuordnungen in einen eigenen Prüf-Reiter.
 */
function writeMonthImportErrors_(ss, sheetName, unmappedEmployees, unmappedSkills) {
  let sheet = ss.getSheetByName(sheetName);

  if (!sheet) {
    sheet = ss.insertSheet(sheetName);
  }

  sheet.clear();

  const rows = [["Typ", "Fehlender Eintrag"]];

  unmappedEmployees.forEach(emp => rows.push(["Mitarbeiter ohne Team", emp]));
  unmappedSkills.forEach(skill => rows.push(["Unbekannte Tätigkeit", skill]));

  if (rows.length === 1) {
    rows.push(["OK", "Alle Mitarbeiter und Tätigkeiten konnten zugeordnet werden."]);
  }

  sheet.getRange(1, 1, rows.length, 2).setValues(rows);
  sheet.getRange(1, 1, 1, 2).setFontWeight("bold");
  sheet.autoResizeColumns(1, 2);
}

/**
 * Setzt einen Wert anhand des Spaltennamens.
 */
function setRowValueByHeader_(rowData, headers, headerName, value) {
  const idx = headers.findIndex(h => String(h).trim().toUpperCase() === headerName.toUpperCase());

  if (idx !== -1) {
    rowData[idx] = value;
  }
}

/**
 * Sortiert nach Datum und danach nach Rest-Key.
 */
function sortRecordKeys_(a, b) {
  const da = parseDateFromKey_(a);
  const db = parseDateFromKey_(b);

  if (da.getTime() !== db.getTime()) {
    return da.getTime() - db.getTime();
  }

  return a.localeCompare(b);
}

/**
 * Holt das Datum aus einem Key wie:
 * 04.05.2026|19|Name
 */
function parseDateFromKey_(key) {
  const dateStr = key.split("|")[0];
  const parts = dateStr.split(".");

  return new Date(Number(parts[2]), Number(parts[1]) - 1, Number(parts[0]));
}
