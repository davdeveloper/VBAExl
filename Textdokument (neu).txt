/**
 * ZENTRALE BERECHTIGUNG
 * Prüft differenziert nach Zugriff, Admin-Status und Dienstplaner-Status.
 */
function getUserPermissions(email) {
  if (!email) return { hasAccess: false, isAdmin: false, isDienstplaner: false };

  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const searchEmail = String(email).toLowerCase().trim();

  let hasAccess = false;
  let isAdmin = false;
  let isDienstplaner = false;

  const sheetTS = ss.getSheetByName('Tagessteuerung');

  if (sheetTS && sheetTS.getLastRow() > 1) {
    const data = sheetTS.getRange(2, 1, sheetTS.getLastRow() - 1, 2).getValues();

    for (let i = 0; i < data.length; i++) {
      const rowEmail = String(data[i][0]).toLowerCase().trim();

      if (rowEmail === searchEmail) {
        hasAccess = true;

        const rowRole = String(data[i][1] || "").toLowerCase().trim();

        if (rowRole.includes('tagessteuerung') || rowRole.includes('dienstplanung')) {
          isAdmin = true;
        }

        if (rowRole.includes('dienstplanung')) {
          isDienstplaner = true;
        }
      }
    }
  }

  if (!isAdmin) {
    const sheetDB = ss.getSheetByName('Data_DB');

    if (sheetDB && sheetDB.getLastRow() > 1) {
      const lastRow = sheetDB.getLastRow();
      const dataDB = sheetDB.getRange(2, 1, lastRow - 1, 14).getValues();

      const isDienstplanerInDB = dataDB.some(row => {
        const dbEmail = String(row[13] || "").trim().toLowerCase();
        const skill = String(row[4] || "").trim().toLowerCase();

        return dbEmail === searchEmail && skill === 'dienstplanung';
      });

      if (isDienstplanerInDB) {
        hasAccess = true;
        isAdmin = true;
        isDienstplaner = true;
      }
    }
  }

  return { hasAccess, isAdmin, isDienstplaner };
}

/**
 * Hauptfunktion: Daten laden.
 */
function getData() {
  try {
    const userEmail = Session.getActiveUser().getEmail().toLowerCase();
    const permissions = getUserPermissions(userEmail);

    if (!permissions.hasAccess) {
      throw new Error("⛔ Zugriff verweigert: Ihre E-Mail (" + userEmail + ") ist nicht berechtigt.");
    }

    const ss = SpreadsheetApp.getActiveSpreadsheet();

    let sheetDB = ss.getSheetByName('Data_DB');
    let sheetAuftrag = ss.getSheetByName('Auftrag');
    let sheetTS = ss.getSheetByName('Tagessteuerung');
    let sheetSchicht = ss.getSheetByName('Schichtmodelle');
    let sheetAbbau = ss.getSheetByName('AbbauKW');

    if (!sheetDB) throw new Error('Blatt "Data_DB" fehlt.');

    if (!sheetAuftrag) {
      sheetAuftrag = ss.insertSheet('Auftrag');
      sheetAuftrag.appendRow(['Name', 'Vorname', 'Datum', 'Stunden', 'Notizen', 'Erstellt von']);
    }

    if (!sheetSchicht) {
      sheetSchicht = ss.insertSheet('Schichtmodelle');
      sheetSchicht.appendRow(['Nachname', 'Vorname', 'Schichtzeit', 'Typ', 'Pause', 'Taetigkeiten']);
    }

    processAuftragExpiry(sheetDB, sheetAuftrag);

    const lastRow = sheetDB.getLastRow();

    if (lastRow < 2) {
      return {
        employees: [],
        stats: {},
        currentWeek: getWeekNumber(new Date()),
        currentYear: new Date().getFullYear(),
        isAdmin: permissions.isAdmin,
        isDienstplaner: permissions.isDienstplaner,
        abbauHistory: []
      };
    }

    const data = sheetDB.getRange(2, 1, lastRow - 1, 14).getValues();
    const rawAuftraege = getAuftraege(sheetAuftrag);

    const tsNames = new Set();

    if (sheetTS && sheetTS.getLastRow() > 1) {
      const tsData = sheetTS.getRange(2, 3, sheetTS.getLastRow() - 1, 1).getValues();

      tsData.forEach(r => {
        const full = String(r[0]).trim();

        if (full) {
          tsNames.add(full.split(' ')[0].toLowerCase());
        }
      });
    }

    const shiftMap = new Map();

    if (sheetSchicht.getLastRow() > 1) {
      const lastColS = sheetSchicht.getLastColumn();
      const colsToLoad = Math.min(lastColS, 6);

      if (colsToLoad >= 1) {
        const sData = sheetSchicht
          .getRange(2, 1, sheetSchicht.getLastRow() - 1, colsToLoad)
          .getValues();

        sData.forEach(r => {
          const key = (String(r[1] || "").trim() + " " + String(r[0] || "").trim()).toLowerCase();

          let activitiesArr = [];
          let detailsArr = [];

          const rawActivities = colsToLoad >= 6 ? String(r[5] || "").trim() : "";

          if (rawActivities) {
            try {
              const parsed = JSON.parse(rawActivities);

              if (Array.isArray(parsed)) {
                detailsArr = parsed
                  .map(item => ({
                    time: String(item.time || "").trim(),
                    activity: String(item.activity || "").trim()
                  }))
                  .filter(item => item.time || item.activity);

                detailsArr.forEach(item => {
                  const activity = item.activity.trim();

                  if (
                    activity &&
                    activity !== '-' &&
                    !activity.toLowerCase().includes('pause') &&
                    !activitiesArr.includes(activity)
                  ) {
                    activitiesArr.push(activity);
                  }
                });
              }
            } catch (jsonErr) {
              const parts = rawActivities.split(/(?=\d{2}:\d{2}\s*-\s*\d{2}:\d{2})/);

              parts.forEach(p => {
                const match = p.match(/(\d{2}:\d{2}\s*-\s*\d{2}:\d{2})\s+(.+)/);

                if (match) {
                  const time = match[1].replace(/\s/g, '').trim();
                  const activity = match[2].replace(/\n/g, '').trim();

                  if (activity) {
                    detailsArr.push({
                      time: time,
                      activity: activity
                    });

                    if (
                      !activity.toLowerCase().includes('pause') &&
                      !activitiesArr.includes(activity)
                    ) {
                      activitiesArr.push(activity);
                    }
                  }
                }
              });

              if (activitiesArr.length === 0) {
                activitiesArr = rawActivities
                  .split(',')
                  .map(a => a.trim())
                  .filter(a => a);
              }
            }
          }

          shiftMap.set(key, {
            time: colsToLoad >= 3 ? String(r[2] || "") : "",
            type: colsToLoad >= 4 ? String(r[3] || "") : "",
            pause: colsToLoad >= 5 ? String(r[4] || "") : "",
            activities: activitiesArr,
            details: detailsArr
          });
        });
      }
    }

    let abbauRaw = [];

    if (sheetAbbau && sheetAbbau.getLastRow() > 1) {
      const rows = sheetAbbau.getDataRange().getValues();
      const headers = rows[0];

      for (let r = 1; r < rows.length; r++) {
        const rowData = rows[r];
        const rYear = rowData[0];
        const rKw = rowData[1];

        for (let c = 3; c < rowData.length; c++) {
          const val = Number(rowData[c]);
          const teamName = String(headers[c]);

          if (val !== 0 && teamName) {
            abbauRaw.push({
              y: rYear,
              kw: rKw,
              val: val,
              team: teamName,
              tma: ""
            });
          }
        }
      }
    }

    const formattedData = data.map((row, index) => {
      if (!row[0]) return null;

      const austrittRaw = row[12];
      let isAustritt = false;

      if (austrittRaw) {
        let aDate = null;

        if (austrittRaw instanceof Date) {
          aDate = austrittRaw;
        } else if (typeof austrittRaw === 'string' && austrittRaw.trim() !== "") {
          const parts = extractDates(austrittRaw);

          if (parts.length > 0) {
            aDate = parts[0];
          }
        }

        if (aDate) {
          aDate.setHours(0, 0, 0, 0);

          const today = new Date();
          today.setHours(0, 0, 0, 0);

          if (aDate <= today) {
            isAustritt = true;
          }
        }
      }

      if (isAustritt) return null;

      const nachname = String(row[0] || "").trim();
      const vorname = String(row[1] || "").trim();
      const dbKey = (vorname + " " + nachname).toLowerCase();

      const myAuftraege = rawAuftraege
        .filter(a =>
          a.name.toLowerCase() === nachname.toLowerCase() &&
          a.vorname.toLowerCase() === vorname.toLowerCase()
        )
        .map(a => {
          let isTs = false;

          if (a.erstelltVon) {
            const creatorFirst = a.erstelltVon.split(' ')[0].toLowerCase();

            if (tsNames.has(creatorFirst)) {
              isTs = true;
            }
          }

          return { ...a, isTsRelevant: isTs };
        });

      const rawEss = row[5];
      const essDecimal = timeToDecimal(rawEss);
      const plannedDecimal = myAuftraege.reduce((sum, order) => sum + (order.stunden || 0), 0);

      let saldoDecimal = essDecimal - plannedDecimal;
      saldoDecimal = Math.round(saldoDecimal * 100) / 100;

      const shiftInfo = shiftMap.get(dbKey) || {
        time: "",
        type: "",
        pause: "",
        activities: [],
        details: []
      };

      const austrittDisplay = row[12] instanceof Date ? formatDateCompat(row[12]) : "";

      return {
        id: index + 2,
        nachname: nachname,
        vorname: vorname,
        fullName: `${vorname} ${nachname}`,
        tmaVorname: String(row[3] || "Ohne TMA"),
        skill: String(row[4] || "-"),
        stundenEss: rawEss,
        stundenSaldo: saldoDecimal,
        letzteAktualisierung: row[6] ? formatDateCompat(row[6]) : "-",
        notizen: String(row[11] || ""),
        austrittDatum: austrittDisplay,
        auftraege: myAuftraege,
        schicht: shiftInfo
      };
    }).filter(item => item !== null);

    return {
      employees: formattedData,
      currentWeek: getWeekNumber(new Date()),
      currentYear: new Date().getFullYear(),
      isAdmin: permissions.isAdmin,
      isDienstplaner: permissions.isDienstplaner,
      abbauHistory: abbauRaw
    };

  } catch (e) {
    Logger.log("Fehler in getData: " + e.toString());
    throw e;
  }
}

/**
 * Cleanup Logic.
 */
function processAuftragExpiry(sheetDB, sheetAuftrag) {
  const lastRowA = sheetAuftrag.getLastRow();

  if (lastRowA < 2) return;

  const dbDataRange = sheetDB.getDataRange();
  const dbData = dbDataRange.getValues();

  const today = new Date();
  today.setHours(0, 0, 0, 0);

  const dbMap = new Map();

  for (let i = 1; i < dbData.length; i++) {
    const key = (String(dbData[i][1]).trim() + " " + String(dbData[i][0]).trim()).toLowerCase();
    dbMap.set(key, i);
  }

  const auftragData = sheetAuftrag.getRange(2, 1, lastRowA - 1, 6).getValues();

  let dbDirty = false;
  let auftragDirty = false;

  const validAuftraege = [];

  auftragData.forEach(row => {
    const n = String(row[0]);
    const v = String(row[1]);
    const rawDate = row[2];
    const hoursVal = row[3];
    const notiz = row[4];
    const creator = row[5];

    let dates = [];

    if (rawDate instanceof Date) {
      dates.push(rawDate);
    } else {
      dates = extractDates(String(rawDate));
    }

    let isExpired = false;

    if (dates.length > 0) {
      const lastDate = dates[dates.length - 1];
      lastDate.setHours(0, 0, 0, 0);

      if (lastDate < today) {
        isExpired = true;
      }
    }

    let processed = false;

    if (isExpired) {
      const dbKey = (v.trim() + " " + n.trim()).toLowerCase();

      if (dbMap.has(dbKey)) {
        const dbIdx = dbMap.get(dbKey);

        const currentEssStr = dbData[dbIdx][5];
        const currentEssDec = timeToDecimal(currentEssStr);
        const orderHoursDec = timeToDecimal(hoursVal);
        const newEssDec = currentEssDec - orderHoursDec;

        dbData[dbIdx][5] = decimalToTime(newEssDec);
        dbData[dbIdx][6] = today;

        dbDirty = true;
        processed = true;
        auftragDirty = true;
      }
    }

    if (!processed) {
      validAuftraege.push([n, v, rawDate, hoursVal, notiz, creator]);
    }
  });

  if (dbDirty) {
    const dataToWrite = dbData.slice(1);
    const rangeFG = sheetDB.getRange(2, 6, dataToWrite.length, 2);

    rangeFG.setNumberFormat("@");

    const valuesFG = rangeFG.getValues();

    let rangeDirty = false;

    for (let i = 0; i < dataToWrite.length; i++) {
      const newF = dataToWrite[i][5];
      const newG = dataToWrite[i][6];

      if (valuesFG[i][0] !== newF) {
        valuesFG[i][0] = newF;
        valuesFG[i][1] = newG;
        rangeDirty = true;
      }
    }

    if (rangeDirty) {
      rangeFG.setValues(valuesFG);
    }
  }

  if (auftragDirty) {
    sheetAuftrag.getRange(2, 1, lastRowA - 1, 6).clearContent();

    if (validAuftraege.length > 0) {
      sheetAuftrag.getRange(2, 1, validAuftraege.length, 6).setValues(validAuftraege);
    }

    SpreadsheetApp.flush();
  }
}

function getAuftraege(sheet) {
  const lastRow = sheet.getLastRow();

  if (lastRow < 2) return [];

  const data = sheet.getRange(2, 1, lastRow - 1, 6).getValues();

  return data.map((row, idx) => ({
    rowId: idx + 2,
    name: String(row[0]),
    vorname: String(row[1]),
    datum: formatDateCompat(row[2]),
    stunden: timeToDecimal(row[3]),
    stundenDisplay: row[3],
    notizen: String(row[4]),
    erstelltVon: String(row[5] || "")
  }));
}

function addAuftrag(data) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();

  let sheetAuftrag = ss.getSheetByName('Auftrag');

  if (!sheetAuftrag) {
    sheetAuftrag = ss.insertSheet('Auftrag');
  }

  const userEmail = Session.getActiveUser().getEmail();
  const editorName = getEditorNameFromEmail(userEmail);
  const lastRow = sheetAuftrag.getLastRow() + 1;

  sheetAuftrag.getRange(lastRow, 4).setNumberFormat("@");

  sheetAuftrag.getRange(lastRow, 1, 1, 6).setValues([[
    data.nachname.trim(),
    data.vorname.trim(),
    data.datum,
    data.stunden,
    data.notizen,
    editorName
  ]]);

  const team = getEmployeeTeam(data.nachname, data.vorname);

  updateAbbauStats(data.datum, timeToDecimal(data.stunden), 'add', team);

  SpreadsheetApp.flush();

  return getData();
}

function deleteAuftrag(rowId) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName('Auftrag');

  if (sheet) {
    const r = Number(rowId);
    const val = sheet.getRange(r, 1, 1, 6).getValues()[0];
    const team = getEmployeeTeam(val[0], val[1]);

    updateAbbauStats(val[2], timeToDecimal(val[3]), 'remove', team);

    sheet.deleteRow(r);
  }

  return getData();
}

function updateEmployeeBase(formData) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName('Data_DB');
  const row = parseInt(formData.id, 10);

  const essCell = sheet.getRange(row, 6);

  essCell.setNumberFormat("@");
  essCell.setValue(formData.stundenEss);

  sheet.getRange(row, 7).setValue(new Date()).setNumberFormat("dd.MM.yyyy");
  sheet.getRange(row, 12).setValue(formData.notizen);

  if (formData.austrittDatum) {
    const parts = formData.austrittDatum.split('-');
    const d = new Date(parts[0], parts[1] - 1, parts[2]);

    sheet.getRange(row, 13).setValue(d).setNumberFormat("dd.MM.yyyy");
  } else {
    sheet.getRange(row, 13).setValue("");
  }

  SpreadsheetApp.flush();

  return getData();
}

function getEditorNameFromEmail(email) {
  if (!email) return "";

  try {
    const localPart = email.split('@')[0];
    const parts = localPart.split('.');

    if (parts.length > 0) {
      const firstNamePart = parts[0].replace(/_/g, ' ');

      return firstNamePart.replace(/\b\w/g, l => l.toUpperCase());
    }

    return email;
  } catch (e) {
    return email;
  }
}

function extractDates(str) {
  const regex = /(\d{1,2})\.(\d{1,2})\.(\d{4})/g;
  const matches = [];

  let match;

  while ((match = regex.exec(str)) !== null) {
    matches.push(new Date(match[3], match[2] - 1, match[1]));
  }

  return matches.sort((a, b) => a - b);
}

function formatDateCompat(dateObj) {
  if (!dateObj) return "-";

  let d = null;

  if (dateObj instanceof Date) {
    d = dateObj;
  } else if (typeof dateObj === 'string') {
    if (dateObj.match(/[A-Za-z]{3}\s[A-Za-z]{3}\s\d{2}/) || dateObj.includes("GMT")) {
      const tryD = new Date(dateObj);

      if (!isNaN(tryD.getTime())) {
        d = tryD;
      } else {
        return dateObj;
      }
    } else {
      return dateObj;
    }
  }

  if (d && !isNaN(d.getTime())) {
    return Utilities.formatDate(d, Session.getScriptTimeZone(), "dd.MM.yyyy");
  }

  return String(dateObj);
}

function getWeekNumber(d) {
  d = new Date(Date.UTC(d.getFullYear(), d.getMonth(), d.getDate()));
  d.setUTCDate(d.getUTCDate() + 4 - (d.getUTCDay() || 7));

  const yearStart = new Date(Date.UTC(d.getUTCFullYear(), 0, 1));

  return Math.ceil((((d - yearStart) / 86400000) + 1) / 7);
}

function timeToDecimal(timeStr) {
  if (timeStr === null || timeStr === undefined || timeStr === "") return 0;

  if (typeof timeStr === 'number') return timeStr;

  if (timeStr instanceof Date) {
    const h = timeStr.getHours();
    const m = timeStr.getMinutes();

    return h + (m / 60);
  }

  timeStr = String(timeStr).trim();

  let isNegative = timeStr.startsWith("-");

  if (isNegative) {
    timeStr = timeStr.substring(1);
  }

  const parts = timeStr.split(':');

  if (parts.length < 2) {
    return parseFloat(timeStr) || 0;
  }

  const hours = parseInt(parts[0], 10);
  const minutes = parseInt(parts[1], 10);

  let decimal = hours + (minutes / 60);

  return isNegative ? -decimal : decimal;
}

function decimalToTime(decimalVal) {
  if (decimalVal === null || decimalVal === undefined || decimalVal === "") return "00:00";

  decimalVal = Number(decimalVal) || 0;

  const isNegative = decimalVal < 0;

  decimalVal = Math.abs(decimalVal);

  let hours = Math.floor(decimalVal);
  let minutes = Math.round((decimalVal - hours) * 60);

  if (minutes === 60) {
    hours += 1;
    minutes = 0;
  }

  const hStr = String(hours).padStart(2, '0');
  const mStr = String(minutes).padStart(2, '0');

  return (isNegative ? "-" : "") + hStr + ":" + mStr;
}

function updateAbbauStats(dateStr, hoursDecimal, operation, teamName) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();

  let sheet = ss.getSheetByName('AbbauKW');

  if (!sheet) {
    sheet = ss.insertSheet('AbbauKW');
    sheet.appendRow(['Jahr', 'KW', 'Gesamt']);
    sheet.setFrozenRows(1);
    sheet.setFrozenColumns(2);
  }

  let dateObj = dateStr instanceof Date ? dateStr : extractDates(String(dateStr))[0];

  if (!dateObj) return;

  const year = dateObj.getFullYear();
  const kw = getWeekNumber(dateObj);
  const team = teamName || "Sonstiges";

  const lastCol = sheet.getLastColumn();
  const headers = sheet.getRange(1, 1, 1, lastCol).getValues()[0];

  let colIndex = -1;

  for (let c = 0; c < headers.length; c++) {
    if (String(headers[c]).toLowerCase() === String(team).toLowerCase()) {
      colIndex = c + 1;
      break;
    }
  }

  if (colIndex === -1) {
    colIndex = lastCol + 1;
    sheet.getRange(1, colIndex).setValue(team).setFontWeight("bold");
  }

  const lastRow = sheet.getLastRow();

  let rowIndex = -1;

  if (lastRow > 1) {
    const timeData = sheet.getRange(2, 1, lastRow - 1, 2).getValues();

    for (let r = 0; r < timeData.length; r++) {
      if (timeData[r][0] == year && timeData[r][1] == kw) {
        rowIndex = r + 2;
        break;
      }
    }
  }

  if (rowIndex === -1) {
    sheet.appendRow([year, kw, 0]);
    rowIndex = sheet.getLastRow();
  }

  const cell = sheet.getRange(rowIndex, colIndex);
  const oldVal = Number(cell.getValue()) || 0;

  let newVal = oldVal;

  if (operation === 'add') {
    newVal += hoursDecimal;
  } else {
    newVal -= hoursDecimal;
  }

  newVal = Math.round(newVal * 100) / 100;

  cell.setValue(newVal);
}

function getEmployeeTeam(nachname, vorname) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheetDB = ss.getSheetByName('Data_DB');

  if (!sheetDB) return "Unbekannt";

  const data = sheetDB.getDataRange().getValues();

  for (let i = 1; i < data.length; i++) {
    if (
      String(data[i][0]).toLowerCase().trim() === String(nachname).toLowerCase().trim() &&
      String(data[i][1]).toLowerCase().trim() === String(vorname).toLowerCase().trim()
    ) {
      return String(data[i][4]);
    }
  }

  return "Unbekannt";
}

   KAPAZITÄTENÜBERSICHT TAGESSTEUERUNG
   Basis: Schichtmodelle aus Plano Backup
   Wichtig: Es werden nur echte Plano-Tätigkeits-/Skill-Zeitblöcke gezählt.
   Keine Fallback-Berechnung über die komplette Arbeitszeit, keine Pause.
   ========================================================================= */

function getCapacityOverviewData() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheetSchicht = ss.getSheetByName('Schichtmodelle');

  const now = new Date();
  const nowMinutes = now.getHours() * 60 + now.getMinutes();

  if (!sheetSchicht || sheetSchicht.getLastRow() < 2) {
    return {
      success: true,
      updatedAt: Utilities.formatDate(now, Session.getScriptTimeZone(), "dd.MM.yyyy HH:mm"),
      sourceRows: 0,
      totalHours: 0,
      remainingHours: 0,
      usedHours: 0,
      employeeCount: 0,
      activeNowCount: 0,
      foundSkills: [],
      areas: []
    };
  }

  const colsToLoad = Math.min(sheetSchicht.getLastColumn(), 6);
  const rows = sheetSchicht.getRange(2, 1, sheetSchicht.getLastRow() - 1, colsToLoad).getValues();

  const groupMap = new Map();

  let totalHoursAll = 0;
  let remainingHoursAll = 0;
  let activeNowCount = 0;
  let sourceSegments = 0;

  rows.forEach(row => {
    const rawActivities = colsToLoad >= 6 ? String(row[5] || "").trim() : "";
    const segments = parseCapacitySkillSegments(rawActivities);

    if (segments.length === 0) return;

    segments.forEach(segment => {
      const rawSkill = String(segment.activity || "").trim();

      if (!rawSkill || isCapacityIgnoredActivity(rawSkill)) return;

      const capacitySkill = mapCapacitySkillGroup(rawSkill);

      if (!capacitySkill || capacitySkill === "-" || capacitySkill.toLowerCase() === "ohne skill") return;

      const duration = calculateCapacitySegmentDuration(segment.start, segment.end, nowMinutes);

      if (duration.totalHours <= 0) return;

      if (!groupMap.has(capacitySkill)) {
        groupMap.set(capacitySkill, {
          area: capacitySkill,
          totalHours: 0,
          remainingHours: 0,
          usedHours: 0,
          activeNowCount: 0,
          originalSkills: new Set()
        });
      }

      const group = groupMap.get(capacitySkill);

      group.totalHours += duration.totalHours;
      group.remainingHours += duration.remainingHours;
      group.usedHours += duration.usedHours;
      group.originalSkills.add(rawSkill);

      if (duration.isActiveNow) {
        group.activeNowCount += 1;
        activeNowCount += 1;
      }

      totalHoursAll += duration.totalHours;
      remainingHoursAll += duration.remainingHours;
      sourceSegments += 1;
    });
  });

  const preferredOrder = [
    "1st Level Filiale",
    "2nd Level Filiale",
    "One 1st Level",
    "One 2nd Level"
  ];

  const areas = Array.from(groupMap.values()).map(group => ({
    area: group.area,
    totalHours: Math.round(group.totalHours * 100) / 100,
    remainingHours: Math.round(group.remainingHours * 100) / 100,
    usedHours: Math.round(group.usedHours * 100) / 100,
    activeNowCount: group.activeNowCount,
    originalSkills: Array.from(group.originalSkills).sort()
  })).sort((a, b) => {
    const aIdx = preferredOrder.indexOf(a.area);
    const bIdx = preferredOrder.indexOf(b.area);

    if (aIdx !== -1 && bIdx !== -1) return aIdx - bIdx;
    if (aIdx !== -1) return -1;
    if (bIdx !== -1) return 1;

    return a.area.localeCompare(b.area);
  });

  return {
    success: true,
    updatedAt: Utilities.formatDate(now, Session.getScriptTimeZone(), "dd.MM.yyyy HH:mm"),
    sourceRows: sourceSegments,
    totalHours: Math.round(totalHoursAll * 100) / 100,
    remainingHours: Math.round(remainingHoursAll * 100) / 100,
    usedHours: Math.round((totalHoursAll - remainingHoursAll) * 100) / 100,
    employeeCount: 0,
    activeNowCount: activeNowCount,
    foundSkills: areas.map(a => a.area),
    areas: areas
  };
}

function parseCapacitySkillSegments(rawActivities) {
  const segments = [];

  if (!rawActivities) return segments;

  try {
    const parsed = JSON.parse(rawActivities);

    if (Array.isArray(parsed)) {
      parsed.forEach(item => {
        const time = String(item.time || "").trim();
        const activity = String(item.activity || "").trim();

        if (!activity || isCapacityIgnoredActivity(activity)) return;

        const range = parseTimeRangeToMinutes(time);

        if (range) {
          segments.push({
            start: range.start,
            end: range.end,
            activity: activity
          });
        }
      });

      return segments;
    }
  } catch (e) {
    // Fallback auf Textparser
  }

  const regex = /(\d{1,2}:\d{2})\s*-\s*(\d{1,2}:\d{2})\s+([\s\S]*?)(?=\d{1,2}:\d{2}\s*-\s*\d{1,2}:\d{2}|$)/g;
  let match;

  while ((match = regex.exec(rawActivities)) !== null) {
    const time = `${match[1]}-${match[2]}`;
    const activity = String(match[3] || "").replace(/\n/g, " ").trim();

    if (!activity || isCapacityIgnoredActivity(activity)) continue;

    const range = parseTimeRangeToMinutes(time);

    if (range) {
      segments.push({
        start: range.start,
        end: range.end,
        activity: activity
      });
    }
  }

  return segments;
}

function calculateCapacitySegmentDuration(startMinutes, endMinutes, nowMinutes) {
  const totalMinutes = Math.max(0, endMinutes - startMinutes);

  let adjustedNow = nowMinutes;

  if (endMinutes > 1440 && adjustedNow < startMinutes) {
    adjustedNow += 1440;
  }

  let remainingMinutes = 0;
  let isActiveNow = false;

  if (adjustedNow < startMinutes) {
    remainingMinutes = totalMinutes;
  } else if (adjustedNow >= startMinutes && adjustedNow < endMinutes) {
    remainingMinutes = Math.max(0, endMinutes - adjustedNow);
    isActiveNow = true;
  }

  const usedMinutes = Math.max(0, totalMinutes - remainingMinutes);

  return {
    totalHours: totalMinutes / 60,
    remainingHours: remainingMinutes / 60,
    usedHours: usedMinutes / 60,
    isActiveNow: isActiveNow
  };
}

function parseTimeRangeToMinutes(timeRange) {
  if (!timeRange) return null;

  const match = String(timeRange).match(/(\d{1,2}):(\d{2})\s*-\s*(\d{1,2}):(\d{2})/);

  if (!match) return null;

  const start = Number(match[1]) * 60 + Number(match[2]);
  let end = Number(match[3]) * 60 + Number(match[4]);

  if (end <= start) {
    end += 1440;
  }

  return {
    start: start,
    end: end
  };
}

function isCapacityIgnoredActivity(activity) {
  const a = String(activity || "").toLowerCase().trim();

  return (
    !a ||
    a === "-" ||
    a.includes("pause") ||
    a.includes("break") ||
    a.includes("systemstörung") ||
    a.includes("systemstoerung") ||
    a.includes("störung") ||
    a.includes("stoerung") ||
    a.includes("meeting") ||
    a.includes("coaching") ||
    a.includes("coaching lks de") ||
    a.includes("schulung") ||
    a.includes("schulung lks") ||
    a.includes("paypal") ||
    a.includes("briefdruck") ||
    a.includes("andere tätigkeiten") ||
    a.includes("andere taetigkeiten") ||
    a.includes("trainer") ||
    a.includes("quality mng") ||
    a.includes("wissen & training") ||
    a.includes("training") ||
    a.includes("bot")
  );
}

function mapCapacitySkillGroup(skill) {
  const original = String(skill || "").trim();
  const s = original.toLowerCase();

  if (!s || s === "-" || s === "ohne skill") return "";

  const hasFiliale = s.includes("filiale");
  const hasOnlineshop = s.includes("onlineshop") || s.includes("online shop");
  const hasLidlPlus = s.includes("lidl plus") || s.includes("lidlplus");
  const hasTelefon = s.includes("telefon");
  const has1st = s.includes("1st") || s.includes("first");
  const has2nd = s.includes("2nd") || s.includes("second");
  const hasOne = s.includes("one");
  const hasKleineLines = s.includes("kleine line") || s.includes("kleine lines") || s.includes("kleinelines");

  if (hasOne && has1st) {
    return "One 1st Level";
  }

  if (hasOne && has2nd) {
    return "One 2nd Level";
  }

  if (
    (has1st && (hasOnlineshop || hasLidlPlus)) ||
    (hasTelefon && (hasOnlineshop || hasLidlPlus))
  ) {
    return "One 1st Level";
  }

  if (
    has2nd &&
    (hasOnlineshop || hasLidlPlus)
  ) {
    return "One 2nd Level";
  }

  if (
    (has2nd && hasFiliale) ||
    hasKleineLines
  ) {
    return "2nd Level Filiale";
  }

  if (
    (has1st && hasFiliale) ||
    (hasTelefon && hasFiliale) ||
    s.includes("telefon filiale/lidl") ||
    (hasTelefon && s.includes("lidl") && !hasLidlPlus)
  ) {
    return "1st Level Filiale";
  }

  return original;
}

    let allActivities = new Set();

      if (activeControlSort === 'minus') return a.stundenSaldo - b.stundenSaldo;

      if (activeControlSort === 'time') {
        const timeA = parseStartTimeMinutes(getRelevantTimeStr(a, activeActivityFilter));
        const timeB = parseStartTimeMinutes(getRelevantTimeStr(b, activeActivityFilter));

        return timeA - timeB;
      }

      return 0;
    });

    sorted.forEach(e => {
      if (e.schicht && e.schicht.activities && e.schicht.activities.length > 0) {
        e.schicht.activities.forEach(act => allActivities.add(act));
      }
    });

    buildActivityFilters(Array.from(allActivities).sort());

    listFrueh.innerHTML = '';
    listMittel.innerHTML = '';
    listSpaet.innerHTML = '';

    let cF = 0;
    let cM = 0;
    let cS = 0;

    let sF = 0;
    let sM = 0;
    let sS = 0;

    sorted.forEach(e => {
      if (e.schicht && e.schicht.type) {
        let showItem = true;

        if (activeActivityFilter !== 'Alle') {
          if (!e.schicht.activities || !e.schicht.activities.includes(activeActivityFilter)) {
            showItem = false;
          }
        }

        if (showItem) {
          const col = getBalanceColor(e.stundenSaldo);

          let pauseBadge = "";

          if (e.schicht.pause) {
            pauseBadge = `<span class="bg-gray-100 text-gray-500 text-[9px] px-1.5 py-0.5 rounded mt-1 whitespace-nowrap inline-block" title="Pause"><i class="fa-solid fa-mug-hot mr-1"></i>${e.schicht.pause}</span>`;
          }

          let actHtml = "";

  function getRelevantTimeStr(emp, filter) {
    if (filter === 'Alle' || !emp.schicht || !emp.schicht.details || emp.schicht.details.length === 0) {
      return emp.schicht ? emp.schicht.time : "";
    }

    const matches = emp.schicht.details.filter(d => d.activity === filter);

    if (matches.length > 0) {
      if (matches.length === 1) {
        return matches[0].time;
      }

      let minStart = null;
      let maxEnd = null;
      let minStartMins = 99999;
      let maxEndMins = -1;

      matches.forEach(m => {
        const tMatch = m.time.match(/(\d{2}:\d{2})\s*-\s*(\d{2}:\d{2})/);

        if (tMatch) {
          const sStr = tMatch[1];
          const eStr = tMatch[2];

          const sMins = parseInt(sStr.split(':')[0], 10) * 60 + parseInt(sStr.split(':')[1], 10);
          let eMins = parseInt(eStr.split(':')[0], 10) * 60 + parseInt(eStr.split(':')[1], 10);

          if (eMins <= sMins) {
            eMins += 24 * 60;
          }

          if (sMins < minStartMins) {
            minStartMins = sMins;
            minStart = sStr;
          }

          if (eMins > maxEndMins) {
            maxEndMins = eMins;
            maxEnd = eStr;
          }
        }
      });

      if (minStart && maxEnd) {
        return `${minStart}-${maxEnd}`;
      }

      return matches.map(m => m.time).join(', ');
    }

    return emp.schicht.time;
  }

  function parseStartTimeMinutes(timeStr) {
    if (!timeStr) return 9999;

    const match = String(timeStr).match(/(\d{2}):(\d{2})/);

    if (match) {
      return parseInt(match[1], 10) * 60 + parseInt(match[2], 10);
    }

    return 9999;
  }

function getTagesplanungData() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const today = new Date();
  today.setHours(0, 0, 0, 0);

  ensureTagesplanungSheets_();

  const capacity = getCapacityOverviewData();
  const productivityMap = getTagesplanungProductivityMap_();
  const savedMap = getTagesplanungSavedMap_(today);
  const dbBacklogMap = getTagesplanungLatestBacklogMap_(today);

  const teams = TAGESPLANUNG_TEAM_ORDER.map(teamName => {
    const capacityArea = capacity.areas.find(a => a.area === teamName) || {};
    const saved = savedMap[teamName] || {};
    const dbBacklog = dbBacklogMap[teamName];

    const totalHours = Number(capacityArea.totalHours || 0);
    const remainingHours = Number(capacityArea.remainingHours || 0);
    const usedHours = Number(capacityArea.usedHours || 0);
    const productivity = saved.productivity !== undefined && saved.productivity !== ''
      ? Number(saved.productivity)
      : Number(productivityMap[teamName] || 5);

    const backlog = saved.backlog !== undefined && saved.backlog !== ''
      ? Number(saved.backlog)
      : Number(dbBacklog || 0);

    const manualInflow = Number(saved.manualInflow || 0);
    const manualOutput = Number(saved.manualOutput || 0);
    const note = String(saved.note || '');

    const possibleCases = Math.round(totalHours * productivity);
    const possibleRemainingCases = Math.round(remainingHours * productivity);
    const workload = backlog + manualInflow;
    const gap = possibleRemainingCases + manualOutput - workload;

    return {
      team: teamName,
      totalHours: round2_(totalHours),
      remainingHours: round2_(remainingHours),
      usedHours: round2_(usedHours),
      productivity: round2_(productivity),
      backlog: backlog,
      manualInflow: manualInflow,
      manualOutput: manualOutput,
      possibleCases: possibleCases,
      possibleRemainingCases: possibleRemainingCases,
      workload: workload,
      gap: gap,
      status: getTagesplanungStatus_(gap),
      note: note
    };
  });

  const payload = {
    success: true,
    date: formatDateIso_(today),
    dateDisplay: Utilities.formatDate(today, Session.getScriptTimeZone(), 'dd.MM.yyyy'),
    kw: getWeekNumber(today),
    updatedAt: Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'dd.MM.yyyy HH:mm'),
    teams: teams
  };

  writeTagesplanungCurrentState_(payload);

  return payload;
}

function importTagesplanungBacklogFromText(level, pastedText, refDate) {
  ensureTagesplanungSheets_();

  const dateObj = refDate ? parseIsoDate_(refDate) : new Date();
  dateObj.setHours(0, 0, 0, 0);

  const config = String(level) === '2nd'
    ? TAGESPLANUNG_BACKLOG_2ND_CONFIG
    : TAGESPLANUNG_BACKLOG_1ST_CONFIG;

  const teamBacklog = {};
  TAGESPLANUNG_TEAM_ORDER.forEach(t => teamBacklog[t] = 0);

  const lines = String(pastedText || '').split(/\r?\n/);
  let currentLabel = null;
  let processedCount = 0;

  lines.forEach(rawLine => {
    const line = String(rawLine || '').trim();

    if (!line) return;

    if (line.startsWith('DEU_')) {
      currentLabel = line;
      return;
    }

    if (currentLabel && /^\d+/.test(line)) {
      const match = line.match(/^(\d+)/);

      if (match) {
        const count = Number(match[1]) || 0;
        const mappedTeam = config[currentLabel] || null;

        if (mappedTeam) {
          teamBacklog[mappedTeam] = Number(teamBacklog[mappedTeam] || 0) + count;
          processedCount += 1;
        }
      }

      currentLabel = null;
    }
  });

  const existing = getTagesplanungData();
  existing.teams = existing.teams.map(team => {
    const imported = Number(teamBacklog[team.team] || 0);

    if (imported > 0) {
      team.backlog = imported;
    }

    return normalizeTagesplanungTeam_(team);
  });

  writeTagesplanungCurrentState_(existing);
  appendBacklogImportToDatabase_(dateObj, level, teamBacklog);

  return {
    success: true,
    message: `${String(level) === '2nd' ? '2nd' : '1st'} Level Backlog importiert.`,
    processedCount: processedCount,
    teamBacklog: teamBacklog,
    data: existing
  };
}

function normalizeTagesplanungTeam_(team) {
  const teamName = String(team.team || '').trim();
  const totalHours = Number(team.totalHours || 0);
  const remainingHours = Number(team.remainingHours || 0);
  const usedHours = Number(team.usedHours || Math.max(0, totalHours - remainingHours));
  const productivity = Number(team.productivity || 5);
  const backlog = Number(team.backlog || 0);
  const manualInflow = Number(team.manualInflow || 0);
  const manualOutput = Number(team.manualOutput || 0);
  const possibleCases = Math.round(totalHours * productivity);
  const possibleRemainingCases = Math.round(remainingHours * productivity);
  const workload = backlog + manualInflow;
  const gap = possibleRemainingCases + manualOutput - workload;

  return {
    team: teamName,
    totalHours: round2_(totalHours),
    remainingHours: round2_(remainingHours),
    usedHours: round2_(usedHours),
    productivity: round2_(productivity),
    backlog: backlog,
    manualInflow: manualInflow,
    manualOutput: manualOutput,
    possibleCases: possibleCases,
    possibleRemainingCases: possibleRemainingCases,
    workload: workload,
    gap: gap,
    status: getTagesplanungStatus_(gap),
    note: String(team.note || '')
  };
}

function getTagesplanungStatus_(gap) {
  if (gap >= 25) return 'Grün';
  if (gap >= 0) return 'Gelb';
  return 'Rot';
}

function formatDateIso_(dateObj) {
  return Utilities.formatDate(dateObj, Session.getScriptTimeZone(), 'yyyy-MM-dd');
}

function parseIsoDate_(iso) {
  if (iso instanceof Date) {
    return iso;
  }

  const parts = String(iso || '').split('-');

  if (parts.length === 3) {
    return new Date(Number(parts[0]), Number(parts[1]) - 1, Number(parts[2]));
  }

  return new Date();
}

function round2_(value) {
  return Math.round((Number(value) || 0) * 100) / 100;
}

function isSameDay_(d1, d2) {
  return d1.getFullYear() === d2.getFullYear() &&
    d1.getMonth() === d2.getMonth() &&
    d1.getDate() === d2.getDate();
}

  function collectDayPlanningInputs() {
    if (!dayPlanningState || !dayPlanningState.teams) return;

    dayPlanningState.teams.forEach((team, index) => {
      const fields = document.querySelectorAll(`[data-team-index="${index}"][data-planning-field]`);

      fields.forEach(field => {
        const key = field.getAttribute('data-planning-field');

        if (key === 'backlog' || key === 'productivity') {
          team[key] = Number(field.value || 0);
        } else {
          team[key] = field.value || '';
        }
      });

      team.possibleCases = Math.round(Number(team.planoHours || 0) * Number(team.productivity || 0));
      team.possibleRemainingCases = Math.round(Number(team.remainingHours || 0) * Number(team.productivity || 0));
      team.gap = Math.round(Number(team.possibleCases || 0) - Number(team.backlog || 0));
      team.remainingGap = Math.round(Number(team.possibleRemainingCases || 0) - Number(team.backlog || 0));
    });

  function recalculateDayPlanningTotals() {
    if (!dayPlanningState || !dayPlanningState.teams) return;

    const totals = dayPlanningState.teams.reduce((acc, team) => {
      acc.planoHours += Number(team.planoHours || 0);
      acc.remainingHours += Number(team.remainingHours || 0);
      acc.usedHours += Number(team.usedHours || 0);
      acc.backlog += Number(team.backlog || 0);
      acc.possibleCases += Number(team.possibleCases || 0);
      acc.possibleRemainingCases += Number(team.possibleRemainingCases || 0);
      return acc;
    }, {
      planoHours: 0,
      remainingHours: 0,
      usedHours: 0,
      backlog: 0,
      possibleCases: 0,
      possibleRemainingCases: 0
    });

    totals.gap = Math.round(totals.possibleCases - totals.backlog);
    totals.remainingGap = Math.round(totals.possibleRemainingCases - totals.backlog);
    dayPlanningState.totals = totals;
  }

  function planningDecimalToTime(decimalVal) {
    decimalVal = Number(decimalVal || 0);

    const isNegative = decimalVal < 0;
    decimalVal = Math.abs(decimalVal);

    let hours = Math.floor(decimalVal);
    let minutes = Math.round((decimalVal - hours) * 60);

    if (minutes === 60) {
      hours += 1;
      minutes = 0;
    }

    return `${isNegative ? '-' : ''}${String(hours).padStart(2, '0')}:${String(minutes).padStart(2, '0')}`;
  }

  function planningEscape(value) {
    return String(value || '')
      .replace(/&/g, '&amp;')
      .replace(/</g, '&lt;')
      .replace(/>/g, '&gt;')
      .replace(/"/g, '&quot;')
      .replace(/'/g, '&#039;');
  }

      function getTmaColorClass(tma) {
        const emps = allData.filter(d => d.tmaVorname === tma);

        if (emps.length === 0) return "bg-gray-50 border-gray-200 text-gray-400";

        let maxDays = 0;

        emps.forEach(e => {
          if (e.notizen && e.notizen.trim() !== "") return;

          if (e.letzteAktualisierung && e.letzteAktualisierung !== '-') {
            const parts = e.letzteAktualisierung.split('.');

            if (parts.length === 3) {
              const d = new Date(parts[2], parts[1] - 1, parts[0]);
              const diff = (new Date() - d) / (1000 * 60 * 60 * 24);

              if (diff > maxDays) {
                maxDays = diff;
              }
            }
          }
        });

        if (maxDays > 14) return "bg-red-600 text-white border-red-600 hover:opacity-90";
        if (maxDays > 7) return "bg-orange-100 text-orange-800 border-orange-300 hover:bg-orange-200";

        return "bg-emerald-50 text-emerald-700 border-emerald-200 hover:bg-emerald-100";
      }

        const set = type === 'team' ? selectedTeams : selectedTmas;

        if (set.has(val)) {
          set.delete(val);

          if (type === 'team') {
            element.classList.remove('active');
            element.classList.add('default');
          } else {
            element.classList.remove('ring-2', 'ring-blue-700', 'ring-offset-1');
          }
        } else {
          set.add(val);

          if (type === 'team') {
            element.classList.remove('default');
            element.classList.add('active');
          } else {
            element.classList.add('ring-2', 'ring-blue-700', 'ring-offset-1');
          }
        }

        applyFilters();
      }

      function applyFilters() {
        let filtered = allData.filter(d => {

          const tmaMatch = selectedTmas.size === 0 || selectedTmas.has(d.tmaVorname);

      function updateTmaAvailability() {
        const tmaChips = document.querySelectorAll('#filter-tma-container .filter-chip');

        if (selectedTeams.size === 0) {
          tmaChips.forEach(c => c.classList.remove('disabled'));
          return;
        }

        const validTmas = new Set();

        allData.forEach(d => {
          if (selectedTeams.has(d.skill) && d.tmaVorname) {
            validTmas.add(d.tmaVorname);
          }
        });

        tmaChips.forEach(chip => {
          if (validTmas.has(chip.dataset.value)) {
            chip.classList.remove('disabled');
          } else {
            chip.classList.add('disabled');

            if (selectedTmas.has(chip.dataset.value)) {
              selectedTmas.delete(chip.dataset.value);
              chip.classList.remove('ring-2', 'ring-blue-700', 'ring-offset-1');
            }
          }
        });
      }

      function renderStatsAndLists(data) {
        const currKW = currentWeekNum;
        const currY = currentYearNum;

        let lastKW = currKW - 1;
        let lastY = currY;

        if (lastKW < 1) {
          lastKW = 52;
          lastY--;
        }

        let nextKW = currKW + 1;
        let nextY = currY;

        if (nextKW > 52) {
          nextKW = 1;
          nextY++;
        }

        let sumLast = 0;
        let sumCurr = 0;
        let sumNext = 0;

        abbauHistoryData.forEach(h => {
          const teamMatch = selectedTeams.size === 0 || selectedTeams.has(h.team);

          if (teamMatch) {
            if (h.y == lastY && h.kw == lastKW) sumLast += Number(h.val) || 0;
            if (h.y == currY && h.kw == currKW) sumCurr += Number(h.val) || 0;
            if (h.y == nextY && h.kw == nextKW) sumNext += Number(h.val) || 0;
          }
        });

        const myKwStats = {
          lastKW: lastKW,
          lastVal: sumLast,
          currKW: currKW,
          currVal: sumCurr,
          nextKW: nextKW,
          nextVal: sumNext
        };

        const saldos = data.map(d => parseFloat(d.stundenSaldo) || 0);

        const critTotal = saldos.filter(h => h > 40).length;
        const critMid = saldos.filter(h => h >= 10 && h <= 40).length;
        const critLow = saldos.filter(h => h > 0 && h < 10).length;

        const minusArr = saldos.filter(h => h < 0);
        const minusSum = minusArr.reduce((a, b) => a + b, 0);

        const plusArr = saldos.filter(h => h > 0);
        const plusSum = plusArr.reduce((a, b) => a + b, 0);
        const avgPlus = plusArr.length > 0 ? plusSum / plusArr.length : 0;

        let totalAbbau = 0;
        let tsAbbau = 0;

        data.forEach(emp => {
          if (emp.auftraege) {
            emp.auftraege.forEach(o => {
              totalAbbau += Number(o.stunden) || 0;

              if (o.isTsRelevant && isDateInWeek(o.datum, currentWeekNum, currentYearNum)) {
                tsAbbau += Number(o.stunden) || 0;
              }
            });
          }
        });

        let targetId = 'dash-stats-container';

        if (currentPage === 'directory') {
          targetId = 'dir-stats-container';
        }

        if (currentPage === 'dashboard' || currentPage === 'directory') {
          updateStatValues(targetId, {
            critTotal,
            critMid,
            critLow,
            minusSum,
            minusCount: minusArr.length,
            plusSum,
            countPlus: plusArr.length,
            avgPlus,
            totalAbbau,
            tsAbbau,
            kwStats: myKwStats
          });
        }

        if (currentPage === 'dashboard') {
          renderDashboardLists(data);
        } else if (currentPage === 'directory') {
          renderDirectoryList(data);
        } else if (currentPage === 'control') {
          renderControl(data);
        }
      }

        if (!container) return;

        if (!container.innerHTML.trim()) {
          container.innerHTML = document.getElementById('stats-row-template').innerHTML;

        const setText = (key, val) => {

          if (el) {
            el.innerText = val;

        };

        setText('crit-total', stats.critTotal);
        setText('crit-mid', stats.critMid);
        setText('crit-low', stats.critLow);
        setText('avg-plus', decimalToTime(stats.avgPlus));
        setText('total-minus', decimalToTime(stats.minusSum));
        setText('count-minus', `(${stats.minusCount} Kol.)`);
        setText('total-plus', "+" + decimalToTime(stats.plusSum));
        setText('count-plus', `(${stats.countPlus} Kol.)`);

        if (stats.kwStats) {

        }

      function renderDashboardLists(data) {
        renderTable(
          'top15-plus-body',
          [...data].filter(d => d.stundenSaldo > 0).sort((a, b) => b.stundenSaldo - a.stundenSaldo).slice(0, 15)
        );

        renderTable(
          'top15-minus-body',
          [...data].filter(d => d.stundenSaldo < 0).sort((a, b) => a.stundenSaldo - b.stundenSaldo).slice(0, 15)
        );

        const grp = prop => {
          const g = {};

          data.forEach(d => {
            const n = d[prop] || "Ohne";

            if (!g[n]) {
              g[n] = { m: 0, p: 0, cP: 0 };
            }

            const v = parseFloat(d.stundenSaldo) || 0;

            if (v < 0) {
              g[n].m += v;
            } else if (v > 0) {
              g[n].p += v;
              g[n].cP++;
            }
          });

          return Object.keys(g)
            .map(k => ({ name: k, ...g[k] }))
            .sort((a, b) => b.p - a.p);
        };

        const rGrp = (id, arr) => {
          const tb = document.getElementById(id);

          if (!tb) return;

          tb.innerHTML = '';

          arr.forEach(x => {
            let avgStr = "-";

            if (x.p > 0 && x.cP > 0) {
              avgStr = decimalToTime(x.p / x.cP);
            }

            tb.innerHTML += `
              <tr class="hover:bg-gray-50 border-b border-gray-50 last:border-0 transition-colors">
                <td class="px-4 py-2 font-medium text-slate-700 truncate max-w-[120px] text-xs">${x.name}</td>
                <td class="px-4 py-2 text-right text-red-600 font-mono text-xs">${x.m ? decimalToTime(x.m) : '-'}</td>

                <td class="px-4 py-2 text-right text-gray-500 font-mono text-xs bg-gray-50/50">${avgStr}</td>
              </tr>`;
          });
        };

        rGrp('team-stats-body', grp('skill'));
        rGrp('tma-stats-body', grp('tmaVorname'));
      }

      function convertIsoToDe(dateStr) {
        if (!dateStr) return "";

        const parts = dateStr.split('-');

        if (parts.length !== 3) return dateStr;

        return `${parts[2]}.${parts[1]}.${parts[0]}`;
      }

      function getInitials(n) {
        return n ? n.split(' ').map(x => x[0]).join('').substring(0, 2).toUpperCase() : '?';
      }

      function isDateInWeek(dateStr, week, year) {
        if (!dateStr) return false;

        if (!String(dateStr).includes('-')) {
          const d = parseGermanDate(dateStr);

          if (!d) return false;

          return getWeekNumber(d) === week && d.getFullYear() === year;
        }

        const parts = String(dateStr).split('-');

        if (parts.length !== 2) return false;

        const d1 = parseGermanDate(parts[0].trim());
        const d2 = parseGermanDate(parts[1].trim());

        if (!d1 || !d2) return false;

        const w1 = getWeekNumber(d1);
        const w2 = getWeekNumber(d2);

        return (w1 === week && d1.getFullYear() === year) ||
               (w2 === week && d2.getFullYear() === year);
      }

      function parseGermanDate(str) {
        if (!str) return null;

        const parts = String(str).split('.');

        if (parts.length !== 3) return null;

        const d = new Date(parts[2], parts[1] - 1, parts[0]);

        return isNaN(d.getTime()) ? null : d;
      }

      function getWeekNumber(d) {
        d = new Date(Date.UTC(d.getFullYear(), d.getMonth(), d.getDate()));
        d.setUTCDate(d.getUTCDate() + 4 - (d.getUTCDay() || 7));

        const yearStart = new Date(Date.UTC(d.getUTCFullYear(), 0, 1));

        return Math.ceil((((d - yearStart) / 86400000) + 1) / 7);
      }

      function isOverdue(d) {
        if (!d || d === '-') return false;

        const p = String(d).split('.');

        if (p.length !== 3) return false;

        const checkDate = new Date(p[2], p[1] - 1, p[0]);

        if (isNaN(checkDate.getTime())) return false;

        return (new Date() - checkDate) / (1000 * 60 * 60 * 24) > 7;
      }

      function decimalToTime(decimalVal) {
        if (decimalVal === null || decimalVal === undefined || decimalVal === "") return "00:00";

        decimalVal = Number(decimalVal) || 0;

        const isNegative = decimalVal < 0;

        decimalVal = Math.abs(decimalVal);

        let hours = Math.floor(decimalVal);
        let minutes = Math.round((decimalVal - hours) * 60);

        if (minutes === 60) {
          hours += 1;
          minutes = 0;
        }

        const hStr = String(hours).padStart(2, '0');
        const mStr = String(minutes).padStart(2, '0');

        return (isNegative ? "-" : "") + hStr + ":" + mStr;
      }

      function getBalanceColor(saldo) {

        if (saldo >= 10) return 'text-orange-500 font-bold';
        if (saldo > 0) return 'text-modern-green font-bold';

        return 'text-slate-400';
