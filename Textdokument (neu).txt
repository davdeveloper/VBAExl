/**
 * Backend-Funktion: Liest alle Dateien aus der Ablage aus.
 */
function getFilesInFolder() {
  const fileList = [];
  try {
    const folder = DriveApp.getFolderById(FOLDER_ID);
    const files = folder.getFiles();
    
    while (files.hasNext()) {
      const file = files.next();
      fileList.push({ id: file.getId(), name: file.getName() });
    }
  } catch (e) {
    throw new Error("Konnte nicht auf den Ordner zugreifen. Hast du Berechtigungen für die geteilte Ablage? " + e.message);
  }
  
  fileList.sort((a, b) => a.name.localeCompare(b.name));
  return fileList;
}

/**
 * Holt das Mapping für die Skills/Tätigkeiten aus dem "Mapping"-Reiter.
 */
function getMapping(ss) {
  const mappingSheetName = "Mapping";
  let mappingSheet = ss.getSheetByName(mappingSheetName);
  
  if (!mappingSheet) {
    throw new Error("Das Arbeitsblatt 'Mapping' wurde nicht gefunden! Bitte lege es an und trage Spalte A (Plano) und Spalte B (Datenbank) ein.");
  }
  
  const mappingData = mappingSheet.getDataRange().getValues();
  const skillMap = {};
  for (let i = 0; i < mappingData.length; i++) {
    const planoSkill = normalizeStr(mappingData[i][0]);
    const dbSkill = String(mappingData[i][1]).trim();
    if (planoSkill && dbSkill && !planoSkill.toLowerCase().includes("plano tätigkeit")) {
      skillMap[planoSkill] = dbSkill;
    }
  }
  return skillMap;
}

/**

 */
function getNameMapping(ss) {
  const nameMappingSheetName = "Name_Mapping";
  let nameSheet = ss.getSheetByName(nameMappingSheetName);
  
  if (!nameSheet) {
    nameSheet = ss.insertSheet(nameMappingSheetName);

    nameSheet.autoResizeColumns(1, 2);
    return {};
  }
  
  const data = nameSheet.getDataRange().getValues();
  const nameMap = {};
  for (let i = 0; i < data.length; i++) {

    const dbName = normalizeStr(data[i][1]);

      nameMap[planoName] = dbName;
    }
  }
  return nameMap;
}

/**
 * Holt das lokale Team-Mapping aus dem "Team_Mapping"-Reiter (Überschreibt externe DB).
 */
function getTeamMapping(ss) {
  const teamMappingSheetName = "Team_Mapping";
  let teamSheet = ss.getSheetByName(teamMappingSheetName);
  
  if (!teamSheet) {
    teamSheet = ss.insertSheet(teamMappingSheetName);
    teamSheet.appendRow(["Mitarbeiter Name", "Team (z.B. One)"]);
    teamSheet.getRange("A1:B1").setFontWeight("bold");
    teamSheet.autoResizeColumns(1, 2);
    return {};
  }
  
  const data = teamSheet.getDataRange().getValues();
  const localTeamMap = {};
  for (let i = 0; i < data.length; i++) {
    const empName = normalizeStr(data[i][0]);
    const teamName = String(data[i][1]).trim();
    if (empName && teamName && !empName.toLowerCase().includes("mitarbeiter name")) {
      localTeamMap[empName] = teamName;
    }
  }
  return localTeamMap;
}

/**
 * Holt die Liste der zu ignorierenden Mitarbeiter aus dem "Ignore_Mapping"-Reiter.
 */
function getIgnoreMapping(ss) {
  const ignoreSheetName = "Ignore_Mapping";
  let ignoreSheet = ss.getSheetByName(ignoreSheetName);
  
  if (!ignoreSheet) {
    ignoreSheet = ss.insertSheet(ignoreSheetName);
    ignoreSheet.appendRow(["Zu ignorierende Namen (wie in Plano)"]);
    ignoreSheet.getRange("A1").setFontWeight("bold");
    ignoreSheet.autoResizeColumns(1, 1);
    return new Set();
  }
  
  const data = ignoreSheet.getDataRange().getValues();
  const ignoreSet = new Set();
  for (let i = 0; i < data.length; i++) {
    const name = normalizeStr(data[i][0]);
    if (name && !name.toLowerCase().includes("zu ignorierende")) {
      ignoreSet.add(name);
    }
  }
  return ignoreSet;
}

/**
 * Backend-Funktion: Zieht die Datei, wertet sie aus und schreibt sie ins Sheet.
 */
function processSelectedFile(fileId) {
  const targetSheetName = "Datenbank_KO_SKILL";
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  
  // --- MAPPINGS & TEAMS LADEN ---
  const skillMap = getMapping(ss);
  const planoSkillsSorted = Object.keys(skillMap).sort((a, b) => b.length - a.length);
  const nameMap = getNameMapping(ss);
  const localTeamMap = getTeamMapping(ss); 
  const ignoreSet = getIgnoreMapping(ss); 
  const teamMap = getEmployeeTeams();
  
  // Sets zum Tracken fehlender Zuordnungen
  const unmappedSkills = new Set();
  const unmappedEmployees = new Set();
  
  // Bestimme, unter welchem Namen ABW in der Datenbank gespeichert wird
  const abwDbName = (skillMap["ABW"] || "ABW").toUpperCase();
  
  // --- ZIEL-BLATT PRÜFEN/ERSTELLEN ---
  let targetSheet = ss.getSheetByName(targetSheetName);
  if (!targetSheet) {
    targetSheet = ss.insertSheet(targetSheetName);
    targetSheet.appendRow(["Datum", "KW", "Mitarbeitername"]);
    targetSheet.getRange("A1:C1").setFontWeight("bold");
  }

  // --- ZIEL-SPALTEN ERMITTELN & NEUE ERGÄNZEN ---
  const lastTargetCol = Math.max(targetSheet.getLastColumn(), 3);
  let targetHeaders = targetSheet.getRange(1, 1, 1, lastTargetCol).getValues()[0];
  
  const uniqueDbSkills = [...new Set(Object.values(skillMap))];
  
  if (!uniqueDbSkills.some(skill => skill.toUpperCase() === "DIFFERENZ")) uniqueDbSkills.push("Differenz");
  if (!uniqueDbSkills.some(skill => skill.toUpperCase() === "TEAM")) uniqueDbSkills.push("Team");
  if (!uniqueDbSkills.some(skill => skill.toUpperCase() === "SCHICHT")) uniqueDbSkills.push("Schicht");
  if (!uniqueDbSkills.some(skill => skill.toUpperCase() === "SOLL")) uniqueDbSkills.push("Soll");
  if (!uniqueDbSkills.some(skill => skill.toUpperCase() === "IST")) uniqueDbSkills.push("Ist");
  
  // NEU: Ausgleichstag Spalte ergänzen, falls nicht vorhanden
  if (!uniqueDbSkills.some(skill => skill.toUpperCase() === "AUSGLEICHSTAG")) uniqueDbSkills.push("Ausgleichstag");

  uniqueDbSkills.forEach(dbSkill => {
    let exists = targetHeaders.some(h => String(h).trim().toUpperCase() === dbSkill.toUpperCase());
    if (!exists) {
      targetHeaders.push(dbSkill);
      targetSheet.getRange(1, targetHeaders.length).setValue(dbSkill);
      targetSheet.getRange(1, targetHeaders.length).setFontWeight("bold");
    }
  });

  const file = DriveApp.getFileById(fileId);
  const mimeType = file.getMimeType();
  let data = [];

  // --- DATEI EINLESEN & EXCEL-KONVERTIERUNG (Drive API v3) ---
  try {
    if (mimeType === MimeType.GOOGLE_SHEETS) {
      data = SpreadsheetApp.openById(fileId).getSheets()[0].getDataRange().getValues();
    } else {
      const blob = file.getBlob();
      const textData = blob.getDataAsString("UTF-8");
      
      if (textData.includes("PK\x03\x04")) {
        try {
          const resource = { name: "Temp_" + file.getName(), mimeType: MimeType.GOOGLE_SHEETS };
          let tempFile;
          if (typeof Drive !== 'undefined') {
             tempFile = Drive.Files.create(resource, blob, {supportsAllDrives: true});
             const tempDoc = SpreadsheetApp.openById(tempFile.id);
             data = tempDoc.getSheets()[0].getDataRange().getValues();
             Drive.Files.remove(tempFile.id); 
          } else {
             throw new Error("Drive API nicht aktiviert.");
          }
        } catch(convertErr) {
          throw new Error("Fehler bei der Excel-Konvertierung! Detail: " + convertErr.message);
        }
      } else {
        if (textData.includes(";") && textData.split(";").length > textData.split(",").length) {
          data = Utilities.parseCsv(textData, ';');
        } else {
          data = Utilities.parseCsv(textData);
        }
      }
    }
  } catch (e) {
    throw new Error("Fehler beim Lesen der Datei: " + e.message);
  }

  if (!data || data.length < 4) throw new Error("Die Datei enthält nicht genug Daten.");

  // --- AUSWERTUNGS LOGIK ---
  let titleStr = "";
  let headerRowIndex = -1;
  
  for (let i = 0; i < Math.min(data.length, 30); i++) {
    const rowStr = data[i].join(" ");
    if (rowStr.includes("Wochenplan KW")) titleStr = rowStr;
    if (String(data[i][0]).trim() === "Name" || rowStr.includes("Dienstag,") || rowStr.includes("Montag,")) {
      headerRowIndex = i;
      break; 
    }
  }
  
  if (headerRowIndex === -1) throw new Error("Konnte die Datum-Kopfzeile in der Datei nicht finden.");

  const kwMatch = titleStr.match(/KW\s+(\d+)/);
  const yearMatch = titleStr.match(/\/\s+(\d{4})/);
  
  const kw = kwMatch ? kwMatch[1] : "";
  const year = yearMatch ? yearMatch[1] : new Date().getFullYear();

  const headerRow = data[headerRowIndex];
  const dateColumns = [];
  
  let differenzColIndex = headerRow.findIndex(h => String(h).trim() === 'Differenz');
  if (differenzColIndex === -1) differenzColIndex = 10; 
  
  let isthColIndex = headerRow.findIndex(h => String(h).trim() === 'Isth');
  if (isthColIndex === -1) isthColIndex = 12; 
  
  let vertragColIndex = headerRow.findIndex(h => String(h).trim() === 'Vertrag h');
  if (vertragColIndex === -1) vertragColIndex = 14; 

  for (let c = 0; c < headerRow.length; c++) {
    let val = headerRow[c];
    let valStr = (val instanceof Date) ? Utilities.formatDate(val, Session.getScriptTimeZone(), "dd.MM.yyyy") : String(val);

    if (valStr.match(/\d{2}\.\d{2}\./)) {
      const dateMatch = valStr.match(/(\d{2}\.\d{2}\.)/);
      if (dateMatch) {
        let dStr = dateMatch[1] + year;
        
        let parts = dStr.split('.');
        let dateObj = new Date(parts[2], parts[1] - 1, parts[0]);
        let isSaturday = (dateObj.getDay() === 6);

        dateColumns.push({ colIndex: c, date: dStr, isSaturday: isSaturday });
      }
    }
  }

  const records = {}; 

  let startRow = headerRowIndex + 2;
  for (let r = startRow; r < data.length; r++) {
    const nameCell = String(data[r][0]);
    
    if (nameCell.includes("Lidl Kundenservice DE")) {
      const planoCleanName = normalizeStr(nameCell.split('\n')[0]);
      
      // Wenn der Kollege im Ignore_Mapping steht, überspringen wir ihn komplett
      if (ignoreSet.has(planoCleanName)) {
        continue;
      }
      
      const mappedName = nameMap[planoCleanName] || planoCleanName;
      const assignedTeam = localTeamMap[mappedName] || localTeamMap[planoCleanName] || teamMap[mappedName];
      
      if (!assignedTeam) {
        unmappedEmployees.add(planoCleanName);
      }
      
      let differenzVal = parseGermanFloat(data[r][differenzColIndex]);
      let isthVal = parseGermanFloat(data[r][isthColIndex]);
      let vertragHVal = parseGermanFloat(data[r][vertragColIndex]);
      
      for (let dc of dateColumns) {
        const dayValueInNameRow = String(data[r][dc.colIndex]).trim();
        const upperDayVal = dayValueInNameRow.toUpperCase();
        
        const recKey = `${dc.date}|${kw}|${mappedName}`;
        
        let currentSollVal = dc.isSaturday ? isthVal : vertragHVal;
        
        if (!records[recKey]) {
          records[recKey] = { 
            date: dc.date, kw: kw, name: mappedName, skills: {}, 
            differenz: differenzVal, soll: currentSollVal, ist: 0, 
            urlaub: 0, abw: 0, isAnw: false, 
            schicht: dayValueInNameRow, team: assignedTeam || "Unbekannt"
          };
        }
        
        let isFullDayAbsence = false;

        // NEU: Prüfung auf Ausgleichstag (- im Plan, aber >0 Vertrag-Stunden und 0 Ist-Stunden)
        let isAusgleichstag = (upperDayVal === "-" && vertragHVal > 0 && isthVal === 0);

        // Logik für Urlaub, ABW, Überstunden, Wunschfrei UND Ausgleichstag
        if (upperDayVal.includes("URLAUB") || upperDayVal.includes("ABW") || upperDayVal.includes("ÜBERSTUNDEN") || upperDayVal.includes("WUNSCHFREI") || isAusgleichstag) {
          let isUrlaub = upperDayVal.includes("URLAUB");
          let isUeberstunden = upperDayVal.includes("ÜBERSTUNDEN");
          let isWunschfrei = upperDayVal.includes("WUNSCHFREI");
          
          let matchedPlanoSkill = "ABW";
          if (isUrlaub) matchedPlanoSkill = "Urlaub";
          if (isUeberstunden) matchedPlanoSkill = "Überstundenfrei";
          if (isWunschfrei) matchedPlanoSkill = "Wunschfrei";
          if (isAusgleichstag) matchedPlanoSkill = "Ausgleichstag";
          
          let dbSkill = skillMap[matchedPlanoSkill] || matchedPlanoSkill; 
          
          if (!skillMap[matchedPlanoSkill] && !isAusgleichstag) {
            unmappedSkills.add(matchedPlanoSkill);
          }
          
          let hoursToUse = ((isUrlaub || isUeberstunden || isWunschfrei) && !dc.isSaturday) ? vertragHVal : isthVal;
          if (isAusgleichstag) {
            hoursToUse = vertragHVal; // Beim Ausgleichstag immer die Vertragsstunden gutschreiben
            records[recKey].soll = 0; // FIX: Das SOLL für diesen Tag auf 0 setzen, damit es das Team-SOLL nicht verändert!
          }
          
          records[recKey].skills[dbSkill] = (records[recKey].skills[dbSkill] || 0) + hoursToUse;
          
          isFullDayAbsence = true;
          
          // Überstundenfrei, Wunschfrei UND Ausgleichstag generieren KEINE Ist-Stunden (0). Urlaub & ABW behalten die Ist-Stunden.
          if (isUeberstunden || isWunschfrei || isAusgleichstag) {
            records[recKey].ist = 0;
          } else {
            records[recKey].ist = isthVal; 
          }
        } else {
          records[recKey].ist = isthVal; 
        }
        
        if (upperDayVal.includes("ANW") || upperDayVal === "ANW") {
          records[recKey].isAnw = true;
        }

        if (!isFullDayAbsence && r + 1 < data.length) {
          const detailRow = data[r + 1];
          const detailCell = String(detailRow[dc.colIndex]);
          if (!detailCell || detailCell.trim() === "") continue;
          
          const lines = detailCell.split('\n').map(l => l.trim());
          let currentTimeRange = null;
          
          for (let i = 0; i < lines.length; i++) {
            const line = lines[i];
            if (line.match(/^\d{2}:\d{2}-\d{2}:\d{2}/)) {
              currentTimeRange = line.split(' ')[0]; 
            } else if (currentTimeRange) {
              let matchedDbSkill = null;
              for (const planoSkill of planoSkillsSorted) {
                if (normalizeStr(line).includes(planoSkill)) {
                  matchedDbSkill = skillMap[planoSkill];
                  break; 
                }
              }
              if (matchedDbSkill) {
                const hrs = calculateHours(currentTimeRange);
                records[recKey].skills[matchedDbSkill] = (records[recKey].skills[matchedDbSkill] || 0) + hrs;
              } else {
                if (line && line.trim().toUpperCase() !== "PAUSE" && !line.toUpperCase().includes("PAUSE") && line.trim().toUpperCase() !== "ANW") {
                  unmappedSkills.add(line.trim());
                }
              }
              currentTimeRange = null; 
            }
          }
        }
      }
    }
  }

  // --- WANDELN FÜR KO_SKILL ---
  const results = [];
  let addedRowsCount = 0;
  for (const key in records) {
    const rec = records[key];
    let rowData = new Array(targetHeaders.length).fill("");
    rowData[0] = rec.date; rowData[1] = rec.kw; rowData[2] = rec.name; 
    
    let hasSkillData = false;
    for (const dbSkill in rec.skills) {
      const colIndex = targetHeaders.findIndex(h => String(h).trim().toUpperCase() === dbSkill.toUpperCase());
      if (colIndex !== -1) { 
        // ABW in KO_SKILL auf exakte Minuten umrechnen und formatieren
        if (dbSkill.toUpperCase() === abwDbName) {
            let totalMins = Math.round(rec.skills[dbSkill] * 60);
            rowData[colIndex] = minutesToHHMM(totalMins);
        } else {
            rowData[colIndex] = rec.skills[dbSkill]; 
        }
        hasSkillData = true; 
      }
    }
    
    if (hasSkillData || rec.isAnw) {
      const setVal = (headerName, val) => {
        const idx = targetHeaders.findIndex(h => String(h).trim().toUpperCase() === headerName.toUpperCase());
        if (idx !== -1) rowData[idx] = val;
      };
      
      setVal("DIFFERENZ", rec.differenz); 
      // SOLL in KO_SKILL auf exakte Minuten umrechnen
      setVal("SOLL", minutesToHHMM(Math.round(rec.soll * 60))); 
      setVal("IST", rec.ist);
      setVal("TEAM", rec.team); 
      setVal("SCHICHT", rec.schicht);

      results.push(rowData);
      addedRowsCount++;
    }
  }

  const lastRow = targetSheet.getLastRow();
  if (lastRow > 1) targetSheet.getRange(2, 1, lastRow - 1, targetSheet.getLastColumn()).clearContent();
  if (results.length > 0) targetSheet.getRange(2, 1, results.length, results[0].length).setValues(results);

  // --- TEAM AGGREGATION FÜR DATENBANK_GESAMT ---
  const gesamtSheetName = "Datenbank_Gesamt";
  let gesamtSheet = ss.getSheetByName(gesamtSheetName);
  const gesamtHeaders = [...targetHeaders];
  gesamtHeaders[2] = "Team"; 
  
  if (!gesamtSheet) {
    gesamtSheet = ss.insertSheet(gesamtSheetName);
    gesamtSheet.appendRow(gesamtHeaders);
    gesamtSheet.getRange(1, 1, 1, gesamtHeaders.length).setFontWeight("bold");
  } else {
    if (gesamtSheet.getLastColumn() < gesamtHeaders.length) {
      gesamtSheet.getRange(1, 1, 1, gesamtHeaders.length).setValues([gesamtHeaders]);
      gesamtSheet.getRange(1, 1, 1, gesamtHeaders.length).setFontWeight("bold");
    }
  }

  const existingGesamtData = gesamtSheet.getDataRange().getValues();
  const existingKeys = new Set();
  for (let i = 1; i < existingGesamtData.length; i++) {
    let d = existingGesamtData[i][0];
    let dStr = (d instanceof Date) ? Utilities.formatDate(d, Session.getScriptTimeZone(), "dd.MM.yyyy") : String(d);
    existingKeys.add(`${dStr}|${String(existingGesamtData[i][2]).trim()}`);
  }

  // DUPLIKAT-PRÜFUNG & OVERWRITE DIALOG
  let overlapDates = new Set();
  for (const key in records) {
      let tName = teamMap[records[key].name] || "Ohne Team";
      if (existingKeys.has(`${records[key].date}|${tName}`)) {
          overlapDates.add(records[key].date);
      }
  }

  let overwrite = false;
  if (overlapDates.size > 0) {
      const ui = SpreadsheetApp.getUi();
      const datesStr = Array.from(overlapDates).join(", ");
      const res = ui.alert(
        "Duplikate gefunden", 
        `Für das/die Datum(e) ${datesStr} existieren bereits Einträge in 'Datenbank_Gesamt'.\n\nSollen die vorhandenen Einträge für diese Tage überschrieben und neu importiert werden?`, 
        ui.ButtonSet.YES_NO
      );
      overwrite = (res === ui.Button.YES);
      
      if (overwrite) {
          for (let i = existingGesamtData.length - 1; i >= 1; i--) {
              let d = existingGesamtData[i][0];
              let dStr = (d instanceof Date) ? Utilities.formatDate(d, Session.getScriptTimeZone(), "dd.MM.yyyy") : String(d);
              if (overlapDates.has(dStr)) {
                  gesamtSheet.deleteRow(i + 1);
              }
          }
          for (let key of existingKeys) {
              if (overlapDates.has(key.split('|')[0])) existingKeys.delete(key);
          }
      }
  }

  // Datensätze aggregieren (Mit präziser Minuten-Summierung)
  const teamRecords = {};
  for (const key in records) {
    const rec = records[key];
    const teamName = rec.team || "Ohne Team";
    const teamKey = `${rec.date}|${rec.kw}|${teamName}`;
    
    if (Object.keys(rec.skills).length === 0 && !rec.isAnw) continue;

    if (!teamRecords[teamKey]) {
      teamRecords[teamKey] = { date: rec.date, kw: rec.kw, team: teamName, skills: {}, differenz: 0, sollMins: 0, ist: 0 };
    }
    
    for (const dbSkill in rec.skills) {
      teamRecords[teamKey].skills[dbSkill] = (teamRecords[teamKey].skills[dbSkill] || 0) + rec.skills[dbSkill];
    }
    teamRecords[teamKey].differenz += (rec.differenz || 0);
    
    // Die Dezimalzahl jedes Mitarbeiters wird SOFORT in exakte Minuten umgewandelt und addiert.
    let employeeSollMins = Math.round((rec.soll || 0) * 60);
    teamRecords[teamKey].sollMins += employeeSollMins;
    
    teamRecords[teamKey].ist += (rec.ist || 0);
  }

  const teamResults = [];
  let addedTeamRowsCount = 0;

  for (const teamKey in teamRecords) {
    const tr = teamRecords[teamKey];
    if (existingKeys.has(`${tr.date}|${tr.team}`)) continue; 
    
    let rowData = new Array(gesamtHeaders.length).fill("");
    rowData[0] = tr.date; rowData[1] = tr.kw; rowData[2] = tr.team;
    
    for (const dbSkill in tr.skills) {
      const colIndex = gesamtHeaders.findIndex(h => String(h).trim().toUpperCase() === dbSkill.toUpperCase());
      if (colIndex !== -1) {
          // GANZ WICHTIG: ABW jetzt ebenfalls in HH:MM formatieren statt als Dezimalzahl!
          if (dbSkill.toUpperCase() === abwDbName) {
              let totalMins = Math.round(tr.skills[dbSkill] * 60);
              rowData[colIndex] = minutesToHHMM(totalMins);
          } else {
              rowData[colIndex] = tr.skills[dbSkill];
          }
      }
    }
    
    const setTeamVal = (headerName, val) => {
      const idx = gesamtHeaders.findIndex(h => String(h).trim().toUpperCase() === headerName.toUpperCase());
      if (idx !== -1) rowData[idx] = val;
    };
    
    setTeamVal("DIFFERENZ", tr.differenz);
    setTeamVal("SOLL", minutesToHHMM(tr.sollMins)); 
    setTeamVal("IST", tr.ist);

    teamResults.push(rowData);
    addedTeamRowsCount++;
  }

  if (teamResults.length > 0) {
    const lastGesamtRow = Math.max(gesamtSheet.getLastRow(), 1);
    gesamtSheet.getRange(lastGesamtRow + 1, 1, teamResults.length, teamResults[0].length).setValues(teamResults);
  }

  // --- NEU: DASHBOARD DATUM AUS DATEINAMEN AKTUALISIEREN ---
  let dashboardDate = "";
  const fileName = file.getName();
  // Sucht nach einem Datum im Format TT.MM.JJJJ oder TT.MM.JJ im Dateinamen
  const dateMatch = fileName.match(/\d{2}\.\d{2}\.\d{2,4}/); 
  
  if (dateMatch) {
    dashboardDate = dateMatch[0]; // Nimmt das Datum aus dem Dateinamen
  } else if (dateColumns.length > 0) {
    dashboardDate = dateColumns[0].date; // Fallback: Nimmt das erste Datum aus den Spalten
  }

  if (dashboardDate) {
    let dashboardSheet = ss.getSheetByName("Dashboard");
    if (dashboardSheet) {
      // Bisherige Daten
      dashboardSheet.getRange("B2").setValue(dashboardDate);
      dashboardSheet.getRange("C13").setValue(dashboardDate); 
      
      // NEU: Zusätzliche Tag-Felder
      dashboardSheet.getRange("C113").setValue(dashboardDate);
      
      // NEU: Kalenderwoche (KW) aus der Variable eintragen
      if (kw) {
        dashboardSheet.getRange("C30").setValue(kw);
        dashboardSheet.getRange("C130").setValue(kw);
      }
      
      // NEU: Monatsersten berechnen (z.B. aus 26.05.2026 wird 01.05.2026)
      let parts = dashboardDate.split(".");
      if (parts.length === 3) {
        let monthDate = "01." + parts[1] + "." + parts[2];
        dashboardSheet.getRange("C48").setValue(monthDate);
        dashboardSheet.getRange("C147").setValue(monthDate);
      }
    }
  }

  // --- FEHLENDE ZUORDNUNGEN PROTOKOLLIEREN ---
  const errorSheetName = "Fehlende_Zuordnungen";
  let errorSheet = ss.getSheetByName(errorSheetName);
  
  if (!errorSheet) {
    errorSheet = ss.insertSheet(errorSheetName);
  }
  errorSheet.clear();
  
  const errorData = [["Typ", "Fehlender Eintrag (Name / Tätigkeit)"]];
  
  unmappedEmployees.forEach(emp => errorData.push(["Mitarbeiter ohne Team", emp]));
  unmappedSkills.forEach(skill => errorData.push(["Unbekannte Tätigkeit (Plano)", skill]));
  
  if (errorData.length > 1) {
    errorSheet.getRange(1, 1, errorData.length, 2).setValues(errorData);
    errorSheet.getRange(1, 1, 1, 2).setFontWeight("bold");
    errorSheet.autoResizeColumns(1, 2);
    
    return `Erfolg! ${addedRowsCount} Mitarbeiter aktualisiert und ${addedTeamRowsCount} neue Teamsätze geschrieben.\n\n⚠️ WICHTIG: Es fehlen ${unmappedEmployees.size} Team-Zuordnungen und ${unmappedSkills.size} Tätigkeiten! Bitte prüfe den Reiter '${errorSheetName}'.`;
  } else {
    errorSheet.appendRow(["Perfekt! Alle Mitarbeiter und Tätigkeiten konnten erfolgreich zugeordnet werden."]);
    return `Erfolg! ${addedRowsCount} Mitarbeiter aktualisiert und ${addedTeamRowsCount} neue Teamsätze in Datenbank_Gesamt geschrieben. Alles wurde sauber gemappt!`;
  }
}

/**
 * Hilfsfunktion zur Normalisierung von Strings (entfernt überflüssige Leerzeichen).
 */
function normalizeStr(str) {
  if (!str) return "";
  return String(str).trim().replace(/\s+/g, ' ');
}

/**
 * Hilfsfunktion zur Berechnung der Dezimalstunden aus einem String wie "07:00-09:15"
 */
function calculateHours(timeRange) {
  try {
    const [start, end] = timeRange.split('-');
    const [h1, m1] = start.split(':').map(Number);
    const [h2, m2] = end.split(':').map(Number);
    
    let diffMinutes = (h2 * 60 + m2) - (h1 * 60 + m1);
    if (diffMinutes < 0) diffMinutes += 24 * 60;
    
    return diffMinutes / 60; 
  } catch (e) {
    return 0;
  }
}

/**
 * Hilfsfunktion zum Umwandeln von Zahlen-Strings in Float-Zahlen
 */
function parseGermanFloat(str) {
  if (!str) return 0;
  let s = String(str).trim().replace(',', '.');
  let f = parseFloat(s);
  return isNaN(f) ? 0 : f;
}

/**
 * Hilfsfunktion zum Umwandeln von Dezimalstunden in das Format HH:MM (z.B. aus 7.4 wird "7:24")
 */
function toHHMM(decimalHours) {
  if (!decimalHours) return "0:00";
  let sign = decimalHours < 0 ? "-" : "";
  let absDec = Math.abs(decimalHours);
  
  let h = Math.floor(absDec);
  let m = Math.round((absDec - h) * 60);
  
  if (m >= 60) {
    h += 1;
    m -= 60;
  }
  
  let mStr = m < 10 ? "0" + m : m;
  
  return sign + h + ":" + mStr;
}

/**
 * Hilfsfunktion, die addierte, exakte Minuten für Datenbank_Gesamt in das Format HH:MM umwandelt
 */
function minutesToHHMM(totalMinutes) {
  if (!totalMinutes) return "0:00";
  let sign = totalMinutes < 0 ? "-" : "";
  let absMins = Math.abs(totalMinutes);
  
  let h = Math.floor(absMins / 60);
  let m = Math.round(absMins % 60);
  
  let mStr = m < 10 ? "0" + m : m;
  
  return sign + h + ":" + mStr;
}

/**

 */
function processStundenerfassung(dateStr) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  
  // --- 1. DATUMS-FORMATIERUNGEN VORBEREITEN ---
  const dateObj = new Date(dateStr);
  const dateDDMMYYYY = Utilities.formatDate(dateObj, Session.getScriptTimeZone(), "dd.MM.yyyy");
  const dateYYYYMM = Utilities.formatDate(dateObj, Session.getScriptTimeZone(), "yyyyMM"); 
  const yearFull = dateObj.getFullYear().toString();
  const yearShort = yearFull.slice(-2);
  
  const monthsFull = ["Januar", "Februar", "März", "April", "Mai", "Juni", "Juli", "August", "September", "Oktober", "November", "Dezember"];
  const monthNameFull = monthsFull[dateObj.getMonth()];
  
  // Namen der Arbeitsblätter in den diversen Dateien
  const sheetNameTarget = monthNameFull + " " + yearFull;     
  const sheetNameSocial = yearFull;                           
  const sheetNameBot = monthNameFull + yearFull;              
  const sheetNameFW = monthNameFull + yearShort;              
  const sheetNameQM = monthNameFull + "_" + yearShort;        
  
  // Ein Array mit allen möglichen Schreibweisen für das Datum (für die Sonderzeitenlisten)
  const dateVariations = [
    dateDDMMYYYY, 
    dateObj.getDate() + "." + (dateObj.getMonth() + 1) + ".", 
    dateObj.getDate() + "." + (dateObj.getMonth() + 1), 
    ("0" + dateObj.getDate()).slice(-2) + "." + ("0" + (dateObj.getMonth() + 1)).slice(-2) + ".", 
    ("0" + dateObj.getDate()).slice(-2) + "." + ("0" + (dateObj.getMonth() + 1)).slice(-2) 
  ];

  
  // --- 2. VALIDIERUNG: DATUM IN DATENBANK_GESAMT? ---
  const dbGesamtSheet = ss.getSheetByName("Datenbank_Gesamt");
  if (!dbGesamtSheet) throw new Error("Arbeitsblatt 'Datenbank_Gesamt' nicht gefunden.");
  
  const dbGesamtRange = dbGesamtSheet.getDataRange();
  const dbData = dbGesamtRange.getValues();
  const dbDisplayData = dbGesamtRange.getDisplayValues(); 
  
  let dateFoundInDb = false;
  for (let i = 1; i < dbData.length; i++) {
    if (normalizeDateSearch(dbData[i][0]) === dateDDMMYYYY) {
      dateFoundInDb = true; break;
    }
  }
  if (!dateFoundInDb) {
    throw new Error(`Das Datum ${dateDDMMYYYY} ist in Datenbank_Gesamt nicht vorhanden. Bitte importiere zuerst das Plano Backup!`);
  }

  
  // --- 3. EINSTELLUNGEN LADEN & STUNDENERFASSUNG ÖFFNEN ---
  const einstellungenSheet = ss.getSheetByName("Einstellungen");
  if (!einstellungenSheet) throw new Error("Reiter 'Einstellungen' nicht gefunden!");

  let dashboardSheet = ss.getSheetByName("Dashboard");
  if (dashboardSheet) {

  let mappingStundenSheet = ss.getSheetByName("Mapping_Stundenerfassung");

  if (!backupSheet) {

/**
 * Hilfsfunktion: Wandelt Strings wie "07:24", "104,60" oder "14.5" sauber in Dezimalstunden um.
 */
function parseToDecimalHours(str) {
  if (!str) return 0;
  let s = String(str).trim();
  
  if (s.includes(':')) {
    let parts = s.split(':');
    let h = parseInt(parts[0], 10) || 0;
    let m = parseInt(parts[1], 10) || 0;
    let sec = parseInt(parts[2], 10) || 0;
    
    let sign = 1;
    if (s.startsWith('-')) {
      sign = -1;
      h = Math.abs(h);
    }
    return sign * (h + (m / 60) + (sec / 3600));
  }
  
  s = s.replace(',', '.');
  let f = parseFloat(s);
  return isNaN(f) ? 0 : f;
}

/**
 * Hilfsfunktion: Wandelt ein Sheets-Datum oder einen String verlässlich in dd.MM.yyyy um
 */
function normalizeDateSearch(cellVal) {
  if (!cellVal) return "";
  if (cellVal instanceof Date) {
    return Utilities.formatDate(cellVal, Session.getScriptTimeZone(), "dd.MM.yyyy");
  }
  return String(cellVal).trim();
}

/**
 * Hilfsfunktion: Wandelt Spaltenbuchstaben (z.B. "E" oder "DI") in den 0-basierten Index um
 */
function colLetterToIndex(letter) {
  let col = 0;
  let len = letter.length;
  for (let i = 0; i < len; i++) {
    col += (letter.toUpperCase().charCodeAt(i) - 64) * Math.pow(26, len - i - 1);
  }
  return col - 1; 
}
