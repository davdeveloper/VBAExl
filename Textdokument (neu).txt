<div class="grid grid-cols-1 gap-3">
            <label class="block">
              <span class="text-[10px] uppercase font-bold text-gray-400">Status</span>
              <select data-planning-field="status" data-team-index="${index}" onchange="handleDayPlanningInput(${index})" class="mt-1 w-full border border-gray-200 rounded-xl px-3 py-2 text-sm font-bold text-slate-700 focus:ring-2 focus:ring-lidl-blue outline-none bg-white">
                <option value="offen" ${status === 'offen' ? 'selected' : ''}>Offen</option>
                <option value="halten" ${status === 'halten' ? 'selected' : ''}>Halten möglich</option>
                <option value="abbauen" ${status === 'abbauen' ? 'selected' : ''}>Abbau möglich</option>
                <option value="risiko" ${status === 'risiko' ? 'selected' : ''}>Risiko</option>
              </select>
            </label>
            <label class="block">
              <span class="text-[10px] uppercase font-bold text-gray-400">Notiz</span>
              <textarea data-planning-field="note" data-team-index="${index}" oninput="handleDayPlanningInput(${index})" rows="2" class="mt-1 w-full border border-gray-200 rounded-xl px-3 py-2 text-sm text-slate-700 focus:ring-2 focus:ring-lidl-blue outline-none resize-none" placeholder="Kurznotiz für die Steuerung...">${planningEscape(note)}</textarea>
            </label>
          </div>
        </div>
      </div>
    `;
  }

  function handleDayPlanningInput(index) {
    collectDayPlanningInputs();
    recalculateDayPlanningCard(index);
    recalculateDayPlanningTotals();
    scheduleDayPlanningSave();
  }

  function collectDayPlanningInputs() {
    if (!dayPlanningState || !dayPlanningState.teams) return;

    dayPlanningState.teams.forEach((team, index) => {
      const fields = document.querySelectorAll(`[data-team-index="${index}"][data-planning-field]`);

      fields.forEach(field => {
        const key = field.getAttribute('data-planning-field');

        if (key === 'backlog' || key === 'productivity') {
          team[key] = Number(field.value || 0);
        } else {
          team[key] = field.value || '';
        }
      });

      team.possibleCases = Math.round(Number(team.planoHours || 0) * Number(team.productivity || 0));
      team.possibleRemainingCases = Math.round(Number(team.remainingHours || 0) * Number(team.productivity || 0));
      team.gap = Math.round(Number(team.possibleCases || 0) - Number(team.backlog || 0));
      team.remainingGap = Math.round(Number(team.possibleRemainingCases || 0) - Number(team.backlog || 0));
    });
  }

  function recalculateDayPlanningCard(index) {
    if (!dayPlanningState || !dayPlanningState.teams || !dayPlanningState.teams[index]) return;

    const team = dayPlanningState.teams[index];
    const possible = Math.round(Number(team.planoHours || 0) * Number(team.productivity || 0));
    const possibleRemaining = Math.round(Number(team.remainingHours || 0) * Number(team.productivity || 0));
    const gap = possible - Number(team.backlog || 0);

    const possibleEl = document.querySelector(`[data-planning-output="possible"][data-team-index="${index}"]`);
    const possibleRemainingEl = document.querySelector(`[data-planning-output="possibleRemaining"][data-team-index="${index}"]`);
    const gapEl = document.querySelector(`[data-planning-output="gap"][data-team-index="${index}"]`);

    if (possibleEl) possibleEl.innerText = possible;
    if (possibleRemainingEl) possibleRemainingEl.innerText = possibleRemaining;

    if (gapEl) {
      gapEl.innerText = `${gap >= 0 ? '+' : ''}${gap}`;
      gapEl.className = `font-mono font-bold ${gap >= 0 ? 'text-modern-green' : 'text-red-600'} text-lg`;
    }
  }

  function recalculateDayPlanningTotals() {
    if (!dayPlanningState || !dayPlanningState.teams) return;

    const totals = dayPlanningState.teams.reduce((acc, team) => {
      acc.planoHours += Number(team.planoHours || 0);
      acc.remainingHours += Number(team.remainingHours || 0);
      acc.usedHours += Number(team.usedHours || 0);
      acc.backlog += Number(team.backlog || 0);
      acc.possibleCases += Number(team.possibleCases || 0);
      acc.possibleRemainingCases += Number(team.possibleRemainingCases || 0);
      return acc;
    }, {
      planoHours: 0,
      remainingHours: 0,
      usedHours: 0,
      backlog: 0,
      possibleCases: 0,
      possibleRemainingCases: 0
    });

    totals.gap = Math.round(totals.possibleCases - totals.backlog);
    totals.remainingGap = Math.round(totals.possibleRemainingCases - totals.backlog);
    dayPlanningState.totals = totals;
  }

  function scheduleDayPlanningSave() {
    setDayPlanningSaveStatus('Änderungen erkannt – speichere gleich automatisch...');

    if (dayPlanningSaveTimer) {
      clearTimeout(dayPlanningSaveTimer);
    }

    dayPlanningSaveTimer = setTimeout(() => {
      saveDayPlanningNow(true);
    }, 900);
  }

  function saveDayPlanningNow(silent) {
    if (!dayPlanningState || dayPlanningIsSaving) return;

    collectDayPlanningInputs();
    dayPlanningIsSaving = true;

    if (!silent) {
      setDayPlanningSaveStatus('Speichere Tagesplanung...');
    }

    google.script.run
      .withSuccessHandler(res => {
        dayPlanningIsSaving = false;
        const savedAt = res && res.savedAt ? res.savedAt : 'gerade eben';
        setDayPlanningSaveStatus(`Gespeichert: ${savedAt}`);
        showDayPlanningSavePill();
      })
      .withFailureHandler(err => {
        dayPlanningIsSaving = false;
        setDayPlanningSaveStatus(`Fehler beim Speichern: ${err.message || err}`);
      })
      .saveDayPlanningData({
        dateKey: dayPlanningState.dateKey,
        teams: dayPlanningState.teams,
        backlogDetails: dayPlanningState.backlogDetails || {},
        saveReason: silent ? 'Auto-Speicherung' : 'Manuelle Speicherung'
      });
  }

  function openPlanningBacklogPasteModal(level) {
    dayPlanningBacklogLevel = level || '1st';

    const modal = document.getElementById('dayPlanningBacklogModal');
    const content = document.getElementById('dayPlanningBacklogModalContent');
    const subtitle = document.getElementById('day-planning-backlog-subtitle');
    const text = document.getElementById('day-planning-backlog-text');
    const result = document.getElementById('day-planning-backlog-result');
    const refDate = document.getElementById('day-planning-backlog-ref-date');

    if (!modal || !content) return;

    if (subtitle) subtitle.innerText = `${level === '2nd' ? '2nd' : '1st'} Level Backlog aus Salesforce einfügen oder aus Mapping übernehmen`;
    if (text) text.value = '';
    if (result) {
      result.classList.add('hidden');
      result.innerHTML = '';
    }
    if (refDate) {
      refDate.value = dayPlanningState && dayPlanningState.dateKey ? dayPlanningState.dateKey : new Date().toISOString().slice(0, 10);
    }

    modal.classList.remove('hidden');
    setTimeout(() => {
      modal.classList.remove('opacity-0');
      content.classList.remove('scale-95');
      content.classList.add('scale-100');
    }, 10);
  }

  function closePlanningBacklogPasteModal() {
    const modal = document.getElementById('dayPlanningBacklogModal');
    const content = document.getElementById('dayPlanningBacklogModalContent');

    if (!modal || !content) return;

    modal.classList.add('opacity-0');
    content.classList.remove('scale-100');
    content.classList.add('scale-95');

    setTimeout(() => {
      modal.classList.add('hidden');
    }, 250);
  }

  function submitPlanningBacklogText() {
    if (!dayPlanningState) return;

    const pastedText = document.getElementById('day-planning-backlog-text').value || '';
    const refDate = document.getElementById('day-planning-backlog-ref-date').value || dayPlanningState.dateKey;

    if (!pastedText.trim()) {
      showPlanningBacklogResult('Bitte füge zuerst Backlog-Text ein.', false);
      return;
    }

    showPlanningBacklogResult('Import läuft...', true);

    google.script.run
      .withSuccessHandler(res => {
        if (res && res.success) {
          dayPlanningState = res.data;
          renderDayPlanning(dayPlanningState);
          showPlanningBacklogResult(res.message || 'Backlog importiert.', true);
          saveDayPlanningNow(true);
          setTimeout(closePlanningBacklogPasteModal, 700);
        } else {
          showPlanningBacklogResult('Backlog konnte nicht importiert werden.', false);
        }
      })
      .withFailureHandler(err => {
        showPlanningBacklogResult(`Fehler beim Import: ${err.message || err}`, false);
      })
      .importDayPlanningBacklogFromText(dayPlanningBacklogLevel, pastedText, refDate);
  }

  function importPlanningBacklogFromMapping() {
    if (!dayPlanningState) return;

    showPlanningBacklogResult('Backlog wird aus Mapping übernommen...', true);

    google.script.run
      .withSuccessHandler(res => {
        if (res && res.success) {
          dayPlanningState = res.data;
          renderDayPlanning(dayPlanningState);
          showPlanningBacklogResult(res.message || 'Backlog aus Mapping übernommen.', true);
          saveDayPlanningNow(true);
          setTimeout(closePlanningBacklogPasteModal, 700);
        } else {
          showPlanningBacklogResult('Backlog konnte nicht aus Mapping übernommen werden.', false);
        }
      })
      .withFailureHandler(err => {
        showPlanningBacklogResult(`Fehler beim Mapping-Import: ${err.message || err}`, false);
      })
      .importDayPlanningBacklogFromMapping();
  }

  function showPlanningBacklogResult(message, success) {
    const result = document.getElementById('day-planning-backlog-result');

    if (!result) return;

    result.classList.remove('hidden');

    if (success) {
      result.className = 'rounded-xl border p-4 text-sm bg-emerald-50 border-emerald-200 text-emerald-700';
      result.innerHTML = `<i class="fa-solid fa-circle-check mr-2"></i>${message}`;
    } else {
      result.className = 'rounded-xl border p-4 text-sm bg-red-50 border-red-200 text-red-700';
      result.innerHTML = `<i class="fa-solid fa-triangle-exclamation mr-2"></i>${message}`;
    }
  }

  function setDayPlanningSaveStatus(text) {
    const el = document.getElementById('day-planning-save-status');

    if (el) {
      el.innerText = text;
    }
  }

  function showDayPlanningSavePill() {
    const pill = document.getElementById('day-planning-save-pill');

    if (!pill) return;

    pill.classList.remove('hidden');

    setTimeout(() => {
      pill.classList.add('hidden');
    }, 2500);
  }

  function planningDecimalToTime(decimalVal) {
    decimalVal = Number(decimalVal || 0);

    const isNegative = decimalVal < 0;
    decimalVal = Math.abs(decimalVal);

    let hours = Math.floor(decimalVal);
    let minutes = Math.round((decimalVal - hours) * 60);

    if (minutes === 60) {
      hours += 1;
      minutes = 0;
    }

    return `${isNegative ? '-' : ''}${String(hours).padStart(2, '0')}:${String(minutes).padStart(2, '0')}`;
  }

  function planningEscape(value) {
    return String(value || '')
      .replace(/&/g, '&amp;')
      .replace(/</g, '&lt;')
      .replace(/>/g, '&gt;')
      .replace(/"/g, '&quot;')
      .replace(/'/g, '&#039;');
  }
