=LET(
    prozent; C344*100;
    bereiche; TEXTSPLIT(TEXTVERKETTEN(",";WAHR;B330:B334); "-,");
    untergrenzen; WERT(INDEX(bereiche;;1));
    obergrenzen; WERT(INDEX(bereiche;;2));
    minwerte; C330:C334;
    maxwerte; D330:D334;
    index; XVERGLEICH(WAHR; (prozent >= untergrenzen)*(prozent <= obergrenzen); 0);
    minwert; INDEX(minwerte; index);
    maxwert; INDEX(maxwerte; index);
    minwert + (maxwert - minwert) * C344
)
