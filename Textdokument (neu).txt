  function getRelevantTimeStr(emp, filter) {
    if (filter === 'Alle' || !emp.schicht || !emp.schicht.details || emp.schicht.details.length === 0) {
      return emp.schicht ? emp.schicht.time : "";
    }

    const matches = emp.schicht.details.filter(d => d.activity === filter);

    if (matches.length > 0) {
      if (matches.length === 1) {
        return matches[0].time;
      }

      let minStart = null;
      let maxEnd = null;
      let minStartMins = 99999;
      let maxEndMins = -1;

      matches.forEach(m => {
        const tMatch = m.time.match(/(\d{2}:\d{2})\s*-\s*(\d{2}:\d{2})/);

        if (tMatch) {
          const sStr = tMatch[1];
          const eStr = tMatch[2];

          const sMins = parseInt(sStr.split(':')[0], 10) * 60 + parseInt(sStr.split(':')[1], 10);
          let eMins = parseInt(eStr.split(':')[0], 10) * 60 + parseInt(eStr.split(':')[1], 10);

          if (eMins <= sMins) {
            eMins += 24 * 60;
          }

          if (sMins < minStartMins) {
            minStartMins = sMins;
            minStart = sStr;
          }

          if (eMins > maxEndMins) {
            maxEndMins = eMins;
            maxEnd = eStr;
          }
        }
      });

      if (minStart && maxEnd) {
        return `${minStart}-${maxEnd}`;
      }

      return matches.map(m => m.time).join(', ');
    }

    return emp.schicht.time;
  }

  function parseStartTimeMinutes(timeStr) {
    if (!timeStr) return 9999;

    const match = String(timeStr).match(/(\d{2}):(\d{2})/);

    if (match) {
      return parseInt(match[1], 10) * 60 + parseInt(match[2], 10);
    }

    return 9999;
  }

          if (
            actLower.includes('pause') ||
            actLower.includes('meeting') ||
            actLower.includes('andere tätigkeiten') ||
            actLower.includes('andere taetigkeiten') ||
            actLower.includes('quality mng') ||
            actLower.includes('wissen & training') ||
            actLower.includes('bot')
          ) {
            return;
          }

          const match = String(det.time || "").match(/(\d{2}):(\d{2})/);

          if (match) {
            const startMins = parseInt(match[1], 10) * 60 + parseInt(match[2], 10);

            validDetails.push({
              activity: det.activity,
              timeStr: det.time,
              startMins: startMins
            });
          }
        });

        validDetails.sort((a, b) => a.startMins - b.startMins);

        for (let i = 0; i < validDetails.length; i++) {
          let det = validDetails[i];

          if (det.startMins > currentMins && det.startMins <= currentMins + 120) {
            if (i > 0) {
              let prevAct = validDetails[i - 1].activity;

              let cleanCurr = String(det.activity || "").trim().toLowerCase();
              let cleanPrev = String(prevAct || "").trim().toLowerCase();

              if (cleanCurr !== cleanPrev) {
                upcoming.push({
                  empId: emp.id,
                  empName: emp.fullName,
                  activity: det.activity,
                  timeStr: det.timeStr,
                  startMins: det.startMins,
                  diffMins: det.startMins - currentMins
                });

                break;
              }
            }
          }
        }
      }
    });

    upcoming.sort((a, b) => a.startMins - b.startMins);

    const list = document.getElementById('radar-list');
    const badge = document.getElementById('radar-badge');

    if (!list || !badge) return;

    if (upcoming.length === 0) {
      list.innerHTML = '<div class="p-6 text-center text-sm text-gray-400 italic"><i class="fa-solid fa-mug-hot text-2xl text-gray-300 mb-2 block"></i>Keine anstehenden Wechsel in den nächsten 2 Stunden.</div>';
      badge.classList.add('hidden');
      return;
    }

    const urgentCount = upcoming.filter(u => u.diffMins <= 15).length;

    if (urgentCount > 0) {
      badge.innerText = urgentCount;
      badge.classList.remove('hidden');
      badge.classList.add('animate-bounce');

      setTimeout(() => badge.classList.remove('animate-bounce'), 3000);
    } else {
      badge.classList.add('hidden');
    }

    let html = '';

    upcoming.slice(0, 10).forEach(u => {
      let colorClass = "text-lidl-blue bg-blue-50 border-blue-100";
      let textCol = "text-blue-800";
      let inText = `in ${u.diffMins} Min`;

      if (u.diffMins <= 5) {
        colorClass = "text-red-600 bg-red-50 border-red-200 animate-pulse";
        textCol = "text-red-800";
      } else if (u.diffMins <= 15) {
        colorClass = "text-orange-600 bg-orange-50 border-orange-200";
        textCol = "text-orange-800";
      }

      html += `
        <div onclick="openShiftDetailsModal(${u.empId})" class="p-3 border-b border-gray-100 last:border-0 hover:bg-gray-100 cursor-pointer transition">
          <div class="flex justify-between items-center mb-1">
            <span class="text-xs font-bold text-slate-700 truncate">${u.empName}</span>
            <span class="text-[10px] font-bold px-2 py-0.5 rounded-full border ${colorClass} shrink-0 ml-2">${inText}</span>
          </div>
          <div class="flex items-center text-[10px] text-gray-500 mt-1">
            <i class="fa-solid fa-arrow-right-arrow-left mr-1.5 text-gray-400"></i>
            <span class="truncate font-bold ${textCol}">${u.activity}</span>
            <span class="ml-auto font-mono text-gray-400 font-bold shrink-0">${String(u.timeStr).split('-')[0]} Uhr</span>
          </div>
        </div>`;
    });

    list.innerHTML = html;
  }

  function openCapacityOverviewModal(forceRefresh) {
    const modal = document.getElementById('capacityOverviewModal');
    const content = document.getElementById('capacity-overview-content');

    if (!modal || !content) return;

    modal.classList.remove('hidden');

    setTimeout(() => {
      modal.classList.remove('opacity-0');
      document.getElementById('capacityOverviewModalContent').classList.remove('scale-95');
      document.getElementById('capacityOverviewModalContent').classList.add('scale-100');
    }, 10);

    content.innerHTML = `
      <div class="flex flex-col items-center justify-center py-16 text-gray-400">
        <i class="fa-solid fa-circle-notch fa-spin text-4xl mb-3 text-lidl-blue"></i>
        <p class="font-medium">Kapazitäten werden berechnet...</p>
        <p class="text-xs mt-1">Basis: aktuelles Schichtmodell aus dem Plano Backup</p>
      </div>
    `;

    capacitySkillFilter = 'Alle';

    google.script.run
      .withSuccessHandler(data => {
        capacityOverviewCache = data;
        renderCapacityOverview(data);
      })
      .withFailureHandler(err => {
        content.innerHTML = `
          <div class="bg-red-50 border border-red-200 text-red-700 rounded-xl p-6 text-center">
            <i class="fa-solid fa-triangle-exclamation text-3xl mb-3"></i>
            <h4 class="font-bold text-lg">Fehler beim Laden</h4>
            <p class="text-sm mt-2">${err.message || err}</p>
          </div>
        `;
      })
      .getCapacityOverviewData();
  }

  function closeCapacityOverviewModal() {
    const modal = document.getElementById('capacityOverviewModal');
    const modalContent = document.getElementById('capacityOverviewModalContent');

    if (!modal || !modalContent) return;

    modal.classList.add('opacity-0');
    modalContent.classList.remove('scale-100');
    modalContent.classList.add('scale-95');

    setTimeout(() => {
      modal.classList.add('hidden');
    }, 300);
  }

  function setCapacitySkillFilter(encodedSkill) {
    capacitySkillFilter = decodeURIComponent(encodedSkill || 'Alle');

    if (capacityOverviewCache) {
      renderCapacityOverview(capacityOverviewCache);
    }
  }

  function renderCapacitySkillFilters(skills) {
    const safeSkills = (skills || []).filter(s => s && String(s).trim() !== "" && String(s).toLowerCase() !== "ohne skill");

    let html = `
      <div class="bg-white border border-gray-100 rounded-xl p-3 shadow-sm">
        <div class="flex flex-wrap gap-2 items-center">
          <span class="text-[10px] uppercase font-bold text-gray-400 mr-1">
            <i class="fa-solid fa-filter text-lidl-blue mr-1"></i> Skill Filter:
          </span>
    `;

    const allClass = capacitySkillFilter === 'Alle'
      ? 'bg-lidl-blue text-white border-lidl-blue'
      : 'bg-white text-gray-500 border-gray-200 hover:bg-blue-50 hover:text-lidl-blue hover:border-blue-200';

    html += `
      <button onclick="setCapacitySkillFilter('${encodeURIComponent('Alle')}')" class="px-3 py-1 rounded-full border text-xs font-bold transition ${allClass}">
        Alle Skills
      </button>
    `;

    safeSkills.forEach(skill => {
      const active = capacitySkillFilter === skill;
      const cls = active
        ? 'bg-lidl-blue text-white border-lidl-blue'
        : 'bg-white text-gray-500 border-gray-200 hover:bg-blue-50 hover:text-lidl-blue hover:border-blue-200';

      html += `
        <button onclick="setCapacitySkillFilter('${encodeURIComponent(skill)}')" class="px-3 py-1 rounded-full border text-xs font-bold transition ${cls}">
          ${skill}
        </button>
      `;
    });

    html += `
        </div>
      </div>
    `;

    return html;
  }

  function renderCapacityOverview(data) {
    const content = document.getElementById('capacity-overview-content');

    if (!content) return;

    if (!data || !data.success || !data.areas || data.areas.length === 0) {
      content.innerHTML = `
        <div class="bg-white border border-gray-200 rounded-xl p-8 text-center text-gray-400">
          <i class="fa-regular fa-folder-open text-4xl mb-3"></i>
          <h4 class="font-bold text-slate-600 text-lg">Keine Kapazitäten gefunden</h4>
          <p class="text-sm mt-2">Es wurden keine verwertbaren Schichtdaten im aktuellen Plano Backup gefunden.</p>
        </div>
      `;
      return;
    }

    const skills = (data.foundSkills || data.areas.map(a => a.area))
      .filter(s => s && String(s).toLowerCase() !== "ohne skill");

    let areas = data.areas.filter(area => area.area && String(area.area).toLowerCase() !== "ohne skill");

    if (capacitySkillFilter !== 'Alle') {
      areas = areas.filter(area => area.area === capacitySkillFilter);
    }

    const totalHours = areas.reduce((sum, area) => sum + Number(area.totalHours || 0), 0);
    const remainingHours = areas.reduce((sum, area) => sum + Number(area.remainingHours || 0), 0);
    const usedHours = areas.reduce((sum, area) => sum + Number(area.usedHours || 0), 0);
    const employeeCount = areas.reduce((sum, area) => sum + Number(area.employeeCount || 0), 0);

    let html = `
      <div class="space-y-5">
        <div class="grid grid-cols-1 md:grid-cols-3 gap-4">
          <div class="bg-white border border-gray-100 rounded-xl p-4 shadow-sm">
            <div class="text-[10px] uppercase font-bold text-gray-400 mb-1">Gesamtkapazität Tag</div>
            <div class="text-3xl font-bold text-slate-700 font-mono">${decimalToTime(totalHours)}</div>
            <div class="text-xs text-gray-400 mt-1">${employeeCount || 0} Kollegen berücksichtigt</div>
          </div>

          <div class="bg-white border border-gray-100 rounded-xl p-4 shadow-sm">
            <div class="text-[10px] uppercase font-bold text-gray-400 mb-1">Bereits vergangen</div>
            <div class="text-3xl font-bold text-lidl-blue font-mono">${decimalToTime(usedHours)}</div>
            <div class="text-xs text-gray-400 mt-1">Seit Tagesbeginn bis jetzt</div>
          </div>

          <div class="bg-white border border-gray-100 rounded-xl p-4 shadow-sm border-l-4 border-modern-green">
            <div class="text-[10px] uppercase font-bold text-gray-400 mb-1">Restkapazität ab jetzt</div>
            <div class="text-3xl font-bold text-modern-green font-mono">${decimalToTime(remainingHours)}</div>
            <div class="text-xs text-gray-400 mt-1">Noch verfügbare Stunden im Tag</div>
          </div>
        </div>

        ${renderCapacitySkillFilters(skills)}

        <div class="bg-white border border-gray-100 rounded-xl p-4 shadow-sm">
          <div class="flex justify-between items-center mb-2">
            <div>
              <h4 class="font-bold text-slate-700 text-sm uppercase flex items-center gap-2">
                <i class="fa-solid fa-layer-group text-lidl-blue"></i> Kapazität nach Skill
              </h4>
              <p class="text-xs text-gray-400 mt-0.5">Anzeige der Tagesstunden und Reststunden aus dem aktuellen Plano Backup.</p>
            </div>
            <span class="text-[10px] uppercase font-bold text-gray-400">${data.sourceRows || 0} Schichtzeilen</span>
          </div>

          <div class="space-y-3 mt-4">
    `;

    if (areas.length === 0) {
      html += `
        <div class="border border-dashed border-gray-200 rounded-xl p-8 text-center text-gray-400">
          <i class="fa-solid fa-filter-circle-xmark text-3xl mb-3"></i>
          <div class="font-bold text-slate-500">Keine Daten für diesen Filter</div>
          <div class="text-xs mt-1">Wähle einen anderen Skill oder setze den Filter zurück.</div>
        </div>
      `;
    }

    areas.forEach(area => {
      const areaTotal = Number(area.totalHours || 0);
      const areaRest = Number(area.remainingHours || 0);
      const areaUsed = Number(area.usedHours || 0);

      html += `
        <div class="border border-gray-100 rounded-xl p-4 hover:shadow-sm transition bg-white">
          <div class="flex justify-between items-start gap-4">
            <div class="min-w-0">
              <h5 class="font-bold text-slate-700 flex items-center gap-2">
                <i class="fa-solid fa-circle text-[7px] text-lidl-blue"></i>
                ${area.area}
              </h5>
              <div class="text-[11px] text-gray-400 mt-1 truncate" title="${(area.originalSkills || []).join(', ')}">
                ${(area.originalSkills || []).join(' • ')}
              </div>
            </div>

            <div class="grid grid-cols-3 gap-8 text-right shrink-0">
              <div>
                <div class="text-[10px] uppercase font-bold text-gray-400">Gesamt</div>
                <div class="font-mono font-bold text-slate-700 text-lg">${decimalToTime(areaTotal)}</div>
              </div>
              <div>
                <div class="text-[10px] uppercase font-bold text-gray-400">Vergangen</div>
                <div class="font-mono font-bold text-lidl-blue text-lg">${decimalToTime(areaUsed)}</div>
              </div>
              <div>
                <div class="text-[10px] uppercase font-bold text-gray-400">Rest</div>
                <div class="font-mono font-bold text-modern-green text-lg">${decimalToTime(areaRest)}</div>
              </div>
            </div>
          </div>

          <div class="mt-3 grid grid-cols-2 gap-3">
            <div class="bg-gray-50 rounded-lg px-3 py-2 border border-gray-100">
              <div class="text-[10px] uppercase font-bold text-gray-400">Kollegen</div>
              <div class="text-sm font-bold text-slate-600">${area.employeeCount || 0}</div>
            </div>
            <div class="bg-emerald-50 rounded-lg px-3 py-2 border border-emerald-100">
              <div class="text-[10px] uppercase font-bold text-emerald-600">Rest je Kollege Ø</div>
              <div class="text-sm font-bold text-modern-green font-mono">${area.employeeCount ? decimalToTime(areaRest / area.employeeCount) : '00:00'}</div>
            </div>
          </div>
        </div>
      `;
    });

    html += `
          </div>
        </div>
      </div>
    `;

    content.innerHTML = html;
  }
