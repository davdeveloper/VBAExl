  <script>
    const SOURCE_ID = '<?= sourceId ?>';
    let isBusy = false;

    document.addEventListener('DOMContentLoaded', function () {
      reloadContent();
    });

    function reloadContent() {
      setBusy(true);
      showStatus('Inhalt wird geladen...', 'info');

      google.script.run
        .withSuccessHandler(function (result) {
          document.getElementById('title').value = result.title || '';
          document.getElementById('editor').innerHTML = result.contentHtml || '<p><br></p>';

          if (result.warnings && result.warnings.length) {
            showStatus(
              '<strong>Hinweis:</strong><br>' + result.warnings.map(escapeHtml).join('<br>'),
              'warning'
            );
          } else {
            hideStatus();
          }

          setBusy(false);
        })
        .withFailureHandler(function (err) {
          showStatus(getErrorMessage(err), 'error');
          setBusy(false);
        })
        .getImportForEditing(SOURCE_ID);
    }

    function saveContent() {
      const title = document.getElementById('title').value.trim();

      if (!title) {
        showStatus('Der Titel darf nicht leer sein.', 'error');
        return;
      }

      const editor = document.getElementById('editor');
      const blocks = editorDomToBlocks(editor);

      setBusy(true);
      showStatus('Änderungen werden gespeichert...', 'info');

      google.script.run
        .withSuccessHandler(function (result) {
          showStatus(result && result.message ? result.message : 'Änderungen wurden gespeichert.', 'success');
          setBusy(false);
        })
        .withFailureHandler(function (err) {
          showStatus(getErrorMessage(err), 'error');
          setBusy(false);
        })
        .saveEditedImport({
          sourceId: SOURCE_ID,
          title: title,
          blocks: blocks
        });
    }

    function closeEditor() {
      google.script.host.close();
    }

    function cmd(command) {
      focusEditor();
      document.execCommand(command, false, null);
    }

    function applyBlockFormat() {
      const value = document.getElementById('blockFormat').value;
      focusEditor();
      document.execCommand('formatBlock', false, value);
    }

    function createLink() {
      focusEditor();

      const url = prompt('Link eingeben:');

      if (!url) {
        return;
      }

      document.execCommand('createLink', false, url);
    }

    function removeFormatting() {
      focusEditor();
      document.execCommand('removeFormat', false, null);
    }

    function insertTable() {
      focusEditor();

      const html = `
        <table>
          <tbody>
            <tr>
              <td><p><br></p></td>
              <td><p><br></p></td>
            </tr>
            <tr>
              <td><p><br></p></td>
              <td><p><br></p></td>
            </tr>
          </tbody>
        </table>
        <p><br></p>
      `;

      document.execCommand('insertHTML', false, html);
    }

    function focusEditor() {
      document.getElementById('editor').focus();
    }

    function editorDomToBlocks(root) {
      const blocks = [];

      Array.from(root.childNodes).forEach(function (node) {
        collectBlockNode(node, blocks, 0);
      });

      return blocks.length ? blocks : [{
        type: 'paragraph',
        heading: 'NORMAL',
        segments: []
      }];
    }

    function collectBlockNode(node, blocks, level) {
      if (node.nodeType === Node.TEXT_NODE) {
        const text = node.nodeValue || '';

        if (text.trim()) {
          blocks.push({
            type: 'paragraph',
            heading: 'NORMAL',
            segments: [{ text: text }]
          });
        }

        return;
      }

      if (node.nodeType !== Node.ELEMENT_NODE) {
        return;
      }

      const tag = node.tagName.toUpperCase();

      if (tag === 'BR') {
        return;
      }

      if (tag === 'P' || tag === 'DIV' || /^H[1-6]$/.test(tag)) {
        blocks.push({
          type: 'paragraph',
          heading: htmlTagToHeading(tag),
          segments: inlineSegmentsFromNode(node)
        });
        return;
      }

      if (tag === 'UL' || tag === 'OL') {
        collectList(node, blocks, tag === 'OL', level);
        return;
      }

      if (tag === 'TABLE') {
        blocks.push(tableNodeToBlock(node));
        return;
      }

      if (tag === 'BLOCKQUOTE') {
        blocks.push({
          type: 'paragraph',
          heading: 'NORMAL',
          segments: inlineSegmentsFromNode(node)
        });
        return;
      }

      blocks.push({
        type: 'paragraph',
        heading: 'NORMAL',
        segments: inlineSegmentsFromNode(node)
      });
    }

    function collectList(listNode, blocks, ordered, level) {
      Array.from(listNode.children).forEach(function (child) {
        if (child.tagName.toUpperCase() !== 'LI') {
          return;
        }

        const storedLevel = child.getAttribute('data-level');
        const realLevel = storedLevel !== null && storedLevel !== ''
          ? Number(storedLevel)
          : level;

        blocks.push({
          type: 'listItem',
          ordered: ordered,
          level: isNaN(realLevel) ? level : realLevel,
          segments: inlineSegmentsFromNode(child, true)
        });

        Array.from(child.children).forEach(function (nested) {
          const nestedTag = nested.tagName.toUpperCase();

          if (nestedTag === 'UL' || nestedTag === 'OL') {
            collectList(nested, blocks, nestedTag === 'OL', level + 1);
          }
        });
      });
    }

    function tableNodeToBlock(tableNode) {
      const rows = [];

      Array.from(tableNode.rows).forEach(function (rowNode) {
        const row = {
          cells: []
        };

        Array.from(rowNode.cells).forEach(function (cellNode) {
          row.cells.push({
            blocks: editorDomToBlocks(cellNode)
          });
        });

        rows.push(row);
      });

      return {
        type: 'table',
        rows: rows
      };
    }

    function inlineSegmentsFromNode(root, ignoreNestedLists) {
      const segments = [];

      walkInline(root, {}, segments, ignoreNestedLists);
      return mergeSegments(segments);
    }

    function walkInline(node, inherited, segments, ignoreNestedLists) {
      if (node.nodeType === Node.TEXT_NODE) {
        const text = node.nodeValue || '';

        if (text) {
          segments.push(Object.assign({}, inherited, { text: text }));
        }

        return;
      }

      if (node.nodeType !== Node.ELEMENT_NODE) {
        return;
      }

      const tag = node.tagName.toUpperCase();

      if (tag === 'BR') {
        segments.push(Object.assign({}, inherited, { text: '\n' }));
        return;
      }

      if (ignoreNestedLists && (tag === 'UL' || tag === 'OL' || tag === 'TABLE')) {
        return;
      }

      const next = Object.assign({}, inherited);

      if (tag === 'B' || tag === 'STRONG') next.bold = true;
      if (tag === 'I' || tag === 'EM') next.italic = true;
      if (tag === 'U') next.underline = true;
      if (tag === 'S' || tag === 'STRIKE' || tag === 'DEL') next.strikethrough = true;

      if (tag === 'A') {
        const href = node.getAttribute('href');

        if (href) {
          next.href = href;
        }
      }

      const style = node.style || {};

      if (style.fontWeight && (style.fontWeight === 'bold' || Number(style.fontWeight) >= 600)) {
        next.bold = true;
      }

      if (style.fontStyle === 'italic') {
        next.italic = true;
      }

      if (style.textDecoration && style.textDecoration.indexOf('underline') !== -1) {
        next.underline = true;
      }

      if (style.textDecoration && style.textDecoration.indexOf('line-through') !== -1) {
        next.strikethrough = true;
      }

      if (style.color) {
        next.color = style.color;
      }

      if (style.backgroundColor) {
        next.backgroundColor = style.backgroundColor;
      }

      if (style.fontFamily) {
        next.fontFamily = style.fontFamily;
      }

      if (style.fontSize) {
        const size = parseFloat(style.fontSize);

        if (!isNaN(size)) {
          next.fontSize = size;
        }
      }

      Array.from(node.childNodes).forEach(function (child) {
        walkInline(child, next, segments, ignoreNestedLists);
      });
    }

    function mergeSegments(segments) {
      const result = [];

      segments.forEach(function (segment) {
        if (!segment.text) {
          return;
        }

        const last = result[result.length - 1];

        if (last && sameAttrs(last, segment)) {
          last.text += segment.text;
          return;
        }

        result.push(segment);
      });

      return result;
    }

    function sameAttrs(a, b) {
      return !!a.bold === !!b.bold &&
        !!a.italic === !!b.italic &&
        !!a.underline === !!b.underline &&
        !!a.strikethrough === !!b.strikethrough &&
        String(a.href || '') === String(b.href || '') &&
        String(a.color || '') === String(b.color || '') &&
        String(a.backgroundColor || '') === String(b.backgroundColor || '') &&
        String(a.fontFamily || '') === String(b.fontFamily || '') &&
        String(a.fontSize || '') === String(b.fontSize || '');
    }

    function htmlTagToHeading(tag) {
      if (tag === 'H1') return 'HEADING2';
      if (tag === 'H2') return 'HEADING2';
      if (tag === 'H3') return 'HEADING3';
      if (tag === 'H4') return 'HEADING4';
      if (tag === 'H5') return 'HEADING5';
      if (tag === 'H6') return 'HEADING6';

      return 'NORMAL';
    }

    function setBusy(value) {
      isBusy = value;

      Array.from(document.querySelectorAll('button, select, input')).forEach(function (element) {
        element.disabled = value;
      });

      document.getElementById('editor').setAttribute('contenteditable', value ? 'false' : 'true');
    }

    function showStatus(message, type) {
      const status = document.getElementById('status');
      status.className = 'status show ' + (type || 'info');
      status.innerHTML = message;
    }

    function hideStatus() {
      const status = document.getElementById('status');
      status.className = 'status';
      status.innerHTML = '';
    }

    function getErrorMessage(err) {
      if (!err) {
        return 'Unbekannter Fehler.';
      }

      return escapeHtml(err.message || String(err));
    }

    function escapeHtml(value) {
      return String(value || '')
        .replace(/&/g, '&amp;')
        .replace(/</g, '&lt;')
        .replace(/>/g, '&gt;')
        .replace(/"/g, '&quot;')
        .replace(/'/g, '&#039;');
    }
