Sub VergleicheNummern()
    Dim ws As Worksheet
    Dim lastRowB As Long, lastRowF As Long
    Dim rngB As Range, rngF As Range, cell As Range
    Dim missingNumbers As Range
    
    ' Arbeitsbl√§tter festlegen
    Set ws = ThisWorkbook.Sheets("DeinBlattName") ' Hier "DeinBlattName" durch den Namen deines Arbeitsblatts ersetzen
    
    ' Letzte Zeile in Spalte B und F finden
    lastRowB = ws.Cells(ws.Rows.Count, "B").End(xlUp).Row
    lastRowF = ws.Cells(ws.Rows.Count, "F").End(xlUp).Row
    
    ' Bereich in Spalte B und F festlegen
    Set rngB = ws.Range("B1:B" & lastRowB)
    Set rngF = ws.Range("F1:F" & lastRowF)
    
    ' Alle fehlenden Nummern in einem Bereich sammeln
    On Error Resume Next
    Set missingNumbers = Application.WorksheetFunction.Match(rngB, rngF, 0)
    On Error GoTo 0
    
    ' Alle fehlenden Nummern in Spalte H ausgeben
    If Not missingNumbers Is Nothing Then
        missingNumbers.EntireRow.Copy Destination:=ws.Cells(ws.Rows.Count, "H").End(xlUp).Offset(1, 0)
    End If
End Sub
