/**
 * Imports and processes data from 'Rohdaten Änderungswünsche Sort.' to 'Änderungswünsche Sortierung'.
 * Replicates VBA macro 'Daten_Aus_RohdatenÄnderungswünscheSort_Importieren'.
 * Key difference: De-duplicates based on BOTH Artikelnummer (Col F) AND Artikel (Col G).
 */
function importToChangeWishesSorting() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const ui = SpreadsheetApp.getUi();
  const sourceSheet = ss.getSheetByName("Rohdaten Änderungswünsche Sort.");
  const destSheet = ss.getSheetByName("Änderungswünsche Sortierung");

  Logger.log("--- Starting importToChangeWishesSorting ---");

  if (!sourceSheet || !destSheet) {
    ui.alert('Error', 'Could not find the source ("Rohdaten...") or destination ("Änderungswünsche Sortierung") sheet.', ui.ButtonSet.OK);
    Logger.log('Error: A required sheet is missing.');
    return;
  }

// Check source sheet using Column H (Artikel) as per VBA
  const columnHValues = sourceSheet.getRange("H1:H" + sourceSheet.getMaxRows()).getValues();
  let sourceLastRow = 0;
  for (let i = columnHValues.length - 1; i >= 0; i--) {
    if (String(columnHValues[i][0]).trim() !== "") {
      sourceLastRow = i + 1;
      break;
    }
  }

  if (sourceLastRow < 8) {
    ui.alert('No Data', 'The "Rohdaten Änderungswünsche Sort." sheet has no data to import (checked Col H).', ui.ButtonSet.OK);
    Logger.log('Warning: No data to process on the source sheet.');
    // Optional: Add redirect logic here if needed
    return;
  }
  Logger.log(`Found data up to row ${sourceLastRow} on source sheet.`);

  try {
    // --- 1. Clear Old Data Content ---
    const destLastRow = destSheet.getLastRow();
    if (destLastRow >= 6) {
      Logger.log(`Clearing old content from A6:R${destLastRow} in 'Änderungswünsche Sortierung'.`);
      destSheet.getRange("A6:R" + destLastRow).clearContent();
    }

    // --- 2. Read, Re-order, and De-duplicate Data in Memory ---
    Logger.log("Reading source data...");
    // Read all relevant columns (A to H) from the source sheet at once.
    const sourceRange = sourceSheet.getRange("A8:H" + sourceLastRow);
    const sourceValues = sourceRange.getValues();

    const uniqueKeys = new Set(); // Use a Set to track unique combinations
    const processedData = [];

    Logger.log("Processing and de-duplicating data based on Artikelnummer AND Artikel...");
    for (const row of sourceValues) {
      const artikelNummer = row[4]; // Column E (becomes Col F in dest)
      const artikel = row[7];       // Column H (becomes Col G in dest)
      const uniqueKey = `${artikelNummer}|||${artikel}`; // Combine both for uniqueness check
      
      // If we haven't seen this combination before, process it.
      if (!uniqueKeys.has(uniqueKey)) {
        processedData.push([
          row[0], // Column A -> A
          row[1], // Column B -> B
          row[2], // Column C -> C
          row[3], // Column D -> D
          row[5], // Column F -> E ('Artikelnummer QUELLE')
          artikelNummer, // Column E -> F ('Artikelnummer') - Part 1 of key
          artikel,       // Column H -> G ('Artikel') - Part 2 of key
        ]);
        uniqueKeys.add(uniqueKey);
      }
    }
    Logger.log(`Finished processing. Found ${processedData.length} unique rows.`);
    
    if (processedData.length === 0) {
        ui.alert('No Data', 'No unique data rows were found to import.', ui.ButtonSet.OK);
        return;
    }

    // --- 3. Write Processed Data to Destination ---
    Logger.log("Writing processed data to 'Änderungswünsche Sortierung' sheet (A:G).");
    destSheet.getRange(6, 1, processedData.length, 7).setValues(processedData);
    
    const newLastRow = 5 + processedData.length;

    // --- 4. Apply Formulas using a Format-Preserving Method ---
    Logger.log(`Applying formulas from H3:R3 to H6:R${newLastRow} using R1C1 notation.`);
    const formulaTemplateRange = destSheet.getRange("H3:R3");
    const templateFormulas = formulaTemplateRange.getFormulasR1C1()[0]; 

    if (templateFormulas.join('').length > 0) {
        const numRows = newLastRow - 5;
        const formulasForRange = [];
        for (let i = 0; i < numRows; i++) {
            formulasForRange.push(templateFormulas);
        }
        
        const formulaDestinationRange = destSheet.getRange("H6:R" + newLastRow);
        formulaDestinationRange.setFormulasR1C1(formulasForRange);
    } else {
        Logger.log("Warning: No formulas found in template range H3:R3.");
    }
    
    // --- 5. Wait for Calculations and Convert to Values ---
    SpreadsheetApp.flush();
    Logger.log("Waiting for formulas to calculate...");
    Utilities.sleep(5000);

    Logger.log(`Converting formulas in H6:R${newLastRow} to static values.`);
    const rangeToConvert = destSheet.getRange("H6:R" + newLastRow);
    rangeToConvert.copyTo(rangeToConvert, { contentsOnly: true });

    // --- 6. Sort the Data ---
    Logger.log("Sorting the data...");
    const fullDataRange = destSheet.getRange("A6:R" + newLastRow); 
    
    fullDataRange.sort({ column: 7, ascending: true }); // First sort: by Column G
    fullDataRange.sort([ // Second sort: multi-level
        { column: 1, ascending: true },
        { column: 2, ascending: true },
        { column: 3, ascending: true }
    ]);

    // --- 7. Apply Formatting (Mimicking VBA) ---
    // Although setFormulasR1C1 preserves format, the VBA explicitly reapplies it.
    Logger.log("Applying formatting from template row A6:R6 to all new rows.");
    const formatTemplateRange = destSheet.getRange("A6:R6");
    if (newLastRow > 6) {
        const formatDestination = destSheet.getRange("A7:R" + newLastRow);
        formatTemplateRange.copyTo(formatDestination, SpreadsheetApp.CopyPasteType.PASTE_FORMAT, false);
    }
    
    // --- 8. Cleanup: Lösche alles unterhalb der formatierten/sortierten Daten ---
    const maxRows = destSheet.getMaxRows();
    if (maxRows > newLastRow) {
        // Löscht alles ab der ersten leeren Zeile bis ganz unten. 
        // Spalten A bis R = 18 Spalten
        destSheet.getRange(newLastRow + 1, 1, maxRows - newLastRow, 18).clear();
        Logger.log(`Tabelle von Zeile ${newLastRow + 1} bis ${maxRows} komplett gesäubert.`);
    }

    destSheet.getRange("A1").activate(); // Activate A1 as per VBA
    if (typeof navigateOnSuccess === "function") {
      navigateOnSuccess();
    } else {
      ss.toast('Die Daten wurden erfolgreich importiert.', 'Erfolg!', 5);
    }
    Logger.log("--- Finished importToChangeWishesSorting ---");
    
    // Optional: Add redirect logic here if needed (e.g., activate another sheet)

  } catch (e) {
    const errorMessage = `An error occurred: ${e.message}\n\nStack:\n${e.stack}`;
    ui.alert('Processing Error', errorMessage, ui.ButtonSet.OK);
    Logger.log(`ERROR: ${e.stack}`);
  }
}
