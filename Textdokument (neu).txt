<script>
  let capacityOverviewCache = null;
  let capacitySkillFilter = 'Alle';

  function toggleControlPanel(panelId) {
    const body = document.getElementById(panelId + '-body');
    const icon = document.getElementById(panelId + '-icon');

    if (!body) return;

    body.classList.toggle('hidden');

    if (icon) {
      icon.classList.toggle('rotate-180');
    }
  }

  function openImportTypeModal() {
    const modal = document.getElementById('importTypeModal');

    modal.classList.remove('hidden');

    setTimeout(() => {
      modal.classList.remove('opacity-0');
      document.getElementById('importTypeModalContent').classList.remove('scale-95');
      document.getElementById('importTypeModalContent').classList.add('scale-100');
    }, 10);
  }

  function closeImportTypeModal() {
    const modal = document.getElementById('importTypeModal');

    modal.classList.add('opacity-0');
    document.getElementById('importTypeModalContent').classList.remove('scale-100');
    document.getElementById('importTypeModalContent').classList.add('scale-95');

    setTimeout(() => {
      modal.classList.add('hidden');
    }, 300);
  }
