function toggleControlPanel(panelId) {
  const body = document.getElementById(panelId + '-body');
  const icon = document.getElementById(panelId + '-icon');

  if (!body) return;

  body.classList.toggle('hidden');

  if (icon) {
    icon.classList.toggle('rotate-180');
  }
}

function openCapacityOverviewModal(forceRefresh) {
  const modal = document.getElementById('capacityOverviewModal');
  const content = document.getElementById('capacity-overview-content');

  if (!modal || !content) return;

  modal.classList.remove('hidden');

  setTimeout(() => {
    modal.classList.remove('opacity-0');
    document.getElementById('capacityOverviewModalContent').classList.remove('scale-95');
    document.getElementById('capacityOverviewModalContent').classList.add('scale-100');
  }, 10);

  content.innerHTML = `
    <div class="flex flex-col items-center justify-center py-16 text-gray-400">
      <i class="fa-solid fa-circle-notch fa-spin text-4xl mb-3 text-lidl-blue"></i>
      <p class="font-medium">Kapazitäten werden berechnet...</p>
      <p class="text-xs mt-1">Basis: aktuelles Schichtmodell aus dem Plano Backup</p>
    </div>
  `;

  google.script.run
    .withSuccessHandler(renderCapacityOverview)
    .withFailureHandler(err => {
      content.innerHTML = `
        <div class="bg-red-50 border border-red-200 text-red-700 rounded-xl p-6 text-center">
          <i class="fa-solid fa-triangle-exclamation text-3xl mb-3"></i>
          <h4 class="font-bold text-lg">Fehler beim Laden</h4>
          <p class="text-sm mt-2">${err.message || err}</p>
        </div>
      `;
    })
    .getCapacityOverviewData();
}

function closeCapacityOverviewModal() {
  const modal = document.getElementById('capacityOverviewModal');
  const modalContent = document.getElementById('capacityOverviewModalContent');

  if (!modal || !modalContent) return;

  modal.classList.add('opacity-0');
  modalContent.classList.remove('scale-100');
  modalContent.classList.add('scale-95');

  setTimeout(() => {
    modal.classList.add('hidden');
  }, 300);
}

function renderCapacityOverview(data) {
  const content = document.getElementById('capacity-overview-content');

  if (!content) return;

  if (!data || !data.success || !data.areas || data.areas.length === 0) {
    content.innerHTML = `
      <div class="bg-white border border-gray-200 rounded-xl p-8 text-center text-gray-400">
        <i class="fa-regular fa-folder-open text-4xl mb-3"></i>
        <h4 class="font-bold text-slate-600 text-lg">Keine Kapazitäten gefunden</h4>
        <p class="text-sm mt-2">Es wurden keine verwertbaren Schichtdaten im aktuellen Plano Backup gefunden.</p>
      </div>
    `;
    return;
  }

  const totalHours = Number(data.totalHours || 0);
  const remainingHours = Number(data.remainingHours || 0);
  const usedHours = Number(data.usedHours || 0);

  const restPercent = totalHours > 0 ? Math.round((remainingHours / totalHours) * 100) : 0;
  const usedPercent = totalHours > 0 ? Math.round((usedHours / totalHours) * 100) : 0;

  let html = `
    <div class="space-y-5">
      <div class="grid grid-cols-1 md:grid-cols-4 gap-4">
        <div class="bg-white border border-gray-100 rounded-xl p-4 shadow-sm">
          <div class="text-[10px] uppercase font-bold text-gray-400 mb-1">Gesamtkapazität Tag</div>
          <div class="text-2xl font-bold text-slate-700 font-mono">${decimalToTime(totalHours)}</div>
          <div class="text-xs text-gray-400 mt-1">${data.employeeCount || 0} Kollegen berücksichtigt</div>
        </div>

        <div class="bg-white border border-gray-100 rounded-xl p-4 shadow-sm">
          <div class="text-[10px] uppercase font-bold text-gray-400 mb-1">Bereits verplant/vergangen</div>
          <div class="text-2xl font-bold text-lidl-blue font-mono">${decimalToTime(usedHours)}</div>
          <div class="text-xs text-gray-400 mt-1">${usedPercent}% des Tages</div>
        </div>

        <div class="bg-white border border-gray-100 rounded-xl p-4 shadow-sm border-l-4 border-modern-green">
          <div class="text-[10px] uppercase font-bold text-gray-400 mb-1">Restkapazität ab jetzt</div>
          <div class="text-2xl font-bold text-modern-green font-mono">${decimalToTime(remainingHours)}</div>
          <div class="text-xs text-gray-400 mt-1">${restPercent}% Rest vom Tag</div>
        </div>

        <div class="bg-white border border-gray-100 rounded-xl p-4 shadow-sm">
          <div class="text-[10px] uppercase font-bold text-gray-400 mb-1">Aktiv im Dienst</div>
          <div class="text-2xl font-bold text-orange-500 font-mono">${data.activeNowCount || 0}</div>
          <div class="text-xs text-gray-400 mt-1">Stand: ${data.updatedAt || '-'}</div>
        </div>
      </div>

      <div class="bg-white border border-gray-100 rounded-xl p-4 shadow-sm">
        <div class="flex justify-between items-center mb-2">
          <div>
            <h4 class="font-bold text-slate-700 text-sm uppercase flex items-center gap-2">
              <i class="fa-solid fa-layer-group text-lidl-blue"></i> Kapazität nach Steuerungsbereich
            </h4>
            <p class="text-xs text-gray-400 mt-0.5">Skills werden automatisch zu den definierten Steuerungsbereichen zusammengeführt.</p>
          </div>
          <span class="text-[10px] uppercase font-bold text-gray-400">${data.sourceRows || 0} Schichtzeilen</span>
        </div>

        <div class="space-y-3 mt-4">
  `;

  data.areas.forEach(area => {
    const areaTotal = Number(area.totalHours || 0);
    const areaRest = Number(area.remainingHours || 0);
    const areaUsed = Number(area.usedHours || 0);

    const areaRestPercent = areaTotal > 0 ? Math.round((areaRest / areaTotal) * 100) : 0;
    const areaUsedPercent = Math.max(0, Math.min(100, 100 - areaRestPercent));

    let statusClass = "bg-emerald-50 text-modern-green border-emerald-100";
    let statusText = "Rest verfügbar";

    if (areaRest <= 0) {
      statusClass = "bg-gray-100 text-gray-500 border-gray-200";
      statusText = "abgeschlossen";
    } else if (areaRestPercent < 25) {
      statusClass = "bg-orange-50 text-orange-600 border-orange-100";
      statusText = "knapp";
    }

    html += `
      <div class="border border-gray-100 rounded-xl p-4 hover:shadow-sm transition bg-white">
        <div class="flex justify-between items-start gap-4">
          <div class="min-w-0">
            <div class="flex items-center gap-2">
              <h5 class="font-bold text-slate-700">${area.area}</h5>
              <span class="text-[10px] font-bold uppercase px-2 py-0.5 rounded-full border ${statusClass}">${statusText}</span>
            </div>
            <div class="text-[11px] text-gray-400 mt-1 truncate" title="${(area.originalSkills || []).join(', ')}">
              ${(area.originalSkills || []).join(' • ')}
            </div>
          </div>

          <div class="grid grid-cols-3 gap-6 text-right shrink-0">
            <div>
              <div class="text-[10px] uppercase font-bold text-gray-400">Gesamt</div>
              <div class="font-mono font-bold text-slate-700">${decimalToTime(areaTotal)}</div>
            </div>
            <div>
              <div class="text-[10px] uppercase font-bold text-gray-400">Rest</div>
              <div class="font-mono font-bold text-modern-green">${decimalToTime(areaRest)}</div>
            </div>
            <div>
              <div class="text-[10px] uppercase font-bold text-gray-400">Aktiv</div>
              <div class="font-mono font-bold text-lidl-blue">${area.activeNowCount || 0}/${area.employeeCount || 0}</div>
            </div>
          </div>
        </div>

        <div class="mt-3">
          <div class="flex justify-between text-[10px] font-bold uppercase text-gray-400 mb-1">
            <span>Verbrauchter Tag: ${areaUsedPercent}%</span>
            <span>Rest: ${areaRestPercent}%</span>
          </div>
          <div class="w-full bg-gray-100 rounded-full h-2 overflow-hidden">
            <div class="bg-lidl-blue h-2 rounded-full transition-all" style="width:${areaUsedPercent}%"></div>
          </div>
        </div>
      </div>
    `;
  });

  html += `
        </div>
      </div>
    </div>
  `;

  content.innerHTML = html;
}
