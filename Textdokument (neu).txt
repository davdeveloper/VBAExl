/**
 * Importiert Telefoniedaten (Analog zu VBA Sub DatenImportieren2)
 */
function importTelephonyData(dateString) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  
  // 1. Datum berechnen (Eingabe - 1 Tag, wie im VBA Dat = Dat - 1)
  const inputDate = new Date(dateString);
  const targetDate = new Date(inputDate);
  targetDate.setDate(inputDate.getDate() - 1); 

  // Formatierungen für Suche und Anzeige
  const day = targetDate.getDate();
  const month = targetDate.getMonth() + 1;
  const year = targetDate.getFullYear();
  
  const dayStr = day < 10 ? '0' + day : '' + day;
  const monthStr = month < 10 ? '0' + month : '' + month;
  
  // Dateiname Telekom: dd.mm.yyyy (z.B. 26.11.2025)
  const dateDotFormat = `${dayStr}.${monthStr}.${year}`; 
  
  // Dateiname Report ist JAHRMONAT_Steuerungsreport_Salesforce (z.B. 202511_...)
  const dateReportPattern = `${year}${monthStr}_Steuerungsreport_Salesforce`;

  // 2. Duplikatsprüfung im Blatt "DTAG" & in "Datenbank" vorbereiten
  let wsDTAG = ss.getSheetByName("DTAG");
  if (!wsDTAG) {
    throw new Error("Blatt 'DTAG' wurde nicht gefunden.");
  }

  let isTelekomImported = false;
  const lastRowDTAG = wsDTAG.getLastRow();
  if (lastRowDTAG >= 3) {
    // Hole alle vorhandenen Daten in Spalte C (Index 3), starte ab Zeile 3
    const existingDates = wsDTAG.getRange(3, 3, lastRowDTAG - 2, 1).getValues().flat();
    for (let d of existingDates) {
      if (d instanceof Date && isSameDay(d, targetDate)) {
        isTelekomImported = true; // Telekom Daten existieren bereits
        break;
      }
    }
  }

  const wsDatenbank = ss.getSheetByName("Datenbank");
  if (!wsDatenbank) {
    throw new Error("Blatt 'Datenbank' wurde nicht gefunden.");
  }

  // Zeile des Datums in der Datenbank ermitteln
  const dbRowIndex = findRowByDate(wsDatenbank, targetDate);
  
  // PRÜFUNG KORRIGIERT: Prüfe jetzt den komplett durchgehenden Block EA bis ET (Spalten 131 bis 150)
  const checkValues = wsDatenbank.getRange(dbRowIndex, 131, 1, 20).getValues()[0];
  const isDatabaseFilled = checkValues.some(val => val !== "" && val !== null);

  // Spezialfall: Wenn bereits BEIDE Datentöpfe befüllt sind, direkt abbrechen
  if (isTelekomImported && isDatabaseFilled) {
    return `Hinweis: Daten für ${dateDotFormat} sind bereits vollständig importiert (DTAG & Datenbank befüllt).`;
  }

  // 3. Dateien suchen (nur wenn der jeweilige Import noch aussteht)
  
  // A) Telekom Datei suchen (nur wenn noch nicht in DTAG vorhanden)
  let telekomFile = null;
  if (!isTelekomImported) {
    const telekomFolder = DriveApp.getFolderById(FOLDER_ID_TELEKOM);
    const telekomFiles = telekomFolder.searchFiles(`title contains '${dateDotFormat}' and trashed = false`);
    if (telekomFiles.hasNext()) {
      telekomFile = telekomFiles.next();
    } else {
      throw new Error(`Telekom-Datei für ${dateDotFormat} nicht gefunden.`);
    }
  }

  // B) Steuerungsreport suchen (nur wenn Datenbank-Spalten noch leer sind)
  let steuerungFile = null;
  if (!isDatabaseFilled) {
    const steuerungFolder = DriveApp.getFolderById(FOLDER_ID_STEUERUNG);
    const steuerungFiles = steuerungFolder.searchFiles(`title contains '${dateReportPattern}' and trashed = false`);
    if (steuerungFiles.hasNext()) {
      steuerungFile = steuerungFiles.next();
    } else {
      throw new Error(`Steuerungsreport nicht gefunden: ${dateReportPattern}`);
    }
  }

  // 4. Import durchführen (Telekom -> DTAG)
  if (!isTelekomImported) {
    // Excel temporär konvertieren
    const tempId = convertExcelToGoogleSheet(telekomFile);
    const tempSS = SpreadsheetApp.openById(tempId);
    const wsStatistik = tempSS.getSheetByName("Statistik");

    if (!wsStatistik) {
      DriveApp.getFileById(tempId).setTrashed(true);
      throw new Error("Blatt 'Statistik' in der Telekom-Datei nicht gefunden.");
    }

    const lastRowSource = wsStatistik.getLastRow();
    const lastColSource = wsStatistik.getLastColumn();

    if (lastRowSource < 18) {
      DriveApp.getFileById(tempId).setTrashed(true);
      throw new Error("Keine Daten im Blatt 'Statistik' ab Zeile 18 gefunden.");
    }

    const dataValues = wsStatistik.getRange(18, 1, lastRowSource - 17, lastColSource).getValues();
    const startRowDest = wsDTAG.getLastRow() + 1;

    // Werte einfügen ab Spalte D (Index 4)
    wsDTAG.getRange(startRowDest, 4, dataValues.length, dataValues[0].length).setValues(dataValues);

    // 5. Nacharbeiten (Datum und Formeln)
    wsDTAG.getRange(startRowDest, 3, dataValues.length, 1).setValue(dateDotFormat).setNumberFormat("dd.MM.yyyy");

    if (startRowDest > 3) {
      const formelA = wsDTAG.getRange("A3").getFormulaR1C1();
      const formelB = wsDTAG.getRange("B3").getFormulaR1C1();
      
      const rangeA = wsDTAG.getRange(startRowDest, 1, dataValues.length, 1);
      if (formelA) rangeA.setFormulaR1C1(formelA);
      
      const rangeB = wsDTAG.getRange(startRowDest, 2, dataValues.length, 1);
      if (formelB) rangeB.setFormulaR1C1(formelB);
      
      SpreadsheetApp.flush(); 
      rangeA.setValues(rangeA.getValues());
      rangeB.setValues(rangeB.getValues());
    }

    // Temp Datei löschen
    DriveApp.getFileById(tempId).setTrashed(true);

    // 6. Abschluss für Tagesplanung
    const wsTagesplanung = ss.getSheetByName("Tagesplanung");
    if (wsTagesplanung) {
      wsTagesplanung.getRange("H6").setValue(dateDotFormat).setNumberFormat("dd.MM.yyyy");
    }
  }

  // 7. Steuerungsreport auslesen und in Datenbank schreiben (nur wenn Zielspalten leer sind)
  if (!isDatabaseFilled) {
    const tempReportId = convertExcelToGoogleSheet(steuerungFile);
    
    try {
      const tempReportSS = SpreadsheetApp.openById(tempReportId);
      const wsLKS1 = tempReportSS.getSheetByName("LKS-1st");
      const wsLKS2 = tempReportSS.getSheetByName("LKS-2nd");
      
      if (!wsLKS1 || !wsLKS2) {
        throw new Error("Blatt 'LKS-1st' oder 'LKS-2nd' im Steuerungsreport nicht gefunden.");
      }
      
      let row1st = -1;
      const data1st = wsLKS1.getRange("A1:A100").getValues();
      for (let i = 0; i < data1st.length; i++) {
        if (data1st[i][0] instanceof Date && isSameDay(data1st[i][0], targetDate)) {
          row1st = i + 1; 
          break;
        }
      }
      
      let row2nd = -1;
      const data2nd = wsLKS2.getRange("A1:A100").getValues();
      for (let i = 0; i < data2nd.length; i++) {
        if (data2nd[i][0] instanceof Date && isSameDay(data2nd[i][0], targetDate)) {
          row2nd = i + 1; 
          break;
        }
      }
      
      if (row1st === -1) throw new Error(`Datum ${dateDotFormat} in LKS-1st nicht gefunden.`);
      if (row2nd === -1) throw new Error(`Datum ${dateDotFormat} in LKS-2nd nicht gefunden.`);
      
      const rowData1st = wsLKS1.getRange(row1st, 1, 1, 100).getValues()[0];
      const rowData2nd = wsLKS2.getRange(row2nd, 1, 1, 50).getValues()[0];
      
      // ALLES IN EINEM BATCH: 20 Werte am Stück für EA(131) bis ET(150)
      const valuesToWrite = [
        [
          // --- LKS-1st Werte ---
          rowData1st[35], rowData1st[37], rowData1st[40], rowData1st[41], // EA-ED (AJ, AL, AO, AP)
          rowData1st[43], rowData1st[45], rowData1st[48], rowData1st[49], // EE-EH (AR, AT, AW, AX)
          rowData1st[51], rowData1st[53], rowData1st[56], rowData1st[57], // EI-EL (AZ, BB, BE, BF)
          rowData1st[67], rowData1st[69], rowData1st[72], rowData1st[73], // EM-EP (BP, BR, BU, BV)
          
          // --- LKS-2nd Werte ---
          rowData2nd[24], rowData2nd[26], rowData2nd[29], rowData2nd[30]  // EQ-ET (Y, AA, AD, AE)
        ]
      ];
      
      // Schreibe alle 20 Spalten in einem einzigen Befehl
      wsDatenbank.getRange(dbRowIndex, 131, 1, 20).setValues(valuesToWrite);
      
    } finally {
      DriveApp.getFileById(tempReportId).setTrashed(true);
    }
  }

  // Dynamischer Rückgabetext je nachdem, welcher Schritt ausgeführt wurde
  if (isTelekomImported && !isDatabaseFilled) {
    return `Telekom-Daten waren bereits vorhanden. Steuerungsreport (LKS) erfolgreich für ${dateDotFormat} nachgetragen.`;
  } else if (!isTelekomImported && isDatabaseFilled) {
    return `Telefoniedaten (Telekom) erfolgreich importiert. LKS-Daten in Datenbank waren bereits befüllt.`;
  } else {
    return `Import erfolgreich abgeschlossen für ${dateDotFormat} (Telekom- & Steuerungsreport-Daten eingelesen).`;
  }
}

function runOutgoingsCalculate() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const wsMapping = ss.getSheetByName("Mapping");
  const wsDBKontakte = ss.getSheetByName("Datenbank");

  const today = new Date();
  const targetDate = new Date(2025, 4, 2); 
  if (today < targetDate) {
    // return "Abbruch"; 
  }

  const datumVal = wsMapping.getRange("C3").getValue();
  if (!(datumVal instanceof Date)) {
    throw new Error("Das Datum im Mapping (C3) ist ungültig.");
  }

  const searchDate = new Date(datumVal);
  searchDate.setDate(searchDate.getDate() - 1); 
  
  const year = searchDate.getFullYear();
  const month = searchDate.getMonth() + 1;
  const monthStr = month < 10 ? '0' + month : '' + month;
  
  const searchPattern = `${year}${monthStr}_Steuerungsreport_Salesforce`;
  
  Logger.log("Suche nach Datei: " + searchPattern);

  const folder = DriveApp.getFolderById(FOLDER_ID_STEUERUNG);
  const files = folder.searchFiles(`title contains '${searchPattern}' and trashed = false`);
  
  let reportFile;
  if (files.hasNext()) {
    reportFile = files.next();
  } else {
    throw new Error(`Datei nicht gefunden: ${searchPattern} (Gesucht für Datum: ${Utilities.formatDate(searchDate, Session.getScriptTimeZone(), "dd.MM.yyyy")})`);
  }

  const dbRowIndex = findRowByDate(wsDBKontakte, searchDate);

  const tempSheetId = convertExcelToGoogleSheet(reportFile);
  const tempSS = SpreadsheetApp.openById(tempSheetId);
  const wsFC_DL = tempSS.getSheetByName("FC_DL");

  if (!wsFC_DL) {
    DriveApp.getFileById(tempSheetId).setTrashed(true);
    throw new Error("Blatt 'FC_DL' im Steuerungsreport nicht gefunden.");
  }

  const rangeBP = wsFC_DL.getRange("BP10:BP40").getValues();
  let foundInReportRow = -1;

  for (let i = 0; i < rangeBP.length; i++) {
    if (rangeBP[i][0] instanceof Date && isSameDay(rangeBP[i][0], searchDate)) {
      foundInReportRow = 10 + i;

  const valCA = sourceValues[78];
  const valCI = sourceValues[86];
  const valCN = sourceValues[91];
  const valCO = sourceValues[92];
  const valCP = sourceValues[93];

  DriveApp.getFileById(tempSheetId).setTrashed(true);

  wsDBKontakte.getRange(dbRowIndex, 121).setValue(valBW); // DQ
  wsDBKontakte.getRange(dbRowIndex, 123).setValue(valCA); // DS
  wsDBKontakte.getRange(dbRowIndex, 125).setValue(valCI); // DU

// --- HELPER ---

function findRowByDate(sheet, dateObj) {
  const data = sheet.getRange("A:A").getValues();
  for (let i = 0; i < data.length; i++) {
    if (data[i][0] instanceof Date && isSameDay(data[i][0], dateObj)) {
      return i + 1;
    }
  }
  throw new Error("Datum " + Utilities.formatDate(dateObj, Session.getScriptTimeZone(), "dd.MM.yyyy") + " in Datenbank Spalte A nicht gefunden.");
}

function isSameDay(d1, d2) {
  return d1.getFullYear() === d2.getFullYear() &&
         d1.getMonth() === d2.getMonth() &&
         d1.getDate() === d2.getDate();
}

function convertExcelToGoogleSheet(file) {
  const resource = {
    name: "Temp_Report_Convert_" + new Date().getTime(),
    mimeType: MimeType.GOOGLE_SHEETS,
    parents: [DriveApp.getRootFolder().getId()] 
  };
  
  const optionalArgs = {
    supportsAllDrives: true
  };
  
  const driveFile = Drive.Files.copy(resource, file.getId(), optionalArgs);
  return driveFile.id;
}

function parseAndFill(formObject, config, levelName) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var rawText = formObject.pastedText;
  var refDateString = formObject.refDate;
  
  var referenceDate = new Date(refDateString);
  if (isNaN(referenceDate.getTime())) {
    return "Fehler: Ungültiges Datum gewählt.";
  }

  var lines = rawText.split(/\r?\n/);
  var currentLabel = null;
  var processedCount = 0;
  var foundConfigKeys = {};

  for (var i = 0; i < lines.length; i++) {
    var line = lines[i].trim();
    if (line === "") continue;

    if (line.startsWith("DEU_")) {
      currentLabel = line;
    } else if (currentLabel && /^\d+/.test(line)) {
      var regex = /^(\d+).*?(\d+)d,\s*(\d+)h\s*(\d+)\s*min/;
      var match = line.match(regex);

      if (match) {
        var count = match[1];
        var days = parseInt(match[2], 10);
        var hours = parseInt(match[3], 10);
        var mins = parseInt(match[4], 10);

        var configKey = currentLabel;
        var setting = config[configKey];
        
        if (!setting) {
           var foundKey = Object.keys(config).find(k => k.toLowerCase() === currentLabel.toLowerCase());
           if (foundKey) {
             configKey = foundKey;
             setting = config[configKey];
           }
        }

        if (setting) {
          foundConfigKeys[configKey] = true;
          if (setting.countCell) {
            sheet.getRange(setting.countCell).setValue(count);
          }
          if (setting.dateCell) {
            var durationMs = (days * 24 * 60 * 60 * 1000) + (hours * 60 * 60 * 1000) + (mins * 60 * 1000);
            var calcDate = new Date(referenceDate.getTime() - durationMs);
            sheet.getRange(setting.dateCell).setValue(calcDate).setNumberFormat("dd.MM.yyyy");
          }
          processedCount++;
        }
      }
      currentLabel = null;
    }
  }

  var allKeys = Object.keys(config);
  for (var k = 0; k < allKeys.length; k++) {
    var key = allKeys[k];
    if (!foundConfigKeys[key]) {
      var setting = config[key];
      if (setting.countCell) {
        sheet.getRange(setting.countCell).setValue(0);
      }
      if (setting.dateCell) {
        sheet.getRange(setting.dateCell).setValue(referenceDate).setNumberFormat("dd.MM.yyyy");
      }
    }
  }

  SpreadsheetApp.flush();
  SpreadsheetApp.getUi().alert("Bitte die Briefversand Vorgänge aus Gmail manuell erfassen.");
  return levelName + ": Fertig. " + processedCount + " Werte aus Text übernommen. Rest auf 0 gesetzt.";
}
