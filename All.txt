      <input type="checkbox" id="fullMonthCheckbox">

    <button id="importBtn" onclick="startImport()" disabled>Auswerten & Importieren</button>

    <div id="status"></div>

    <script>
      google.script.run
        .withSuccessHandler(function(files) {
          const select = document.getElementById('fileSelect');
          select.innerHTML = '';

          if (files.length === 0) {
            select.innerHTML = '<option value="">Keine Dateien im Ordner gefunden</option>';
            return;
          }

          files.forEach(function(f) {
            const option = document.createElement('option');
            option.value = f.id;
            option.text = f.name;
            select.appendChild(option);
          });

          document.getElementById('importBtn').disabled = false;
        })
        .withFailureHandler(function(err) {
          showStatus("Fehler beim Laden der Dateien: " + err.message, "error");
        })
        .getFilesInFolder();

      function startImport() {
        const select = document.getElementById('fileSelect');
        const fileId = select.value;
        const importFullMonth = document.getElementById('fullMonthCheckbox').checked;

        if (!fileId) return;

        document.getElementById('importBtn').disabled = true;
        select.disabled = true;
        document.getElementById('fullMonthCheckbox').disabled = true;

        if (importFullMonth) {
          showStatus("Monatsimport läuft. Die Datei wird blockweise ausgewertet und in die Testdatenbank geschrieben.", "loading");
        } else {
          showStatus("Datei wird eingelesen und ausgewertet. Bitte warten.", "loading");
        }

        google.script.run
          .withSuccessHandler(function(res) {
            showStatus("Erfolgreich! " + res, "success");

            setTimeout(function() {
              google.script.host.close();
            }, 5000);
          })
          .withFailureHandler(function(err) {
            showStatus("Fehler: " + err.message, "error");
            document.getElementById('importBtn').disabled = false;
            select.disabled = false;
            document.getElementById('fullMonthCheckbox').disabled = false;
          })
          .processPlanoImport(fileId, importFullMonth);
      }

      function showStatus(msg, type) {
        const el = document.getElementById('status');
        el.style.display = 'block';
        el.className = type;
        el.innerText = msg;
      }
    </script>
