    <h3>Plano-Daten importieren</h3>
    <p>Waehle eine Datei aus der geteilten Ablage:</p>

    <select id="fileSelect">
      <option value="">Lade Dateien...</option>
    </select>

    <label class="check-row">
      <input id="wholeMonth" type="checkbox">
      <span>gesamten Monat importieren</span>
    </label>

    <button id="importBtn" onclick="startImport()" disabled>Auswerten & Importieren</button>
    <div id="status"></div>

    <script>
      google.script.run.withSuccessHandler(function(files) {
        const select = document.getElementById('fileSelect');
        select.innerHTML = '';
        if (files.length === 0) {
          select.innerHTML = '<option value="">Keine Dateien im Ordner gefunden</option>';
          return;
        }
        files.forEach(function(f) {
          const option = document.createElement('option');
          option.value = f.id;
          option.text = f.name;
          select.appendChild(option);
        });
        document.getElementById('importBtn').disabled = false;
      }).withFailureHandler(function(err) {
        showStatus("Fehler beim Laden der Dateien: " + err.message, "error");
      }).getFilesInFolder();

      function startImport() {
        const select = document.getElementById('fileSelect');
        const fileId = select.value;
        const importWholeMonth = document.getElementById('wholeMonth').checked;
        if (!fileId) return;

        document.getElementById('importBtn').disabled = true;
        select.disabled = true;
        document.getElementById('wholeMonth').disabled = true;

        const modeText = importWholeMonth ? "Monat wird" : "Datei wird";
        showStatus(modeText + " eingelesen und ausgewertet... Bitte warten.", "loading");

        google.script.run.withSuccessHandler(function(res) {
          showStatus("Erfolgreich! " + res, "success");
          setTimeout(() => google.script.host.close(), 4000);
        }).withFailureHandler(function(err) {
          showStatus("Fehler: " + err.message, "error");
          document.getElementById('importBtn').disabled = false;
          select.disabled = false;
          document.getElementById('wholeMonth').disabled = false;
        }).processSelectedFile(fileId, importWholeMonth);
      }

      function showStatus(msg, type) {
        const el = document.getElementById('status');
        el.style.display = 'block';
        el.className = type;
        el.innerText = msg;
      }
    </script>
